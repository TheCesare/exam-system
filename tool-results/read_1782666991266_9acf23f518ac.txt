    43→function parseTimeRange(tStr: string) {
    44→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    45→  try {
    46→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    47→    const parts = clean.split('-');
    48→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    49→    let start = toMinutes(parts[0]);
    50→    let end = toMinutes(parts[1]);
    51→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    52→    if (start < 300 && end > start + 120) start += 720;
    53→    if (end <= start) end += 720;
    54→    let duration = (end - start) / 60;
    55→    // Sanity cap: no exam is longer than 5 hours
    56→    if (duration > 5) duration = 2.0; // fallback to standard
    57→    return { start, end, duration };
    58→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    59→}

    61→function getStage(grade: string) {
    62→  const g = grade.toLowerCase();
    63→  if (g.includes('primary')) return 'primary';
    64→  if (g.includes('prep')) return 'prep';
    65→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    66→  return 'any';
    67→}

   495→    // ---- Initialize tracking ----
   496→    const tracking: Record<string, TrackingEntry> = {};
   497→    teachers.forEach(t => {
   498→      tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {} as Record<string,number>, assignedSlots: [], gradeHistory: [] };
   499→      DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   500→    });
   501→
   502→    // ---- Build exam slots grouped by day (chronological order) ----
   503→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   504→    const slotsByDay: Record<string, Slot[]> = {};
   505→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   506→    DAYS.forEach((day, dayIndex) => {
   507→      GRADES.forEach(grade => {
   508→        const cell = scheduleBuffer[`${grade}__${day}`];
   509→        if (cell && cell.committees > 0) {
   510→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   511→          for (let c = 1; c <= cell.committees; c++) {
   512→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   513→          }
   514→        }
   515→      });
   516→    });
   517→
   518→    // ---- Build final assignments structure ----
   519→    const finalAssignments: Record<string, SessionResult[]> = {};
   520→    DAYS.forEach(d => { finalAssignments[d] = []; });
   521→    DAYS.forEach(day => {
   522→      GRADES.forEach(grade => {
   523→        const cell = scheduleBuffer[`${grade}__${day}`];
   524→        if (cell && cell.committees > 0) {
   525→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   526→          if (!existing) {
   527→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   528→          }
   529→        }
   530→      });
   531→    });
   532→
   533→    // ---- Helper: Stage notes — HARD constraint ----
   534→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   535→      if (!ruleNotes) return true;
   536→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   537→      const n = teacher.notes.toLowerCase();
   538→      const allowed: string[] = [];
   539→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   540→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   541→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   542→      if (allowed.length === 0) return true;
   543→      return allowed.includes(slotStage);
   544→    };
   545→
   546→    // ---- Helper: Consecutive-day same-grade check ----
   547→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   548→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   549→    };
   550→
   551→    // ---- Helper: Fisher-Yates shuffle ----
   552→    const shuffle = <T,>(arr: T[]): T[] => {
   553→      const a = [...arr];
   554→      for (let i = a.length - 1; i > 0; i--) {
   555→        const j = Math.floor(Math.random() * (i + 1));
   556→        [a[i], a[j]] = [a[j], a[i]];
   557→      }
   558→      return a;
   559→    };
   560→
   561→    // ---- Admin minimum assignment tracking ----
   562→    const adminAssignmentCount: Record<string, number> = {};
   563→    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   564→    const adminMinTarget = 2; // each admin gets at least 2 assignments
   565→    let totalAdminAssigned = 0;
   566→
   567→    // ---- Core: Find best teacher for a slot ----
   568→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   569→    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   570→    const findBest = (
   571→      blockedId: string | null,
   572→      slot: Slot,
   573→      pool: Teacher[],
   574→      relaxAdj: boolean
   575→    ): Teacher | null => {
   576→      const candidates: { teacher: Teacher; score: number }[] = [];
   577→      for (const t of pool) {
   578→        if (blockedId && t.id === blockedId) continue;
   579→        const tr = tracking[t.id];
   580→        // HARD: No own-subject supervision
   581→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   582→        // HARD: Stage notes filtering
   583→        if (!canSuperviseStage(t, slot.stage)) continue;
   584→        // HARD: No time overlap on same day
   585→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   586→        // HARD: MAX 1 committee per teacher per day (NEVER relaxed)
   587→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   588→        // SOFT: No same grade on consecutive days (can be relaxed)
   589→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   590→        // Scoring: hours are king, tiny random for variety between equal candidates
   591→        const isAdmin = t.subject === 'Admin';
   592→        const adminBonus = (adminAssignmentCount[t.id] || 0) < adminMinTarget ? -3 : 0;
   593→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (isAdmin ? 3 + adminBonus : 0) + Math.random() * 0.2;
   594→        candidates.push({ teacher: t, score });
   595→      }
   596→      if (candidates.length === 0) return null;
   597→      candidates.sort((a, b) => a.score - b.score);
   598→      const best = candidates[0].score;
   599→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   600→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   601→    };
   602→
   603→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 1-per-day + time) ----
   604→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   605→      const candidates: { teacher: Teacher; score: number }[] = [];
   606→      for (const t of pool) {
   607→        if (blockedId && t.id === blockedId) continue;
   608→        const tr = tracking[t.id];
   609→        // HARD: Still no own-subject supervision even in relaxed mode
   610→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   611→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   612→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   613→        const score = tr.totalHours * 200 + tr.totalComm * 5 + Math.random() * 0.2;
   614→        candidates.push({ teacher: t, score });
   615→      }
   616→      if (candidates.length === 0) return null;
   617→      candidates.sort((a, b) => a.score - b.score);
   618→      return candidates[0].teacher;
   619→    };
   620→
   621→    // ---- Fallback 3: Allow 2nd assignment same day (only time overlap + own-subject blocked) ----
   622→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   623→      const candidates: { teacher: Teacher; score: number }[] = [];
   624→      for (const t of pool) {
   625→        if (blockedId && t.id === blockedId) continue;
   626→        const tr = tracking[t.id];
   627→        // HARD: Still no own-subject supervision even in force-day mode
   628→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   629→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   630→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (tr.dayComm[slot.day] || 0) * 50 + Math.random() * 0.2;
   631→        candidates.push({ teacher: t, score });
   632→      }
   633→      if (candidates.length === 0) return null;
   634→      candidates.sort((a, b) => a.score - b.score);
   635→      return candidates[0].teacher;
   636→    };
   637→
   638→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   639→    let standbyCount = 0;
   640→    const allPool = [...teachers]; // ALL teachers including admin
   641→
   642→    for (const day of DAYS) {
   643→      const daySlots = shuffle(slotsByDay[day]);
   644→      if (daySlots.length === 0) continue;
   645→
   646→      for (const slot of daySlots) {
   647→        let t1: Teacher | null = null;
   648→        let t2: Teacher | null = null;
   649→
   650→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   651→        t1 = findBest(null, slot, allPool, false);
   652→        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   653→        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   654→        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   655→
   656→        if (t1) {
   657→          const tr = tracking[t1.id];
   658→          tr.totalComm++; tr.dayComm[slot.day]++;
   659→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   660→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   661→        } else { standbyCount++; }
   662→
   663→        // T2: same pool, blocked=T1, cascade fallbacks
   664→        const blocked = t1?.id || null;
   665→        t2 = findBest(blocked, slot, allPool, false);
   666→        if (!t2) t2 = findBest(blocked, slot, allPool, true);
   667→        if (!t2) t2 = findBestRelaxed(blocked, slot, allPool);
   668→        if (!t2) t2 = findBestForceDay(blocked, slot, allPool);
   669→
   670→        if (t2) {
   671→          if (t2.subject === 'Admin') {
   672→            adminAssignmentCount[t2.id] = (adminAssignmentCount[t2.id] || 0) + 1;
   673→            totalAdminAssigned++;
   674→          }
   675→          const tr = tracking[t2.id];
   676→          tr.totalComm++; tr.dayComm[slot.day]++;
   677→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   678→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   679→        } else { standbyCount++; }
   680→
   681→        // Record assignment
   682→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   683→        if (session) {
   684→          session.committees.push({
   685→            serial: slot.comId,
   686→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   687→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   688→          });
   689→        }
   690→      }
   691→    }
   692→
   693→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   694→    // This eliminates any tracking accumulation bugs
   695→    const teacherCommCount: Record<string, number> = {};
   696→    const teacherHoursFromAssign: Record<string, number> = {};
   697→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   698→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   699→    teachers.forEach(t => {
   700→      teacherCommCount[t.id] = 0;
   701→      teacherHoursFromAssign[t.id] = 0;
   702→      teacherDayCommFromAssign[t.id] = {};
   703→      teacherSlotsFromAssign[t.id] = [];
   704→    });
   705→
   706→    for (const day of DAYS) {
   707→      for (const sess of finalAssignments[day]) {
   708→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   709→        for (const c of sess.committees) {
   710→          [c.t1, c.t2].forEach(who => {
   711→            if (!who.id) return;
   712→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   713→            // Deduplicate: same day + same time = counted once
   714→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   715→            const existing = teacherSlotsFromAssign[who.id];
   716→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   717→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   718→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   719→            }
   720→            // Day committee count
   721→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   722→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   723→          });
   724→        }
   725→      }
   726→    }
   727→
   728→    // Override tracking with verified data from assignments
   729→    teachers.forEach(t => {
   730→      if (tracking[t.id]) {
   731→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   732→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   733→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   734→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   735→      }
   736→    });
   737→
   738→    // ---- Post-distribution: Ensure admin gets at least 2 assignments ----
   739→    for (const admin of adminTeachers) {
   740→      const adminComm = tracking[admin.id].totalComm;
   741→      if (adminComm >= adminMinTarget) continue;
   742→      // Try to swap: find a teacher with > average hours who has an assignment
   743→      // that admin could take (passing all constraints)
   744→      const allTeacherHours = teachers.map(t => tracking[t.id]?.totalHours || 0).filter(h => h > 0);
   745→      const avgH = allTeacherHours.length > 0 ? allTeacherHours.reduce((a, b) => a + b, 0) / allTeacherHours.length : 0;
   746→      const overburdened = teachers.filter(t => t.id !== admin.id && tracking[t.id]?.totalHours > avgH);
   747→      // Sort by most overburdened first
   748→      overburdened.sort((a, b) => (tracking[b.id]?.totalHours || 0) - (tracking[a.id]?.totalHours || 0));
   749→
   750→      for (const victim of overburdened) {
   751→        if (tracking[admin.id].totalComm >= adminMinTarget) break;
   752→        // Find an assignment of victim that admin can take
   753→        for (const day of DAYS) {
   754→          if (tracking[admin.id].totalComm >= adminMinTarget) break;
   755→          const daySlots = teacherSlotsFromAssign[victim.id].filter(s => s.day === day);
   756→          if (daySlots.length === 0) continue;
   757→          // Admin already has assignment on this day?
   758→          if ((tracking[admin.id].dayComm[day] || 0) >= 1) continue;
   759→          // Find the session+committee for this slot
   760→          for (const sess of finalAssignments[day]) {
   761→            if (tracking[admin.id].totalComm >= adminMinTarget) break;
   762→            const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   763→            if (daySlots[0].start !== sessTimeInfo.start || daySlots[0].end !== sessTimeInfo.end) continue;
   764→            // Check admin constraints
   765→            if (ruleSubject && sess.subject && admin.subject === sess.subject) continue;
   766→            if (!canSuperviseStage(admin, getStage(sess.grade))) continue;
   767→            // Check admin no time overlap (should be fine since we checked dayComm above)
   768→            if (tracking[admin.id].assignedSlots.some(s => s.day === day && !(sessTimeInfo.end <= s.start || sessTimeInfo.start >= s.end))) continue;
   769→            // Find which position (t1 or t2) victim has in this session
   770→            for (const c of sess.committees) {
   771→              if (c.t1?.id === victim.id || c.t2?.id === victim.id) {
   772→                // Check consecutive grade constraint
   773→                if (wasSameGradeAdjacent(tracking[admin.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTimeInfo, comId: 0 })) continue;
   774→                // Do the swap
   775→                const isT1 = c.t1?.id === victim.id;
   776→                const swappedTeacher = isT1 ? c.t1 : c.t2;
   777→                if (isT1) { c.t1 = { id: admin.id, name: admin.name }; }
   778→                else { c.t2 = { id: admin.id, name: admin.name }; }
   779→                // Update tracking: remove from victim, add to admin
   780→                tracking[victim.id].totalComm--;
   781→                tracking[victim.id].dayComm[day] = Math.max(0, (tracking[victim.id].dayComm[day] || 0) - 1);
   782→                tracking[admin.id].totalComm++;
   783→                tracking[admin.id].dayComm[day] = (tracking[admin.id].dayComm[day] || 0) + 1;
   784→                tracking[admin.id].assignedSlots.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   785→                tracking[admin.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   786→                break;
   787→              }
   788→            }
   789→          }
   790→        }
   791→      }
   792→    }
   793→
   794→    // ---- Post-distribution: BALANCE PASS ----
   795→    // Iteratively swap assignments from overburdened → underburdened teachers
   796→    // to minimize the max-min spread in hours
   797→    const BALANCE_ITERATIONS = 150;
   798→    const recalcHours = () => {
   799→      teachers.forEach(t => {
   800→        const slots = tracking[t.id].assignedSlots || [];
   801→        const seen = new Set<string>();
   802→        let hrs = 0;
   803→        slots.forEach(s => {
   804→          const key = s.day + '_' + s.start + '_' + s.end;
   805→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   806→        });
   807→        tracking[t.id].totalHours = hrs;
   808→      });
   809→    };
   810→    recalcHours(); // recalc after admin swaps first
   811→
   812→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   813→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   814→      if (active.length < 2) break;
   815→      const sorted = [...active].sort((a, b) => tracking[b.id].totalHours - tracking[a.id].totalHours);
   816→      const maxH = tracking[sorted[0].id].totalHours;
   817→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   818→      if (maxH - minH <= 1.0) break; // spread ≤ 1h — good enough
   819→
   820→      let swapped = false;
   821→      // Try top-3 donors × bottom-3 recipients
   822→      for (let di = 0; di < Math.min(sorted.length, 3) && !swapped; di++) {
   823→        const donor = sorted[di];
   824→        // Don't strip admins below their minimum
   825→        const isDonorAdmin = donor.subject === 'Admin';
   826→        if (tracking[donor.id].totalComm <= 1) continue;
   827→        if (isDonorAdmin && tracking[donor.id].totalComm <= adminMinTarget) continue;
   828→
   829→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 3) && !swapped; ri--) {
   830→          const recip = sorted[ri];
   831→          if (donor.id === recip.id) continue;
   832→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   833→
   834→          // Try each day where donor has an assignment and recipient doesn't
   835→          for (const day of DAYS) {
   836→            if (swapped) break;
   837→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   838→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   839→
   840→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   841→            if (!donorSlot) continue;
   842→
   843→            // Find the session matching donor's slot
   844→            for (const sess of finalAssignments[day]) {
   845→              if (swapped) break;
   846→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   847→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   848→
   849→              // Check recipient constraints
   850→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   851→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   852→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   853→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   854→
   855→              // Find donor in this session's committees
   856→              for (const c of sess.committees) {
   857→                const isT1 = c.t1?.id === donor.id;
   858→                const isT2 = c.t2?.id === donor.id;
   859→                if (!isT1 && !isT2) continue;
   860→
   861→                // Execute swap
   862→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   863→                else c.t2 = { id: recip.id, name: recip.name };
   864→
   865→                tracking[donor.id].totalComm--;
   866→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   867→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   868→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   869→
   870→                tracking[recip.id].totalComm++;
   871→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   872→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   873→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   874→
   875→                swapped = true;
   876→                break;
   877→              }
   878→            }
   879→          }
   880→        }
   881→      }
   882→      if (!swapped) break; // no valid swap found
   883→      recalcHours(); // update hours after each swap
   884→    }
   885→
   886→    // Final hour recalculation after all balancing
   887→    recalcHours();
   888→
   889→    // ---- Sort committees by serial within each session (1, 2, 3...) ----
   890→    for (const day of DAYS) {
   891→      for (const sess of finalAssignments[day]) {
   892→        sess.committees.sort((a, b) => a.serial - b.serial);
   893→      }
   894→    }
   895→
   896→    // ---- Post-distribution: STANDBY ASSIGNMENT ----
   897→    // 1 standby teacher per stage per day from unassigned teachers
   898→    // CRITICAL: Standby must NOT be the same subject as any exam that day for that stage
   899→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   900→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   901→    const STANDBY_PER_STAGE = 1;
