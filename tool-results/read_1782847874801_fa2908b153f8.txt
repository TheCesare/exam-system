    46→function parseTimeRange(tStr: string) {
    47→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    48→  try {
    49→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    50→    const parts = clean.split('-');
    51→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    52→    let start = toMinutes(parts[0]);
    53→    let end = toMinutes(parts[1]);
    54→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    55→    if (start < 300 && end > start + 120) start += 720;
    56→    if (end <= start) end += 720;
    57→    let duration = (end - start) / 60;
    58→    // Sanity cap: no exam is longer than 5 hours
    59→    if (duration > 5) duration = 2.0; // fallback to standard
    60→    return { start, end, duration };
    61→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    62→}

    64→function getStage(grade: string) {
    65→  const g = grade.toLowerCase();
    66→  if (g.includes('primary')) return 'primary';
    67→  if (g.includes('prep')) return 'prep';
    68→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    69→  return 'any';
    70→}

   594→  // ========== DISTRIBUTION ENGINE v4 ==========
   595→  const runDistribution = () => {
   596→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   597→
   598→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   599→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   600→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   601→    const includeAdminW2 = (document.getElementById('rule-admin-w2') as HTMLInputElement)?.checked ?? false;
   602→
   603→    // ---- Admin handling: Admin subject excluded by default, included only for W2 peak days ----

   605→    // ---- Initialize tracking ----

   612→    // ---- Build exam slots grouped by day (chronological order) ----
   613→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   614→    const slotsByDay: Record<string, Slot[]> = {};
   615→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   616→    DAYS.forEach((day, dayIndex) => {

   621→          for (let c = 1; c <= cell.committees; c++) {
   622→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   623→          }
   624→        }
   625→      });
   626→    });
   627→
   628→    // ---- Build final assignments structure ----
   629→    const finalAssignments: Record<string, SessionResult[]> = {};
   630→    DAYS.forEach(d => { finalAssignments[d] = []; });
   631→    DAYS.forEach(day => {
   632→      GRADES.forEach(grade => {
   633→        const cell = scheduleBuffer[`${grade}__${day}`];
   634→        if (cell && cell.committees > 0) {

   636→          if (!existing) {

   643→    // ---- Helper: Stage notes — HARD constraint ----
   644→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   645→      if (!ruleNotes) return true;
   646→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   647→      const n = teacher.notes.toLowerCase();
   648→      const allowed: string[] = [];
   649→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   650→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   651→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   652→      if (allowed.length === 0) return true;
   653→      return allowed.includes(slotStage);
   654→    };
   655→
   656→    // ---- Helper: Consecutive-day same-grade check ----
   657→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   658→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   659→    };
   660→
   661→    // ---- Helper: Special subject pairing rules ----
   662→    // Religion exam: Arabic + Religion teachers blocked
   663→    // Arabic exam: Religion teachers blocked from primary only
   664→    const isBlockedBySpecialRule = (teacher: Teacher, slot: Slot): boolean => {
   665→      if (slot.subject === 'Religion' && (teacher.subject === 'Arabic' || teacher.subject === 'Religion')) return true;
   666→      if (slot.subject === 'Arabic' && teacher.subject === 'Religion' && slot.stage === 'primary') return true;
   667→      return false;
   668→    };
   669→
   670→    // ---- Helper: Fisher-Yates shuffle ----
   671→    const shuffle = <T,>(arr: T[]): T[] => {
   672→      const a = [...arr];
   673→      for (let i = a.length - 1; i > 0; i--) {
   674→        const j = Math.floor(Math.random() * (i + 1));
   675→        [a[i], a[j]] = [a[j], a[i]];
   676→      }
   677→      return a;
   678→    };
   679→
   680→    // ---- Helper: Special subject exclusion rules (Religion/Arabic) ----
   681→    // Rule 1: If exam subject is Religion (الدين), teachers with subject Arabic (العربي) are excluded (Religion teachers already caught by own-subject)
   682→    // Rule 2: If exam subject is Arabic (العربي), teachers with subject Religion (الدين) are excluded for primary stage only
   683→    const isSubjectExcluded = (t: Teacher, slot: Slot): boolean => {
   684→      if (!ruleSubject || !slot.subject) return false;
   685→      if (slot.subject === 'الدين') {
   686→        return t.subject === 'العربي';
   687→      }
   688→      if (slot.subject === 'العربي' && slot.stage === 'primary') {
   689→        return t.subject === 'الدين';
   690→      }
   691→      return false;
   692→    };
   693→
   694→    // ---- Core: Find best teacher for a slot ----
   695→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   696→    // Scoring: hours dominate (500x), tiny noise (0.1) for variety
   697→    const findBest = (
   698→      blockedId: string | null,
   699→      slot: Slot,
   700→      pool: Teacher[],
   701→      relaxAdj: boolean
   702→    ): Teacher | null => {
   703→      const candidates: { teacher: Teacher; score: number }[] = [];
   704→      for (const t of pool) {
   705→        if (blockedId && t.id === blockedId) continue;
   706→        const tr = tracking[t.id];
   707→        // HARD: No own-subject supervision
   708→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   709→        // HARD: Special subject exclusion (Religion/Arabic rules)
   710→        if (isSubjectExcluded(t, slot)) continue;
   711→        // HARD: Stage notes filtering
   712→        if (!canSuperviseStage(t, slot.stage)) continue;
   713→        // HARD: No time overlap on same day
   714→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   715→        // HARD: MAX 2 committees per teacher per day (NEVER relaxed)
   716→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   717→        // SOFT: No same grade on consecutive days (can be relaxed)
   718→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   719→        // Scoring: HOURS dominate (500x), then committees (10x), tiny noise for variety
   720→        // Admin subject = HEAVY penalty (always last choice)
   721→        const isAdmin = t.subject === 'Admin';
   722→        const adminPenalty = isAdmin ? 50000 : 0;
   723→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   724→        candidates.push({ teacher: t, score });
   725→      }
   726→      if (candidates.length === 0) return null;
   727→      candidates.sort((a, b) => a.score - b.score);
   728→      const best = candidates[0].score;
   729→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   730→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   731→    };
   732→
   733→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 2-per-day + time) ----
   734→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   735→      const candidates: { teacher: Teacher; score: number }[] = [];
   736→      for (const t of pool) {
   737→        if (blockedId && t.id === blockedId) continue;
   738→        const tr = tracking[t.id];
   739→        // HARD: Still no own-subject supervision even in relaxed mode
   740→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   741→        // HARD: Special subject exclusion (Religion/Arabic rules) — never relaxed
   742→        if (isSubjectExcluded(t, slot)) continue;
   743→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   744→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   745→        const isAdmin = t.subject === 'Admin';
   746→        const adminPenalty = isAdmin ? 50000 : 0;
   747→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   748→        candidates.push({ teacher: t, score });
   749→      }
   750→      if (candidates.length === 0) return null;
   751→      candidates.sort((a, b) => a.score - b.score);
   752→      return candidates[0].teacher;
   753→    };
   754→
   755→    // ---- Fallback 3: Allow 2nd assignment same day (time overlap + own-subject + max 2/day blocked) ----
   756→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   757→      const candidates: { teacher: Teacher; score: number }[] = [];
   758→      for (const t of pool) {
   759→        if (blockedId && t.id === blockedId) continue;
   760→        const tr = tracking[t.id];
   761→        // HARD: Still no own-subject supervision even in force-day mode
   762→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   763→        // HARD: Special subject exclusion (Religion/Arabic rules) — never relaxed
   764→        if (isSubjectExcluded(t, slot)) continue;
   765→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   766→        // HARD: ABSOLUTE MAX 2 committees per teacher per day (never exceeded)
   767→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   768→        const isAdmin = t.subject === 'Admin';
   769→        const adminPenalty = isAdmin ? 50000 : 0;
   770→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (tr.dayComm[slot.day] || 0) * 100 + adminPenalty + Math.random() * 0.1;
   771→        candidates.push({ teacher: t, score });
   772→      }
   773→      if (candidates.length === 0) return null;
   774→      candidates.sort((a, b) => a.score - b.score);
   775→      return candidates[0].teacher;
   776→    };
   777→
   778→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   779→    let standbyCount = 0;
   780→
   781→    // ---- 'Old' teacher handling: senior teachers excluded from default pool, used as last resort ----
   782→    const isOldTeacher = (t: Teacher) => !!(t.notes && t.notes.toLowerCase().includes('old'));
   783→    const oldTeachers = teachers.filter(t => t.subject !== 'Admin' && isOldTeacher(t));
   784→    const nonAdminTeachers = teachers.filter(t => t.subject !== 'Admin' && !isOldTeacher(t));
   785→
   786→    // Track old teacher supervision days (max 3, min 1)
   787→    const oldTeacherDays: Record<string, number> = {};
   788→    oldTeachers.forEach(t => { oldTeacherDays[t.id] = 0; });
   789→
   790→    // Calculate Week 2 peak days (days with above-average committee count)
   791→    const w2SlotCounts = WEEK2_DAYS.map(d => slotsByDay[d].length).filter(c => c > 0);
   792→    const w2Avg = w2SlotCounts.length > 0 ? w2SlotCounts.reduce((a, b) => a + b, 0) / w2SlotCounts.length : 0;
   793→    const peakW2Days = new Set(WEEK2_DAYS.filter(d => slotsByDay[d].length >= w2Avg && slotsByDay[d].length > 0));
   794→
   795→    for (const day of DAYS) {
   796→      // Determine pool: Admin only included on W2 peak days when toggle is ON
   797→      const isW2Peak = WEEK2_DAYS.includes(day) && peakW2Days.has(day);
   798→      const dayPool = (includeAdminW2 && isW2Peak) ? [...teachers] : nonAdminTeachers;
   799→      const daySlots = shuffle(slotsByDay[day]);
   800→      if (daySlots.length === 0) continue;
   801→
   802→      for (const slot of daySlots) {
   803→        let t1: Teacher | null = null;
   804→        let t2: Teacher | null = null;
   805→
   806→        // Helper: filter old teachers not exceeding 3-day max and not already assigned today
   807→        const availableOldTeachers = (blockedId: string | null, day: string) => {
   808→          return oldTeachers.filter(ot => {
   809→            if (blockedId && ot.id === blockedId) return false;
   810→            if ((oldTeacherDays[ot.id] || 0) >= 3) return false;
   811→            const tr = tracking[ot.id];
   812→            if (tr.assignedSlots.some(s => s.day === day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) return false;
   813→            return true;
   814→          });
   815→        };
   816→
   817→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day → OLD teacher last resort
   818→        t1 = findBest(null, slot, dayPool, false);
   819→        if (!t1) t1 = findBest(null, slot, dayPool, true); // relax consecutive-day
   820→        if (!t1) t1 = findBestRelaxed(null, slot, dayPool); // relax subject + stage
   821→        if (!t1) t1 = findBestForceDay(null, slot, dayPool); // allow 2nd assignment same day
   822→        // Last resort: try old teachers (max 3 days)
   823→        if (!t1) t1 = findBest(null, slot, availableOldTeachers(null, slot.day), false);
   824→        if (!t1) t1 = findBestRelaxed(null, slot, availableOldTeachers(null, slot.day));
   825→
   826→        if (t1) {
   827→          const tr = tracking[t1.id];
   828→          tr.totalComm++; tr.dayComm[slot.day]++;
   829→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   830→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   831→          // Track old teacher days
   832→          if (isOldTeacher(t1) && !tr.assignedSlots.some((s, idx) => idx < tr.assignedSlots.length - 1 && s.day === slot.day)) {
   833→            oldTeacherDays[t1.id] = (oldTeacherDays[t1.id] || 0) + 1;
   834→          }
   835→        } else { standbyCount++; }
   836→
   837→        // T2: pairing rule — classified teacher (has notes) must pair with unclassified
   838→        const blocked = t1?.id || null;
   839→        const t1Classified = t1 ? !!(t1.notes && t1.notes.trim() !== '') : false;
   840→        const t2Filtered = t1Classified ? dayPool.filter(t => t.id !== blocked && (!t.notes || t.notes.trim() === '')) : dayPool;
   841→
   842→        t2 = findBest(blocked, slot, t2Filtered, false);
   843→        if (!t2) t2 = findBest(blocked, slot, t2Filtered, true);
   844→        if (!t2) t2 = findBestRelaxed(blocked, slot, t2Filtered);
   845→        if (!t2) t2 = findBestForceDay(blocked, slot, t2Filtered);
   846→        // Relax pairing rule if no unclassified teacher available
   847→        if (!t2 && t1Classified) {
   848→          t2 = findBest(blocked, slot, dayPool, false);
   849→          if (!t2) t2 = findBest(blocked, slot, dayPool, true);
   850→          if (!t2) t2 = findBestRelaxed(blocked, slot, dayPool);
   851→          if (!t2) t2 = findBestForceDay(blocked, slot, dayPool);
   852→        }
   853→        // Last resort: try old teachers for T2
   854→        if (!t2) {
   855→          const oldPool = availableOldTeachers(blocked, slot.day);
   856→          t2 = findBest(blocked, slot, oldPool, false);
   857→          if (!t2) t2 = findBestRelaxed(blocked, slot, oldPool);
   858→        }
   859→
   860→        if (t2) {
   861→          const tr = tracking[t2.id];
   862→          tr.totalComm++; tr.dayComm[slot.day]++;
   863→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   864→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   865→          // Track old teacher days
   866→          if (isOldTeacher(t2) && !tr.assignedSlots.some((s, idx) => idx < tr.assignedSlots.length - 1 && s.day === slot.day)) {
   867→            oldTeacherDays[t2.id] = (oldTeacherDays[t2.id] || 0) + 1;
   868→          }
   869→        } else { standbyCount++; }
   870→
   871→        // Record assignment
   872→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   873→        if (session) {
   874→          session.committees.push({
   875→            serial: slot.comId,
   876→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   877→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   878→          });
   879→        }
   880→      }
   881→    }
   882→
   883→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   884→    // This eliminates any tracking accumulation bugs
   885→    const teacherCommCount: Record<string, number> = {};
   886→    const teacherHoursFromAssign: Record<string, number> = {};
   887→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   888→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   889→    teachers.forEach(t => {
   890→      teacherCommCount[t.id] = 0;
   891→      teacherHoursFromAssign[t.id] = 0;
   892→      teacherDayCommFromAssign[t.id] = {};
   893→      teacherSlotsFromAssign[t.id] = [];
   894→    });
   895→
   896→    for (const day of DAYS) {
   897→      for (const sess of finalAssignments[day]) {
   898→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   899→        for (const c of sess.committees) {
   900→          [c.t1, c.t2].forEach(who => {
   901→            if (!who.id) return;
   902→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   903→            // Deduplicate: same day + same time = counted once
   904→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   905→            const existing = teacherSlotsFromAssign[who.id];
   906→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   907→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   908→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   909→            }
   910→            // Day committee count
   911→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   912→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   913→          });
   914→        }
   915→      }
   916→    }
   917→
   918→    // Override tracking with verified data from assignments
   919→    teachers.forEach(t => {
   920→      if (tracking[t.id]) {
   921→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   922→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   923→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   924→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   925→      }
   926→    });
   927→
   928→    // ---- Post-distribution: Admin already handled by pool exclusion ----
   929→
   930→    // ---- Post-distribution: BALANCE PASS ----
   931→    // Iteratively swap assignments from overburdened → underburdened teachers
   932→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   933→    const BALANCE_ITERATIONS = 500;
   934→    const recalcHours = () => {
   935→      teachers.forEach(t => {
   936→        const slots = tracking[t.id].assignedSlots || [];
   937→        const seen = new Set<string>();
   938→        let hrs = 0;
   939→        slots.forEach(s => {
   940→          const key = s.day + '_' + s.start + '_' + s.end;
   941→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   942→        });
   943→        tracking[t.id].totalHours = hrs;
   944→      });
   945→    };
   946→    recalcHours(); // recalc after admin swaps first
   947→
   948→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   949→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   950→      if (active.length < 2) break;
   951→      // Sort by HOURS first (primary), then committees (secondary)
   952→      const sorted = [...active].sort((a, b) => {
   953→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   954→        if (Math.abs(hDiff) > 0.1) return hDiff;
   955→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   956→      });
   957→      const maxH = tracking[sorted[0].id].totalHours;
   958→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   959→      if (maxH - minH <= 0.5) break; // spread ≤ 0.5h — tight balance
   960→
   961→      let swapped = false;
   962→      // Try top-5 donors × bottom-5 recipients for wider swap search
   963→      for (let di = 0; di < Math.min(sorted.length, 5) && !swapped; di++) {
   964→        const donor = sorted[di];
   965→        // Don't strip admins below 1 assignment in balance pass
   966→        // Never swap OUT old teachers (they are last-resort only)
   967→        if (tracking[donor.id].totalComm <= 1) continue;
   968→        if (isOldTeacher(donor)) continue;
   969→
   970→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 5) && !swapped; ri--) {
   971→          const recip = sorted[ri];
   972→          if (donor.id === recip.id) continue;
   973→          // NEVER give admin more assignments during balance pass
   974→          // Old teachers CAN be recipients (swapped IN) but check their day limit

   980→            const recipCommAfter = tracking[recip.id].totalComm + 1;

   988→
   989→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   990→            if (!donorSlot) continue;
   991→
   992→            // Find the session matching donor's slot
   993→            for (const sess of finalAssignments[day]) {
   994→              if (swapped) break;

   996→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;

   998→              // Check recipient constraints
   999→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;

  1002→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;

  1004→
  1005→              // Find donor in this session's committees
  1006→              for (const c of sess.committees) {
  1007→                const isT1 = c.t1?.id === donor.id;
  1008→                const isT2 = c.t2?.id === donor.id;
  1009→                if (!isT1 && !isT2) continue;
  1010→
  1011→                // Execute swap
  1012→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
  1013→                else c.t2 = { id: recip.id, name: recip.name };
  1014→
  1015→                tracking[donor.id].totalComm--;
  1016→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
  1017→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
  1018→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
  1019→
  1020→                tracking[recip.id].totalComm++;
  1021→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;

  1026→                if (isOldTeacher(recip)) {
  1027→                  oldTeacherDays[recip.id] = (oldTeacherDays[recip.id] || 0) + 1;
  1028→                }
  1029→
  1030→                swapped = true;
  1031→                break;
  1032→              }
  1033→            }
  1034→          }
  1035→        }
  1036→      }
  1037→      if (!swapped) break; // no valid swap found

  1104→
  1105→          // Not already standby at this same time on this day

  1107→

  1109→        });
  1110→
  1111→        // Rotation: deprioritize teachers who were standby YESTERDAY for ANY session

  1114→          const prevDay = DAYS[di - 1];

  1118→        }
  1119→

  1121→        const isCompletelyFreeToday = (t: Teacher): boolean => {
  1122→          const tr = tracking[t.id];
  1123→          return !tr.assignedSlots.some(s => s.day === day);
  1124→        };
  1125→
  1126→        candidates.sort((a, b) => {
  1127→          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
  1128→          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
  1129→          const aFree = isCompletelyFreeToday(a) ? 0 : 10;
  1130→          const bFree = isCompletelyFreeToday(b) ? 0 : 10;

  1132→        });
  1133→
  1134→        // Pick min(2, candidates.length) teachers
  1135→        const picked = candidates.slice(0, Math.min(2, candidates.length));
  1136→        session.standbys = picked.map(t => ({ id: t.id, name: t.name }));
  1137→
  1138→        // Track rotation count and mark as standby at this time slot
  1139→        for (const s of picked) {
  1140→          teacherStandbyCount[s.id]++;
  1141→          todayStandbyByTime[timeKey].add(s.id);
  1142→        }
  1143→      }
  1144→    }
  1145→

  1147→    const violations: string[] = [];

  1151→        for (const sess of finalAssignments[day]) {
  1152→          for (const c of sess.committees) {
  1153→            [c.t1, c.t2].forEach(who => {
  1154→              if (who.id) {
  1155→                const tch = teachers.find(x => x.id === who.id);
  1156→                if (tch && tch.subject === sess.subject) violations.push(`${tch.name} -> own subject (${sess.subject}) on ${day}`);
  1157→              }
  1158→            });
  1159→          }
  1160→        }
  1161→      }

  1172→        }
  1173→      }

  1177→      const tr = tracking[t.id];

  1179→        if ((tr.dayComm[day] || 0) > 2) {

  1182→      }
  1183→    }

  1189→          for (const c of sess.committees) {
  1190→            [c.t1, c.t2].forEach(who => {
