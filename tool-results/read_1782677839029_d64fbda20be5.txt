    45→function parseTimeRange(tStr: string) {
    46→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    47→  try {
    48→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    49→    const parts = clean.split('-');
    50→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    51→    let start = toMinutes(parts[0]);
    52→    let end = toMinutes(parts[1]);
    53→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    54→    if (start < 300 && end > start + 120) start += 720;
    55→    if (end <= start) end += 720;
    56→    let duration = (end - start) / 60;
    57→    // Sanity cap: no exam is longer than 5 hours
    58→    if (duration > 5) duration = 2.0; // fallback to standard
    59→    return { start, end, duration };
    60→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    61→}

    63→function getStage(grade: string) {
    64→  const g = grade.toLowerCase();
    65→  if (g.includes('primary')) return 'primary';
    66→  if (g.includes('prep')) return 'prep';
    67→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    68→  return 'any';
    69→}

   475→  const importCSV = () => {
   476→    const input = document.createElement('input');
   477→    input.type = 'file'; input.accept = '.csv';
   478→    input.onchange = async (e) => {
   479→      const file = (e.target as HTMLInputElement).files?.[0];
   480→      if (!file) return;
   481→      const text = await file.text();
   482→      const lines = text.split('\n');
   483→      const imported: { name: string; subject: string; notes: string }[] = [];
   484→      lines.forEach(line => {
   485→        const parts = line.split(',').map(p => p.trim());
   486→        if (parts.length >= 2 && parts[0]) imported.push({ name: parts[0], subject: parts[1], notes: parts[2] || '' });
   487→      });
   488→      if (imported.length === 0) { showToast('No valid data in CSV', 'error'); return; }
   489→      try {
   490→        await fetch('/api/distribute', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ teachers: imported }) });
   491→        // Reset order - will be re-initialized from fresh data

   530→  // ========== DISTRIBUTION ENGINE v4 ==========
   531→  const runDistribution = () => {
   532→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   533→
   534→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   535→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   536→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   537→
   538→    // ---- Admin handling: Admin subject = LAST resort, fewer assignments ----
   539→    const adminTeachers = teachers.filter(t => t.subject === 'Admin');
   540→    const nonAdminTeachers = teachers.filter(t => t.subject !== 'Admin');
   541→
   542→    // ---- Initialize tracking ----
   543→    const tracking: Record<string, TrackingEntry> = {};
   544→    teachers.forEach(t => {
   545→      tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {} as Record<string,number>, assignedSlots: [], gradeHistory: [] };
   546→      DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   547→    });
   548→
   549→    // ---- Build exam slots grouped by day (chronological order) ----
   550→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   551→    const slotsByDay: Record<string, Slot[]> = {};
   552→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   553→    DAYS.forEach((day, dayIndex) => {
   554→      GRADES.forEach(grade => {
   555→        const cell = scheduleBuffer[`${grade}__${day}`];
   556→        if (cell && cell.committees > 0) {
   557→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   558→          for (let c = 1; c <= cell.committees; c++) {
   559→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   560→          }
   561→        }
   562→      });
   563→    });
   564→
   565→    // ---- Build final assignments structure ----
   566→    const finalAssignments: Record<string, SessionResult[]> = {};
   567→    DAYS.forEach(d => { finalAssignments[d] = []; });
   568→    DAYS.forEach(day => {
   569→      GRADES.forEach(grade => {
   570→        const cell = scheduleBuffer[`${grade}__${day}`];
   571→        if (cell && cell.committees > 0) {
   572→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   573→          if (!existing) {
   574→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   575→          }
   576→        }
   577→      });
   578→    });
   579→
   580→    // ---- Helper: Stage notes — HARD constraint ----
   581→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   582→      if (!ruleNotes) return true;
   583→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   584→      const n = teacher.notes.toLowerCase();
   585→      const allowed: string[] = [];
   586→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   587→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   588→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   589→      if (allowed.length === 0) return true;
   590→      return allowed.includes(slotStage);
   591→    };
   592→
   593→    // ---- Helper: Consecutive-day same-grade check ----
   594→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   595→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   596→    };
   597→
   598→    // ---- Helper: Fisher-Yates shuffle ----
   599→    const shuffle = <T,>(arr: T[]): T[] => {
   600→      const a = [...arr];
   601→      for (let i = a.length - 1; i > 0; i--) {
   602→        const j = Math.floor(Math.random() * (i + 1));
   603→        [a[i], a[j]] = [a[j], a[i]];
   604→      }
   605→      return a;
   606→    };
   607→
   608→    // ---- Admin minimum assignment tracking ----
   609→    const adminAssignmentCount: Record<string, number> = {};
   610→    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   611→    const adminMinTarget = 0; // admin = last resort, no minimum
   612→    let totalAdminAssigned = 0;
   613→
   614→    // ---- Core: Find best teacher for a slot ----
   615→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   616→    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   617→    const findBest = (
   618→      blockedId: string | null,
   619→      slot: Slot,
   620→      pool: Teacher[],
   621→      relaxAdj: boolean
   622→    ): Teacher | null => {
   623→      const candidates: { teacher: Teacher; score: number }[] = [];
   624→      for (const t of pool) {
   625→        if (blockedId && t.id === blockedId) continue;
   626→        const tr = tracking[t.id];
   627→        // HARD: No own-subject supervision
   628→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   629→        // HARD: Stage notes filtering
   630→        if (!canSuperviseStage(t, slot.stage)) continue;
   631→        // HARD: No time overlap on same day
   632→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   633→        // HARD: MAX 2 committees per teacher per day (NEVER relaxed)
   634→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   635→        // SOFT: No same grade on consecutive days (can be relaxed)
   636→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   637→        // Scoring: HOURS dominate (500x), then committees (10x), tiny noise for variety
   638→        // Admin subject = HEAVY penalty (always last choice)
   639→        const isAdmin = t.subject === 'Admin';
   640→        const adminPenalty = isAdmin ? 50000 : 0;
   641→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   642→        candidates.push({ teacher: t, score });
   643→      }
   644→      if (candidates.length === 0) return null;
   645→      candidates.sort((a, b) => a.score - b.score);
   646→      const best = candidates[0].score;
   647→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   648→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   649→    };
   650→
   651→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 2-per-day + time) ----
   652→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   653→      const candidates: { teacher: Teacher; score: number }[] = [];
   654→      for (const t of pool) {
   655→        if (blockedId && t.id === blockedId) continue;
   656→        const tr = tracking[t.id];
   657→        // HARD: Still no own-subject supervision even in relaxed mode
   658→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   659→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   660→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   661→        const isAdmin = t.subject === 'Admin';
   662→        const adminPenalty = isAdmin ? 50000 : 0;
   663→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   664→        candidates.push({ teacher: t, score });
   665→      }
   666→      if (candidates.length === 0) return null;
   667→      candidates.sort((a, b) => a.score - b.score);
   668→      return candidates[0].teacher;
   669→    };
   670→
   671→    // ---- Fallback 3: Allow 3rd assignment same day (only time overlap + own-subject blocked) ----
   672→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   673→      const candidates: { teacher: Teacher; score: number }[] = [];
   674→      for (const t of pool) {
   675→        if (blockedId && t.id === blockedId) continue;
   676→        const tr = tracking[t.id];
   677→        // HARD: Still no own-subject supervision even in force-day mode
   678→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   679→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   680→        const isAdmin = t.subject === 'Admin';
   681→        const adminPenalty = isAdmin ? 50000 : 0;
   682→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (tr.dayComm[slot.day] || 0) * 100 + adminPenalty + Math.random() * 0.1;
   683→        candidates.push({ teacher: t, score });
   684→      }
   685→      if (candidates.length === 0) return null;
   686→      candidates.sort((a, b) => a.score - b.score);
   687→      return candidates[0].teacher;
   688→    };
   689→
   690→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   691→    let standbyCount = 0;
   692→    const allPool = [...teachers]; // ALL teachers including admin
   693→
   694→    for (const day of DAYS) {
   695→      const daySlots = shuffle(slotsByDay[day]);
   696→      if (daySlots.length === 0) continue;
   697→
   698→      for (const slot of daySlots) {
   699→        let t1: Teacher | null = null;
   700→        let t2: Teacher | null = null;
   701→
   702→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   703→        t1 = findBest(null, slot, allPool, false);
   704→        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   705→        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   706→        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   707→
   708→        if (t1) {
   709→          const tr = tracking[t1.id];
   710→          tr.totalComm++; tr.dayComm[slot.day]++;
   711→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   712→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   713→        } else { standbyCount++; }
   714→
   715→        // T2: same pool, blocked=T1, cascade fallbacks
   716→        const blocked = t1?.id || null;
   717→        t2 = findBest(blocked, slot, allPool, false);
   718→        if (!t2) t2 = findBest(blocked, slot, allPool, true);
   719→        if (!t2) t2 = findBestRelaxed(blocked, slot, allPool);
   720→        if (!t2) t2 = findBestForceDay(blocked, slot, allPool);
   721→
   722→        if (t2) {
   723→          const tr = tracking[t2.id];
   724→          tr.totalComm++; tr.dayComm[slot.day]++;
   725→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   726→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   727→        } else { standbyCount++; }
   728→
   729→        // Record assignment
   730→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   731→        if (session) {
   732→          session.committees.push({
   733→            serial: slot.comId,
   734→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   735→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   736→          });
   737→        }
   738→      }
   739→    }
   740→
   741→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   742→    // This eliminates any tracking accumulation bugs
   743→    const teacherCommCount: Record<string, number> = {};
   744→    const teacherHoursFromAssign: Record<string, number> = {};
   745→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   746→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   747→    teachers.forEach(t => {
   748→      teacherCommCount[t.id] = 0;
   749→      teacherHoursFromAssign[t.id] = 0;
   750→      teacherDayCommFromAssign[t.id] = {};
   751→      teacherSlotsFromAssign[t.id] = [];
   752→    });
   753→
   754→    for (const day of DAYS) {
   755→      for (const sess of finalAssignments[day]) {
   756→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   757→        for (const c of sess.committees) {
   758→          [c.t1, c.t2].forEach(who => {
   759→            if (!who.id) return;
   760→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   761→            // Deduplicate: same day + same time = counted once
   762→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   763→            const existing = teacherSlotsFromAssign[who.id];
   764→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   765→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   766→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   767→            }
   768→            // Day committee count
   769→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   770→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   771→          });
   772→        }
   773→      }
   774→    }
   775→
   776→    // Override tracking with verified data from assignments
   777→    teachers.forEach(t => {
   778→      if (tracking[t.id]) {
   779→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   780→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   781→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   782→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   783→      }
   784→    });
   785→
   786→    // ---- Post-distribution: Admin is LAST resort, no minimum targets ----
   787→    // Admin teachers only get assignments when no other teachers are available
   788→    // (handled by the heavy penalty in scoring)
   789→
   790→    // ---- Post-distribution: BALANCE PASS ----
   791→    // Iteratively swap assignments from overburdened → underburdened teachers
   792→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   793→    const BALANCE_ITERATIONS = 500;
   794→    const recalcHours = () => {
   795→      teachers.forEach(t => {
   796→        const slots = tracking[t.id].assignedSlots || [];
   797→        const seen = new Set<string>();
   798→        let hrs = 0;
   799→        slots.forEach(s => {
   800→          const key = s.day + '_' + s.start + '_' + s.end;
   801→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   802→        });
   803→        tracking[t.id].totalHours = hrs;
   804→      });
   805→    };
   806→    recalcHours(); // recalc after admin swaps first
   807→
   808→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   809→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   810→      if (active.length < 2) break;
   811→      // Sort by HOURS first (primary), then committees (secondary)
   812→      const sorted = [...active].sort((a, b) => {
   813→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   814→        if (Math.abs(hDiff) > 0.1) return hDiff;
   815→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   816→      });
   817→      const maxH = tracking[sorted[0].id].totalHours;
   818→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   819→      if (maxH - minH <= 0.5) break; // spread ≤ 0.5h — tight balance
   820→
   821→      let swapped = false;
   822→      // Try top-5 donors × bottom-5 recipients for wider swap search
   823→      for (let di = 0; di < Math.min(sorted.length, 5) && !swapped; di++) {
   824→        const donor = sorted[di];
   825→        // Don't strip admins below 1 assignment in balance pass
   826→        const isDonorAdmin = donor.subject === 'Admin';
   827→        if (tracking[donor.id].totalComm <= 1) continue;
   828→
   829→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 5) && !swapped; ri--) {
   830→          const recip = sorted[ri];
   831→          if (donor.id === recip.id) continue;
   832→          // NEVER give admin more assignments during balance pass
   833→          if (recip.subject === 'Admin') continue;
   834→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   835→            // Also check if committees would be more balanced after swap
   836→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   837→            const recipCommAfter = tracking[recip.id].totalComm + 1;
   838→            // Skip if swap would make committee imbalance worse (donor already has fewer comms)
   839→
   840→          // Try each day where donor has an assignment and recipient doesn't
   841→          for (const day of DAYS) {
   842→            if (swapped) break;
   843→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   844→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   845→
   846→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   847→            if (!donorSlot) continue;
   848→
   849→            // Find the session matching donor's slot
   850→            for (const sess of finalAssignments[day]) {
   851→              if (swapped) break;
   852→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   853→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   854→
   855→              // Check recipient constraints
   856→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   857→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   858→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   859→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   860→
   861→              // Find donor in this session's committees
   862→              for (const c of sess.committees) {
   863→                const isT1 = c.t1?.id === donor.id;
   864→                const isT2 = c.t2?.id === donor.id;
   865→                if (!isT1 && !isT2) continue;
   866→
   867→                // Execute swap
   868→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   869→                else c.t2 = { id: recip.id, name: recip.name };
   870→
   871→                tracking[donor.id].totalComm--;
   872→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   873→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   874→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   875→
   876→                tracking[recip.id].totalComm++;
   877→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   878→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   879→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   880→
   881→                swapped = true;
   882→                break;
   883→              }
   884→            }
   885→          }
   886→        }
   887→      }
   888→      if (!swapped) break; // no valid swap found
   889→      recalcHours(); // update hours after each swap
   890→    }
   891→
   892→    // Final hour recalculation after all balancing
   893→    recalcHours();
   894→
   895→    // ---- Sort committees by serial within each session (1, 2, 3...) ----
   896→    for (const day of DAYS) {
   897→      for (const sess of finalAssignments[day]) {
   898→        sess.committees.sort((a, b) => a.serial - b.serial);
   899→      }
   900→    }
   901→
   902→    // ---- Post-distribution: STANDBY ASSIGNMENT ----
   903→    // 1 standby teacher per stage per day from unassigned teachers
   904→    // CRITICAL: Standby must NOT be the same subject as any exam that day for that stage
   905→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   906→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   907→    const STANDBY_PER_STAGE = 1;
   908→    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
   909→    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
   910→
   911→    // Track standby count per teacher for rotation fairness
   912→    const teacherStandbyCount: Record<string, number> = {};
   913→    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
   914→
   915→    // Track which teachers are already standby for another stage today
   916→    const todayStandby: Record<string, Set<string>> = {};
   917→    DAYS.forEach(d => { todayStandby[d] = new Set(); });
   918→
   919→    for (let di = 0; di < DAYS.length; di++) {
   920→      const day = DAYS[di];
