    45→function parseTimeRange(tStr: string) {
    46→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    47→  try {
    48→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    49→    const parts = clean.split('-');
    50→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    51→    let start = toMinutes(parts[0]);
    52→    let end = toMinutes(parts[1]);
    53→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    54→    if (start < 300 && end > start + 120) start += 720;
    55→    if (end <= start) end += 720;
    56→    let duration = (end - start) / 60;
    57→    // Sanity cap: no exam is longer than 5 hours
    58→    if (duration > 5) duration = 2.0; // fallback to standard
    59→    return { start, end, duration };
    60→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    61→}

    63→function getStage(grade: string) {
    64→  const g = grade.toLowerCase();
    65→  if (g.includes('primary')) return 'primary';
    66→  if (g.includes('prep')) return 'prep';
    67→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    68→  return 'any';
    69→}

   491→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   492→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   493→
   494→    // ---- All teachers in ONE pool (admin participates normally) ----
   495→    const adminTeachers = teachers.filter(t => t.subject === 'Admin');
   496→
   497→    // ---- Initialize tracking ----
   498→    const tracking: Record<string, TrackingEntry> = {};
   499→    teachers.forEach(t => {
   500→      tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {} as Record<string,number>, assignedSlots: [], gradeHistory: [] };
   501→      DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   502→    });
   503→
   504→    // ---- Build exam slots grouped by day (chronological order) ----
   505→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   506→    const slotsByDay: Record<string, Slot[]> = {};
   507→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   508→    DAYS.forEach((day, dayIndex) => {
   509→      GRADES.forEach(grade => {
   510→        const cell = scheduleBuffer[`${grade}__${day}`];
   511→        if (cell && cell.committees > 0) {
   512→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   513→          for (let c = 1; c <= cell.committees; c++) {
   514→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   515→          }
   516→        }
   517→      });
   518→    });
   519→
   520→    // ---- Build final assignments structure ----
   521→    const finalAssignments: Record<string, SessionResult[]> = {};
   522→    DAYS.forEach(d => { finalAssignments[d] = []; });
   523→    DAYS.forEach(day => {
   524→      GRADES.forEach(grade => {
   525→        const cell = scheduleBuffer[`${grade}__${day}`];
   526→        if (cell && cell.committees > 0) {
   527→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   528→          if (!existing) {
   529→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   530→          }
   531→        }
   532→      });
   533→    });
   534→
   535→    // ---- Helper: Stage notes — HARD constraint ----
   536→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   537→      if (!ruleNotes) return true;
   538→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   539→      const n = teacher.notes.toLowerCase();
   540→      const allowed: string[] = [];
   541→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   542→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   543→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   544→      if (allowed.length === 0) return true;
   545→      return allowed.includes(slotStage);
   546→    };
   547→
   548→    // ---- Helper: Consecutive-day same-grade check ----
   549→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   550→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   551→    };
   552→
   553→    // ---- Helper: Fisher-Yates shuffle ----
   554→    const shuffle = <T,>(arr: T[]): T[] => {
   555→      const a = [...arr];
   556→      for (let i = a.length - 1; i > 0; i--) {
   557→        const j = Math.floor(Math.random() * (i + 1));
   558→        [a[i], a[j]] = [a[j], a[i]];
   559→      }
   560→      return a;
   561→    };
   562→
   563→    // ---- Admin minimum assignment tracking ----
   564→    const adminAssignmentCount: Record<string, number> = {};
   565→    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   566→    const adminMinTarget = 2; // each admin gets at least 2 assignments
   567→    let totalAdminAssigned = 0;
   568→
   569→    // ---- Core: Find best teacher for a slot ----
   570→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   571→    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   572→    const findBest = (
   573→      blockedId: string | null,
   574→      slot: Slot,
   575→      pool: Teacher[],
   576→      relaxAdj: boolean
   577→    ): Teacher | null => {
   578→      const candidates: { teacher: Teacher; score: number }[] = [];
   579→      for (const t of pool) {
   580→        if (blockedId && t.id === blockedId) continue;
   581→        const tr = tracking[t.id];
   582→        // HARD: No own-subject supervision
   583→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   584→        // HARD: Stage notes filtering
   585→        if (!canSuperviseStage(t, slot.stage)) continue;
   586→        // HARD: No time overlap on same day
   587→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   588→        // HARD: MAX 1 committee per teacher per day (NEVER relaxed)
   589→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   590→        // SOFT: No same grade on consecutive days (can be relaxed)
   591→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   592→        // Scoring: hours are king, tiny random for variety between equal candidates
   593→        const isAdmin = t.subject === 'Admin';
   594→        const adminBonus = (adminAssignmentCount[t.id] || 0) < adminMinTarget ? -3 : 0;
   595→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (isAdmin ? 3 + adminBonus : 0) + Math.random() * 0.2;
   596→        candidates.push({ teacher: t, score });
   597→      }
   598→      if (candidates.length === 0) return null;
   599→      candidates.sort((a, b) => a.score - b.score);
   600→      const best = candidates[0].score;
   601→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   602→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   603→    };
   604→
   605→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 1-per-day + time) ----
   606→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   607→      const candidates: { teacher: Teacher; score: number }[] = [];
   608→      for (const t of pool) {
   609→        if (blockedId && t.id === blockedId) continue;
   610→        const tr = tracking[t.id];
   611→        // HARD: Still no own-subject supervision even in relaxed mode
   612→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   613→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   614→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   615→        const score = tr.totalHours * 200 + tr.totalComm * 5 + Math.random() * 0.2;
   616→        candidates.push({ teacher: t, score });
   617→      }
   618→      if (candidates.length === 0) return null;
   619→      candidates.sort((a, b) => a.score - b.score);
   620→      return candidates[0].teacher;
   621→    };
   622→
   623→    // ---- Fallback 3: Allow 2nd assignment same day (only time overlap + own-subject blocked) ----
   624→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   625→      const candidates: { teacher: Teacher; score: number }[] = [];
   626→      for (const t of pool) {
   627→        if (blockedId && t.id === blockedId) continue;
   628→        const tr = tracking[t.id];
   629→        // HARD: Still no own-subject supervision even in force-day mode
   630→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   631→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   632→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (tr.dayComm[slot.day] || 0) * 50 + Math.random() * 0.2;
   633→        candidates.push({ teacher: t, score });
   634→      }
   635→      if (candidates.length === 0) return null;
   636→      candidates.sort((a, b) => a.score - b.score);
   637→      return candidates[0].teacher;
   638→    };
   639→
   640→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   641→    let standbyCount = 0;
   642→    const allPool = [...teachers]; // ALL teachers including admin
   643→
   644→    for (const day of DAYS) {
   645→      const daySlots = shuffle(slotsByDay[day]);
   646→      if (daySlots.length === 0) continue;
   647→
   648→      for (const slot of daySlots) {
   649→        let t1: Teacher | null = null;
   650→        let t2: Teacher | null = null;
   651→
   652→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   653→        t1 = findBest(null, slot, allPool, false);
   654→        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   655→        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   656→        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   657→
   658→        if (t1) {
   659→          const tr = tracking[t1.id];
   660→          tr.totalComm++; tr.dayComm[slot.day]++;
   661→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   662→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   663→        } else { standbyCount++; }
   664→
   665→        // T2: same pool, blocked=T1, cascade fallbacks
   666→        const blocked = t1?.id || null;
   667→        t2 = findBest(blocked, slot, allPool, false);
   668→        if (!t2) t2 = findBest(blocked, slot, allPool, true);
   669→        if (!t2) t2 = findBestRelaxed(blocked, slot, allPool);
   670→        if (!t2) t2 = findBestForceDay(blocked, slot, allPool);
   671→
   672→        if (t2) {
   673→          if (t2.subject === 'Admin') {
   674→            adminAssignmentCount[t2.id] = (adminAssignmentCount[t2.id] || 0) + 1;
   675→            totalAdminAssigned++;
   676→          }
   677→          const tr = tracking[t2.id];
   678→          tr.totalComm++; tr.dayComm[slot.day]++;
   679→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   680→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   681→        } else { standbyCount++; }
   682→
   683→        // Record assignment
   684→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   685→        if (session) {
   686→          session.committees.push({
   687→            serial: slot.comId,
   688→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   689→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   690→          });
   691→        }
   692→      }
   693→    }
   694→
   695→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   696→    // This eliminates any tracking accumulation bugs
   697→    const teacherCommCount: Record<string, number> = {};
   698→    const teacherHoursFromAssign: Record<string, number> = {};
   699→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   700→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   701→    teachers.forEach(t => {
   702→      teacherCommCount[t.id] = 0;
   703→      teacherHoursFromAssign[t.id] = 0;
   704→      teacherDayCommFromAssign[t.id] = {};
   705→      teacherSlotsFromAssign[t.id] = [];
   706→    });
   707→
   708→    for (const day of DAYS) {
   709→      for (const sess of finalAssignments[day]) {
   710→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   711→        for (const c of sess.committees) {
   712→          [c.t1, c.t2].forEach(who => {
   713→            if (!who.id) return;
   714→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   715→            // Deduplicate: same day + same time = counted once
   716→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   717→            const existing = teacherSlotsFromAssign[who.id];
   718→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   719→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   720→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   721→            }
   722→            // Day committee count
   723→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   724→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   725→          });
   726→        }
   727→      }
   728→    }
   729→
   730→    // Override tracking with verified data from assignments
   731→    teachers.forEach(t => {
   732→      if (tracking[t.id]) {
   733→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   734→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   735→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   736→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   737→      }
   738→    });
   739→
   740→    // ---- Post-distribution: Ensure admin gets at least 2 assignments ----
   741→    for (const admin of adminTeachers) {
   742→      const adminComm = tracking[admin.id].totalComm;
   743→      if (adminComm >= adminMinTarget) continue;
   744→      // Try to swap: find a teacher with > average hours who has an assignment
   745→      // that admin could take (passing all constraints)
   746→      const allTeacherHours = teachers.map(t => tracking[t.id]?.totalHours || 0).filter(h => h > 0);
   747→      const avgH = allTeacherHours.length > 0 ? allTeacherHours.reduce((a, b) => a + b, 0) / allTeacherHours.length : 0;
   748→      const overburdened = teachers.filter(t => t.id !== admin.id && tracking[t.id]?.totalHours > avgH);
   749→      // Sort by most overburdened first
   750→      overburdened.sort((a, b) => (tracking[b.id]?.totalHours || 0) - (tracking[a.id]?.totalHours || 0));
   751→
   752→      for (const victim of overburdened) {
   753→        if (tracking[admin.id].totalComm >= adminMinTarget) break;
   754→        // Find an assignment of victim that admin can take
   755→        for (const day of DAYS) {
   756→          if (tracking[admin.id].totalComm >= adminMinTarget) break;
   757→          const daySlots = teacherSlotsFromAssign[victim.id].filter(s => s.day === day);
   758→          if (daySlots.length === 0) continue;
   759→          // Admin already has assignment on this day?
   760→          if ((tracking[admin.id].dayComm[day] || 0) >= 1) continue;
   761→          // Find the session+committee for this slot
   762→          for (const sess of finalAssignments[day]) {
   763→            if (tracking[admin.id].totalComm >= adminMinTarget) break;
   764→            const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   765→            if (daySlots[0].start !== sessTimeInfo.start || daySlots[0].end !== sessTimeInfo.end) continue;
   766→            // Check admin constraints
   767→            if (ruleSubject && sess.subject && admin.subject === sess.subject) continue;
   768→            if (!canSuperviseStage(admin, getStage(sess.grade))) continue;
   769→            // Check admin no time overlap (should be fine since we checked dayComm above)
   770→            if (tracking[admin.id].assignedSlots.some(s => s.day === day && !(sessTimeInfo.end <= s.start || sessTimeInfo.start >= s.end))) continue;
   771→            // Find which position (t1 or t2) victim has in this session
   772→            for (const c of sess.committees) {
   773→              if (c.t1?.id === victim.id || c.t2?.id === victim.id) {
   774→                // Check consecutive grade constraint
   775→                if (wasSameGradeAdjacent(tracking[admin.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTimeInfo, comId: 0 })) continue;
   776→                // Do the swap
   777→                const isT1 = c.t1?.id === victim.id;
   778→                const swappedTeacher = isT1 ? c.t1 : c.t2;
   779→                if (isT1) { c.t1 = { id: admin.id, name: admin.name }; }
   780→                else { c.t2 = { id: admin.id, name: admin.name }; }
   781→                // Update tracking: remove from victim, add to admin
   782→                tracking[victim.id].totalComm--;
   783→                tracking[victim.id].dayComm[day] = Math.max(0, (tracking[victim.id].dayComm[day] || 0) - 1);
   784→                tracking[admin.id].totalComm++;
   785→                tracking[admin.id].dayComm[day] = (tracking[admin.id].dayComm[day] || 0) + 1;
   786→                tracking[admin.id].assignedSlots.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   787→                tracking[admin.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   788→                break;
   789→              }
   790→            }
   791→          }
   792→        }
   793→      }
   794→    }
   795→
   796→    // ---- Post-distribution: BALANCE PASS ----
   797→    // Iteratively swap assignments from overburdened → underburdened teachers
   798→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   799→    const BALANCE_ITERATIONS = 300;
   800→    const recalcHours = () => {
   801→      teachers.forEach(t => {
   802→        const slots = tracking[t.id].assignedSlots || [];
   803→        const seen = new Set<string>();
   804→        let hrs = 0;
   805→        slots.forEach(s => {
   806→          const key = s.day + '_' + s.start + '_' + s.end;
   807→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   808→        });
   809→        tracking[t.id].totalHours = hrs;
   810→      });
   811→    };
   812→    recalcHours(); // recalc after admin swaps first
   813→
   814→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   815→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   816→      if (active.length < 2) break;
   817→      // Sort by HOURS first (primary), then committees (secondary)
   818→      const sorted = [...active].sort((a, b) => {
   819→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   820→        if (Math.abs(hDiff) > 0.1) return hDiff;
   821→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   822→      });
   823→      const maxH = tracking[sorted[0].id].totalHours;
   824→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   825→      if (maxH - minH <= 1.0) break; // spread ≤ 1h — good enough
   826→
   827→      let swapped = false;
   828→      // Try top-3 donors × bottom-3 recipients
   829→      for (let di = 0; di < Math.min(sorted.length, 3) && !swapped; di++) {
   830→        const donor = sorted[di];
   831→        // Don't strip admins below their minimum
   832→        const isDonorAdmin = donor.subject === 'Admin';
   833→        if (tracking[donor.id].totalComm <= 1) continue;
   834→        if (isDonorAdmin && tracking[donor.id].totalComm <= adminMinTarget) continue;
   835→
   836→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 3) && !swapped; ri--) {
   837→          const recip = sorted[ri];
   838→          if (donor.id === recip.id) continue;
   839→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   840→            // Also check if committees would be more balanced after swap
   841→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   842→            const recipCommAfter = tracking[recip.id].totalComm + 1;
   843→            // Skip if swap would make committee imbalance worse (donor already has fewer comms)
   844→
   845→          // Try each day where donor has an assignment and recipient doesn't
   846→          for (const day of DAYS) {
   847→            if (swapped) break;
   848→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   849→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   850→
   851→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   852→            if (!donorSlot) continue;
   853→
   854→            // Find the session matching donor's slot
   855→            for (const sess of finalAssignments[day]) {
   856→              if (swapped) break;
   857→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   858→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   859→
   860→              // Check recipient constraints
   861→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   862→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   863→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   864→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   865→
   866→              // Find donor in this session's committees
   867→              for (const c of sess.committees) {
   868→                const isT1 = c.t1?.id === donor.id;
   869→                const isT2 = c.t2?.id === donor.id;
   870→                if (!isT1 && !isT2) continue;
   871→
   872→                // Execute swap
   873→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   874→                else c.t2 = { id: recip.id, name: recip.name };
   875→
   876→                tracking[donor.id].totalComm--;
   877→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   878→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   879→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   880→
   881→                tracking[recip.id].totalComm++;
   882→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   883→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   884→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   885→
   886→                swapped = true;
   887→                break;
   888→              }
   889→            }
   890→          }
   891→        }
   892→      }
   893→      if (!swapped) break; // no valid swap found
   894→      recalcHours(); // update hours after each swap
   895→    }

   910→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   911→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   912→    const STANDBY_PER_STAGE = 1;
   913→    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
   914→    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
   915→
   916→    // Track standby count per teacher for rotation fairness
   917→    const teacherStandbyCount: Record<string, number> = {};
   918→    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
   919→
   920→    // Track which teachers are already standby for another stage today
   921→    const todayStandby: Record<string, Set<string>> = {};
   922→    DAYS.forEach(d => { todayStandby[d] = new Set(); });
   923→
   924→    for (let di = 0; di < DAYS.length; di++) {
   925→      const day = DAYS[di];
   926→
   927→      for (const stage of STAGES_LIST) {

   929→        const stageSessions = (finalAssignments[day] || []).filter(s => getStage(s.grade) === stage);
   930→        if (stageSessions.length === 0) continue;
   931→

   933→        const dayStageSubjects = new Set<string>();

   935→
   936→        // Find teachers NOT assigned on this day who can supervise this stage
   937→        // AND whose subject doesn't match any exam subject for this stage today

   939→          if ((tracking[t.id].dayComm[day] || 0) >= 1) return false;
   940→          if (todayStandby[day].has(t.id)) return false; // already standby for another stage today
   941→          if (!canSuperviseStage(t, stage)) return false;
   942→          // CRITICAL: Don't pick a teacher whose subject is being examined today

   944→          return true;

   952→            (standbys[prevDay][st] || []).forEach(s => yesterdayStandby.add(s.id));
   953→          }
   954→        }
   955→        candidates.sort((a, b) => {
   956→          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
   957→          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
   958→          return (teacherStandbyCount[a.id] + aWasYesterday) - (teacherStandbyCount[b.id] + bWasYesterday);
   959→        });
   960→
   961→        // Pick up to STANDBY_PER_STAGE
   962→        const picked = candidates.slice(0, STANDBY_PER_STAGE);
   963→        standbys[day][stage] = picked.map(t => ({ id: t.id, name: t.name }));
   964→
   965→        // Track rotation count and mark as standby for today (so same teacher isn't picked for another stage)
   966→        for (const s of picked) {
   967→          teacherStandbyCount[s.id]++;
