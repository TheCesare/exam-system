   583→    // ---- Helper: Stage notes — HARD constraint ----
   584→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   585→      if (!ruleNotes) return true;
   586→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   587→      const n = teacher.notes.toLowerCase();
   588→      const allowed: string[] = [];
   589→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   590→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   591→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   592→      if (allowed.length === 0) return true;
   593→      return allowed.includes(slotStage);
   594→    };
   595→
   596→    // ---- Helper: Consecutive-day same-grade check ----
   597→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   598→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   599→    };
   600→
   601→    // ---- Helper: Special subject pairing rules ----
   602→    // Religion exam: Arabic + Religion teachers blocked
   603→    // Arabic exam: Religion teachers blocked from primary only
   604→    const isBlockedBySpecialRule = (teacher: Teacher, slot: Slot): boolean => {
   605→      if (slot.subject === 'Religion' && (teacher.subject === 'Arabic' || teacher.subject === 'Religion')) return true;
   606→      if (slot.subject === 'Arabic' && teacher.subject === 'Religion' && slot.stage === 'primary') return true;
   607→      return false;
   608→    };

   610→    // ---- Helper: Fisher-Yates shuffle ----
   611→    const shuffle = <T,>(arr: T[]): T[] => {
   612→      const a = [...arr];
   613→      for (let i = a.length - 1; i > 0; i--) {
   614→        const j = Math.floor(Math.random() * (i + 1));
   615→        [a[i], a[j]] = [a[j], a[i]];
   616→      }
   617→      return a;
   618→    };
   619→
   620→    // ---- Helper: Special subject exclusion rules (الدين/العربي) ----
   621→    // Rule 1: If exam subject is "الدين", teachers with subject "العربي" are excluded (الدين teachers already caught by own-subject)
   622→    // Rule 2: If exam subject is "العربي", teachers with subject "الدين" are excluded for primary stage only
   623→    const isSubjectExcluded = (t: Teacher, slot: Slot): boolean => {
   624→      if (!ruleSubject || !slot.subject) return false;
   625→      if (slot.subject === 'الدين') {
   626→        return t.subject === 'العربي';
   627→      }
   628→      if (slot.subject === 'العربي' && slot.stage === 'primary') {
   629→        return t.subject === 'الدين';
   630→      }
   631→      return false;
   632→    };
   633→
   634→    // ---- Core: Find best teacher for a slot ----
   635→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   636→    // Scoring: hours dominate (500x), tiny noise (0.1) for variety
   637→    const findBest = (
   638→      blockedId: string | null,
   639→      slot: Slot,
   640→      pool: Teacher[],
   641→      relaxAdj: boolean
   642→    ): Teacher | null => {
   643→      const candidates: { teacher: Teacher; score: number }[] = [];
   644→      for (const t of pool) {
   645→        if (blockedId && t.id === blockedId) continue;
   646→        const tr = tracking[t.id];
   647→        // HARD: No own-subject supervision
   648→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   649→        // HARD: Special subject exclusion (الدين/العربي rules)
   650→        if (isSubjectExcluded(t, slot)) continue;
   651→        // HARD: Stage notes filtering
   652→        if (!canSuperviseStage(t, slot.stage)) continue;
   653→        // HARD: No time overlap on same day
   654→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   655→        // HARD: MAX 2 committees per teacher per day (NEVER relaxed)
   656→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   657→        // SOFT: No same grade on consecutive days (can be relaxed)
   658→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   659→        // Scoring: HOURS dominate (500x), then committees (10x), tiny noise for variety
   660→        // Admin subject = HEAVY penalty (always last choice)
   661→        const isAdmin = t.subject === 'Admin';
   662→        const adminPenalty = isAdmin ? 50000 : 0;
   663→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   664→        candidates.push({ teacher: t, score });
   665→      }
   666→      if (candidates.length === 0) return null;
   667→      candidates.sort((a, b) => a.score - b.score);
   668→      const best = candidates[0].score;
   669→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   670→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   671→    };
   672→
   673→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 2-per-day + time) ----
   674→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   675→      const candidates: { teacher: Teacher; score: number }[] = [];
   676→      for (const t of pool) {
   677→        if (blockedId && t.id === blockedId) continue;
   678→        const tr = tracking[t.id];
   679→        // HARD: Still no own-subject supervision even in relaxed mode
   680→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   681→        // HARD: Special subject exclusion (الدين/العربي rules) — never relaxed
   682→        if (isSubjectExcluded(t, slot)) continue;
   683→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   684→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   685→        const isAdmin = t.subject === 'Admin';
   686→        const adminPenalty = isAdmin ? 50000 : 0;
   687→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   688→        candidates.push({ teacher: t, score });
   689→      }
   690→      if (candidates.length === 0) return null;
   691→      candidates.sort((a, b) => a.score - b.score);
   692→      return candidates[0].teacher;
   693→    };
   694→
   695→    // ---- Fallback 3: Allow 2nd assignment same day (time overlap + own-subject + max 2/day blocked) ----
   696→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   697→      const candidates: { teacher: Teacher; score: number }[] = [];
   698→      for (const t of pool) {
   699→        if (blockedId && t.id === blockedId) continue;
   700→        const tr = tracking[t.id];
   701→        // HARD: Still no own-subject supervision even in force-day mode
   702→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   703→        // HARD: Special subject exclusion (الدين/العربي rules) — never relaxed
   704→        if (isSubjectExcluded(t, slot)) continue;
   705→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   706→        // HARD: ABSOLUTE MAX 2 committees per teacher per day (never exceeded)
   707→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   708→        const isAdmin = t.subject === 'Admin';
   709→        const adminPenalty = isAdmin ? 50000 : 0;
   710→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (tr.dayComm[slot.day] || 0) * 100 + adminPenalty + Math.random() * 0.1;
   711→        candidates.push({ teacher: t, score });
   712→      }
   713→      if (candidates.length === 0) return null;
   714→      candidates.sort((a, b) => a.score - b.score);
   715→      return candidates[0].teacher;
   716→    };
   717→
   718→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   719→    let standbyCount = 0;
   720→    const nonAdminTeachers = teachers.filter(t => t.subject !== 'Admin');
   721→
   722→    // Calculate Week 2 peak days (days with above-average committee count)
   723→    const w2SlotCounts = WEEK2_DAYS.map(d => slotsByDay[d].length).filter(c => c > 0);
   724→    const w2Avg = w2SlotCounts.length > 0 ? w2SlotCounts.reduce((a, b) => a + b, 0) / w2SlotCounts.length : 0;
   725→    const peakW2Days = new Set(WEEK2_DAYS.filter(d => slotsByDay[d].length >= w2Avg && slotsByDay[d].length > 0));
   726→
   727→    for (const day of DAYS) {
   728→      // Determine pool: Admin only included on W2 peak days when toggle is ON
   729→      const isW2Peak = WEEK2_DAYS.includes(day) && peakW2Days.has(day);
   730→      const dayPool = (includeAdminW2 && isW2Peak) ? [...teachers] : nonAdminTeachers;
   731→      const daySlots = shuffle(slotsByDay[day]);
   732→      if (daySlots.length === 0) continue;
   733→
   734→      for (const slot of daySlots) {
   735→        let t1: Teacher | null = null;
   736→        let t2: Teacher | null = null;
   737→
   738→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   739→        t1 = findBest(null, slot, dayPool, false);
   740→        if (!t1) t1 = findBest(null, slot, dayPool, true); // relax consecutive-day
   741→        if (!t1) t1 = findBestRelaxed(null, slot, dayPool); // relax subject + stage
   742→        if (!t1) t1 = findBestForceDay(null, slot, dayPool); // allow 2nd assignment same day
   743→
   744→        if (t1) {
   745→          const tr = tracking[t1.id];
   746→          tr.totalComm++; tr.dayComm[slot.day]++;
   747→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   748→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   749→        } else { standbyCount++; }
   750→
   751→        // T2: pairing rule — classified teacher (has notes) must pair with unclassified
   752→        const blocked = t1?.id || null;
   753→        const t1Classified = t1 ? !!(t1.notes && t1.notes.trim() !== '') : false;
   754→        const t2Filtered = t1Classified ? dayPool.filter(t => t.id !== blocked && (!t.notes || t.notes.trim() === '')) : dayPool;
   755→
   756→        t2 = findBest(blocked, slot, t2Filtered, false);
   757→        if (!t2) t2 = findBest(blocked, slot, t2Filtered, true);
   758→        if (!t2) t2 = findBestRelaxed(blocked, slot, t2Filtered);
   759→        if (!t2) t2 = findBestForceDay(blocked, slot, t2Filtered);
   760→        // Relax pairing rule if no unclassified teacher available
   761→        if (!t2 && t1Classified) {
   762→          t2 = findBest(blocked, slot, dayPool, false);
   763→          if (!t2) t2 = findBest(blocked, slot, dayPool, true);
   764→          if (!t2) t2 = findBestRelaxed(blocked, slot, dayPool);
   765→          if (!t2) t2 = findBestForceDay(blocked, slot, dayPool);
   766→        }
   767→
   768→        if (t2) {
   769→          const tr = tracking[t2.id];
   770→          tr.totalComm++; tr.dayComm[slot.day]++;
   771→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   772→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   773→        } else { standbyCount++; }
   774→
   775→        // Record assignment
   776→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   777→        if (session) {
   778→          session.committees.push({
   779→            serial: slot.comId,
   780→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   781→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   782→          });
   783→        }
   784→      }
   785→    }
   786→
   787→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   788→    // This eliminates any tracking accumulation bugs
   789→    const teacherCommCount: Record<string, number> = {};
   790→    const teacherHoursFromAssign: Record<string, number> = {};
   791→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   792→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   793→    teachers.forEach(t => {
   794→      teacherCommCount[t.id] = 0;
   795→      teacherHoursFromAssign[t.id] = 0;
   796→      teacherDayCommFromAssign[t.id] = {};
   797→      teacherSlotsFromAssign[t.id] = [];
   798→    });
   799→
   800→    for (const day of DAYS) {
   801→      for (const sess of finalAssignments[day]) {
   802→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   803→        for (const c of sess.committees) {
   804→          [c.t1, c.t2].forEach(who => {
   805→            if (!who.id) return;
   806→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   807→            // Deduplicate: same day + same time = counted once
   808→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   809→            const existing = teacherSlotsFromAssign[who.id];
   810→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   811→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   812→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   813→            }
   814→            // Day committee count
   815→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   816→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   817→          });
   818→        }
   819→      }
   820→    }
   821→
   822→    // Override tracking with verified data from assignments
   823→    teachers.forEach(t => {
   824→      if (tracking[t.id]) {
   825→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   826→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   827→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   828→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   829→      }
   830→    });
   831→
   832→    // ---- Post-distribution: Admin already handled by pool exclusion ----
   833→
   834→    // ---- Post-distribution: BALANCE PASS ----
   835→    // Iteratively swap assignments from overburdened → underburdened teachers
   836→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   837→    const BALANCE_ITERATIONS = 500;
   838→    const recalcHours = () => {
   839→      teachers.forEach(t => {
   840→        const slots = tracking[t.id].assignedSlots || [];
   841→        const seen = new Set<string>();
   842→        let hrs = 0;
   843→        slots.forEach(s => {
   844→          const key = s.day + '_' + s.start + '_' + s.end;
   845→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   846→        });
   847→        tracking[t.id].totalHours = hrs;
   848→      });
   849→    };
   850→    recalcHours(); // recalc after admin swaps first
   851→
   852→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   853→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   854→      if (active.length < 2) break;
   855→      // Sort by HOURS first (primary), then committees (secondary)
   856→      const sorted = [...active].sort((a, b) => {
   857→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   858→        if (Math.abs(hDiff) > 0.1) return hDiff;
   859→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   860→      });
   861→      const maxH = tracking[sorted[0].id].totalHours;
   862→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   863→      if (maxH - minH <= 0.5) break; // spread ≤ 0.5h — tight balance
   864→
   865→      let swapped = false;
   866→      // Try top-5 donors × bottom-5 recipients for wider swap search
   867→      for (let di = 0; di < Math.min(sorted.length, 5) && !swapped; di++) {
   868→        const donor = sorted[di];
   869→        // Don't strip admins below 1 assignment in balance pass
   870→        const isDonorAdmin = donor.subject === 'Admin';
   871→        if (tracking[donor.id].totalComm <= 1) continue;
   872→
   873→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 5) && !swapped; ri--) {
   874→          const recip = sorted[ri];
   875→          if (donor.id === recip.id) continue;
   876→          // NEVER give admin more assignments during balance pass
   877→          if (recip.subject === 'Admin') continue;
   878→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   879→            // Also check if committees would be more balanced after swap
   880→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   881→            const recipCommAfter = tracking[recip.id].totalComm + 1;
   882→            // Skip if swap would make committee imbalance worse (donor already has fewer comms)
   883→
   884→          // Try each day where donor has an assignment and recipient doesn't
   885→          for (const day of DAYS) {
   886→            if (swapped) break;
   887→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   888→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   889→
   890→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   891→            if (!donorSlot) continue;
   892→
   893→            // Find the session matching donor's slot
   894→            for (const sess of finalAssignments[day]) {
   895→              if (swapped) break;
   896→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   897→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   898→
   899→              // Check recipient constraints
   900→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   901→              if (isSubjectExcluded(recip, { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   902→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   903→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   904→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   905→
   906→              // Find donor in this session's committees
   907→              for (const c of sess.committees) {
   908→                const isT1 = c.t1?.id === donor.id;
   909→                const isT2 = c.t2?.id === donor.id;
   910→                if (!isT1 && !isT2) continue;
   911→
   912→                // Execute swap
   913→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   914→                else c.t2 = { id: recip.id, name: recip.name };
   915→
   916→                tracking[donor.id].totalComm--;
   917→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   918→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   919→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   920→
   921→                tracking[recip.id].totalComm++;
   922→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   923→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   924→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   925→
   926→                swapped = true;
   927→                break;
   928→              }
   929→            }
   930→          }
   931→        }
   932→      }
   933→      if (!swapped) break; // no valid swap found
   934→      recalcHours(); // update hours after each swap
   935→    }
   936→
   937→    // Final hour recalculation after all balancing
   938→    recalcHours();
   939→
   940→    // ---- Sort committees by serial within each session (1, 2, 3...) ----
   941→    for (const day of DAYS) {
   942→      for (const sess of finalAssignments[day]) {
   943→        sess.committees.sort((a, b) => a.serial - b.serial);
   944→      }
   945→    }
   946→
   947→    // ---- Post-distribution: STANDBY ASSIGNMENT ----
   948→    // 1 standby teacher per stage per day from unassigned teachers
   949→    // CRITICAL: Standby must NOT be the same subject as any exam that day for that stage
   950→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   951→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   952→    const STANDBY_PER_STAGE = 1;
   953→    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
   954→    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
   955→
   956→    // Track standby count per teacher for rotation fairness
   957→    const teacherStandbyCount: Record<string, number> = {};
   958→    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
   959→
   960→    // Track which teachers are already standby for another stage today
   961→    const todayStandby: Record<string, Set<string>> = {};
   962→    DAYS.forEach(d => { todayStandby[d] = new Set(); });
   963→
   964→    for (let di = 0; di < DAYS.length; di++) {
   965→      const day = DAYS[di];
   966→
   967→      // Determine pool for standby (same logic as distribution)
   968→      const isW2Peak = WEEK2_DAYS.includes(day) && peakW2Days.has(day);
   969→      const dayStandbyPool = (includeAdminW2 && isW2Peak) ? [...teachers] : nonAdminTeachers;
   970→
   971→      for (const stage of STAGES_LIST) {
   972→        // Only assign standby if there are exams for this stage on this day
   973→        const stageSessions = (finalAssignments[day] || []).filter(s => getStage(s.grade) === stage);
   974→        if (stageSessions.length === 0) continue;
   975→
   976→        // Collect subjects being examined today for this stage
   977→        const dayStageSubjects = new Set<string>();
   978→        stageSessions.forEach(s => { if (s.subject) dayStageSubjects.add(s.subject); });
   979→
   980→        // Find teachers NOT assigned on this day who can supervise this stage
   981→        // AND whose subject doesn't match any exam subject for this stage today
   982→        let candidates = dayStandbyPool.filter(t => {
   983→          if ((tracking[t.id].dayComm[day] || 0) >= 2) return false;
