    43тЖТ    43тЖТfunction parseTimeRange(tStr: string) {
    44тЖТ    44тЖТ  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    45тЖТ    45тЖТ  try {
    46тЖТ    46тЖТ    let clean = tStr.replace(/\s+/g, '').replace(/тАФ/g, '-').replace(/тАУ/g, '-');
    47тЖТ    47тЖТ    const parts = clean.split('-');
    48тЖТ    48тЖТ    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    49тЖТ    49тЖТ    let start = toMinutes(parts[0]);
    50тЖТ    50тЖТ    let end = toMinutes(parts[1]);
    51тЖТ    51тЖТ    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    52тЖТ    52тЖТ    if (start < 300 && end > start + 120) start += 720;
    53тЖТ    53тЖТ    if (end <= start) end += 720;
    54тЖТ    54тЖТ    let duration = (end - start) / 60;
    55тЖТ    55тЖТ    // Sanity cap: no exam is longer than 5 hours
    56тЖТ    56тЖТ    if (duration > 5) duration = 2.0; // fallback to standard
    57тЖТ    57тЖТ    return { start, end, duration };
    58тЖТ    58тЖТ  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    59тЖТ    59тЖТ}

   503тЖТ   503тЖТ    // ---- Helper: Fisher-Yates shuffle ----
   504тЖТ   504тЖТ    const shuffle = <T>(arr: T[]): T[] => {
   505тЖТ   505тЖТ      const a = [...arr];
   506тЖТ   506тЖТ      for (let i = a.length - 1; i > 0; i--) {
   507тЖТ   507тЖТ        const j = Math.floor(Math.random() * (i + 1));
   508тЖТ   508тЖТ        [a[i], a[j]] = [a[j], a[i]];
   509тЖТ   509тЖТ      }
   510тЖТ   510тЖТ      return a;
   511тЖТ   511тЖТ    };

   514тЖТ   514тЖТ    const adminAssignmentCount: Record<string, number> = {};
   515тЖТ   515тЖТ    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   516тЖТ   516тЖТ    const adminMinTarget = 2; // each admin gets at least 2 assignments
   517тЖТ   517тЖТ    let totalAdminAssigned = 0;
   518тЖТ   518тЖТ
   519тЖТ   519тЖТ    // ---- Core: Find best teacher for a slot ----
   520тЖТ   520тЖТ    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   521тЖТ   521тЖТ    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   522тЖТ   522тЖТ    const findBest = (
   523тЖТ   523тЖТ      blockedId: string | null,
   524тЖТ   524тЖТ      slot: Slot,
   525тЖТ   525тЖТ      pool: Teacher[],
   526тЖТ   526тЖТ      relaxAdj: boolean
   527тЖТ   527тЖТ    ): Teacher | null => {
   528тЖТ   528тЖТ      const candidates: { teacher: Teacher; score: number }[] = [];
   529тЖТ   529тЖТ      for (const t of pool) {
   530тЖТ   530тЖТ        if (blockedId && t.id === blockedId) continue;
   531тЖТ   531тЖТ        const tr = tracking[t.id];
   532тЖТ   532тЖТ        // HARD: No own-subject supervision
   533тЖТ   533тЖТ        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   534тЖТ   534тЖТ        // HARD: Stage notes filtering
   535тЖТ   535тЖТ        if (!canSuperviseStage(t, slot.stage)) continue;
   536тЖТ   536тЖТ        // HARD: No time overlap on same day
   537тЖТ   537тЖТ        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   538тЖТ   538тЖТ        // HARD: MAX 1 committee per teacher per day (NEVER relaxed)
   539тЖТ   539тЖТ        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   540тЖТ   540тЖТ        // SOFT: No same grade on consecutive days (can be relaxed)
   541тЖТ   541тЖТ        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   542тЖТ   542тЖТ        // Scoring: hours are king, tiny random for variety between equal candidates
   543тЖТ   543тЖТ        const isAdmin = t.subject === 'Admin';
   544тЖТ   544тЖТ        const adminBonus = (adminAssignmentCount[t.id] || 0) < adminMinTarget ? -3 : 0;
   545тЖТ   545тЖТ        const score = tr.totalHours * 200 + tr.totalComm * 5 + (isAdmin ? 3 + adminBonus : 0) + Math.random() * 0.2;
   546тЖТ   546тЖТ        candidates.push({ teacher: t, score });
   547тЖТ   547тЖТ      }
   548тЖТ   548тЖТ      if (candidates.length === 0) return null;
   549тЖТ   549тЖТ      candidates.sort((a, b) => a.score - b.score);
   550тЖТ   550тЖТ      const best = candidates[0].score;
   551тЖТ   551тЖТ      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   552тЖТ   552тЖТ      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   553тЖТ   553тЖТ    };
   554тЖТ   554тЖТ
   555тЖТ   555тЖТ    // ---- Fallback 2: Relax subject + stage constraints (keep 1-per-day + time) ----
   556тЖТ   556тЖТ    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {

   563тЖТ   563тЖТ        const score = tr.totalHours * 200 + tr.totalComm * 5 + Math.random() * 0.2;
   564тЖТ   564тЖТ        candidates.push({ teacher: t, score });
   565тЖТ   565тЖТ      }
   566тЖТ   566тЖТ      if (candidates.length === 0) return null;
   567тЖТ   567тЖТ      candidates.sort((a, b) => a.score - b.score);
   568тЖТ   568тЖТ      return candidates[0].teacher;
   569тЖТ   569тЖТ    };
   570тЖТ   570тЖТ
   571тЖТ   571тЖТ    // ---- Fallback 3: Allow 2nd assignment same day (only time overlap blocked) ----
   572тЖТ   572тЖТ    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   573тЖТ   573тЖТ      const candidates: { teacher: Teacher; score: number }[] = [];
   574тЖТ   574тЖТ      for (const t of pool) {
   575тЖТ   575тЖТ        if (blockedId && t.id === blockedId) continue;
   576тЖТ   576тЖТ        const tr = tracking[t.id];
   577тЖТ   577тЖТ        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   578тЖТ   578тЖТ        const score = tr.totalHours * 200 + tr.totalComm * 5 + (tr.dayComm[slot.day] || 0) * 50 + Math.random() * 0.2;
   579тЖТ   579тЖТ        candidates.push({ teacher: t, score });
   580тЖТ   580тЖТ      }
   581тЖТ   581тЖТ      if (candidates.length === 0) return null;
   582тЖТ   582тЖТ      candidates.sort((a, b) => a.score - b.score);
   583тЖТ   583тЖТ      return candidates[0].teacher;

   585тЖТ   585тЖТ
   586тЖТ   586тЖТ    // ---- Process each day chronologically, shuffle slots within day ----
   587тЖТ   587тЖТ    let standbyCount = 0;
   588тЖТ   588тЖТ    const allPool = [...teachers]; // ALL teachers including admin
   589тЖТ   589тЖТ
   590тЖТ   590тЖТ    for (const day of DAYS) {
   591тЖТ   591тЖТ      const daySlots = shuffle(slotsByDay[day]);
   592тЖТ   592тЖТ      if (daySlots.length === 0) continue;
   593тЖТ   593тЖТ
   594тЖТ   594тЖТ      for (const slot of daySlots) {
   595тЖТ   595тЖТ        let t1: Teacher | null = null;
   596тЖТ   596тЖТ        let t2: Teacher | null = null;
   597тЖТ   597тЖТ
   598тЖТ   598тЖТ        // T1: strict тЖТ relax consecutive тЖТ relax subject+stage тЖТ allow 2nd same-day
   599тЖТ   599тЖТ        t1 = findBest(null, slot, allPool, false);
   600тЖТ   600тЖТ        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   601тЖТ   601тЖТ        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   602тЖТ   602тЖТ        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   603тЖТ   603тЖТ
   604тЖТ   604тЖТ        if (t1) {
   605тЖТ   605тЖТ          const tr = tracking[t1.id];
   606тЖТ   606тЖТ          tr.totalComm++; tr.dayComm[slot.day]++;

   609тЖТ   609тЖТ        } else { standbyCount++; }

   623тЖТ   623тЖТ          const tr = tracking[t2.id];
   624тЖТ   624тЖТ          tr.totalComm++; tr.dayComm[slot.day]++;
   625тЖТ   625тЖТ          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   626тЖТ   626тЖТ          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   627тЖТ   627тЖТ        } else { standbyCount++; }
   628тЖТ   628тЖТ
   629тЖТ   629тЖТ        // Record assignment
   630тЖТ   630тЖТ        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   631тЖТ   631тЖТ        if (session) {
   632тЖТ   632тЖТ          session.committees.push({
   633тЖТ   633тЖТ            serial: slot.comId,
   634тЖТ   634тЖТ            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: 'тАФ' },

   640тЖТ   640тЖТ
   641тЖТ   641тЖТ    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   642тЖТ   642тЖТ    // This eliminates any tracking accumulation bugs
   643тЖТ   643тЖТ    const teacherCommCount: Record<string, number> = {};
   644тЖТ   644тЖТ    const teacherHoursFromAssign: Record<string, number> = {};
   645тЖТ   645тЖТ    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   646тЖТ   646тЖТ    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};

   648тЖТ   648тЖТ      teacherCommCount[t.id] = 0;
   649тЖТ   649тЖТ      teacherHoursFromAssign[t.id] = 0;
   650тЖТ   650тЖТ      teacherDayCommFromAssign[t.id] = {};
   651тЖТ   651тЖТ      teacherSlotsFromAssign[t.id] = [];
   652тЖТ   652тЖТ    });
   653тЖТ   653тЖТ
   654тЖТ   654тЖТ    for (const day of DAYS) {
   655тЖТ   655тЖТ      for (const sess of finalAssignments[day]) {
   656тЖТ   656тЖТ        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   657тЖТ   657тЖТ        for (const c of sess.committees) {
   658тЖТ   658тЖТ          [c.t1, c.t2].forEach(who => {
   659тЖТ   659тЖТ            if (!who.id) return;
   660тЖТ   660тЖТ            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   661тЖТ   661тЖТ            // Deduplicate: same day + same time = counted once
   662тЖТ   662тЖТ            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   663тЖТ   663тЖТ            const existing = teacherSlotsFromAssign[who.id];
   664тЖТ   664тЖТ            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   665тЖТ   665тЖТ              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   666тЖТ   666тЖТ              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   667тЖТ   667тЖТ            }
   668тЖТ   668тЖТ            // Day committee count
   669тЖТ   669тЖТ            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   670тЖТ   670тЖТ            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   671тЖТ   671тЖТ          });
   672тЖТ   672тЖТ        }
   673тЖТ   673тЖТ      }
   674тЖТ   674тЖТ    }
   675тЖТ   675тЖТ
   676тЖТ   676тЖТ    // Override tracking with verified data from assignments
   677тЖТ   677тЖТ    teachers.forEach(t => {
   678тЖТ   678тЖТ      if (tracking[t.id]) {
   679тЖТ   679тЖТ        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   680тЖТ   680тЖТ        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   681тЖТ   681тЖТ        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   682тЖТ   682тЖТ        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   683тЖТ   683тЖТ      }
   684тЖТ   684тЖТ    });
   685тЖТ   685тЖТ
   686тЖТ   686тЖТ    // ---- Post-distribution: Ensure admin gets at least 2 assignments ----
   687тЖТ   687тЖТ    for (const admin of adminTeachers) {
   688тЖТ   688тЖТ      const adminComm = tracking[admin.id].totalComm;
   689тЖТ   689тЖТ      if (adminComm >= adminMinTarget) continue;
   690тЖТ   690тЖТ      // Try to swap: find a teacher with > average hours who has an assignment
   691тЖТ   691тЖТ      // that admin could take (passing all constraints)
   692тЖТ   692тЖТ      const allTeacherHours = teachers.map(t => tracking[t.id]?.totalHours || 0).filter(h => h > 0);
   693тЖТ   693тЖТ      const avgH = allTeacherHours.length > 0 ? allTeacherHours.reduce((a, b) => a + b, 0) / allTeacherHours.length : 0;
   694тЖТ   694тЖТ      const overburdened = teachers.filter(t => t.id !== admin.id && tracking[t.id]?.totalHours > avgH);
   695тЖТ   695тЖТ      // Sort by most overburdened first
   696тЖТ   696тЖТ      overburdened.sort((a, b) => (tracking[b.id]?.totalHours || 0) - (tracking[a.id]?.totalHours || 0));
   697тЖТ   697тЖТ
   698тЖТ   698тЖТ      for (const victim of overburdened) {
   699тЖТ   699тЖТ        if (tracking[admin.id].totalComm >= adminMinTarget) break;
   700тЖТ   700тЖТ        // Find an assignment of victim that admin can take
   701тЖТ   701тЖТ        for (const day of DAYS) {
   702тЖТ   702тЖТ          if (tracking[admin.id].totalComm >= adminMinTarget) break;
   703тЖТ   703тЖТ          const daySlots = teacherSlotsFromAssign[victim.id].filter(s => s.day === day);
   704тЖТ   704тЖТ          if (daySlots.length === 0) continue;
   705тЖТ   705тЖТ          // Admin already has assignment on this day?
   706тЖТ   706тЖТ          if ((tracking[admin.id].dayComm[day] || 0) >= 1) continue;
   707тЖТ   707тЖТ          // Find the session+committee for this slot
   708тЖТ   708тЖТ          for (const sess of finalAssignments[day]) {
   709тЖТ   709тЖТ            if (tracking[admin.id].totalComm >= adminMinTarget) break;
   710тЖТ   710тЖТ            const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   711тЖТ   711тЖТ            if (daySlots[0].start !== sessTimeInfo.start || daySlots[0].end !== sessTimeInfo.end) continue;
   712тЖТ   712тЖТ            // Check admin constraints
   713тЖТ   713тЖТ            if (ruleSubject && sess.subject && admin.subject === sess.subject) continue;
   714тЖТ   714тЖТ            if (!canSuperviseStage(admin, getStage(sess.grade))) continue;
   715тЖТ   715тЖТ            // Check admin no time overlap (should be fine since we checked dayComm above)
   716тЖТ   716тЖТ            if (tracking[admin.id].assignedSlots.some(s => s.day === day && !(sessTimeInfo.end <= s.start || sessTimeInfo.start >= s.end))) continue;
   717тЖТ   717тЖТ            // Find which position (t1 or t2) victim has in this session
   718тЖТ   718тЖТ            for (const c of sess.committees) {
   719тЖТ   719тЖТ              if (c.t1?.id === victim.id || c.t2?.id === victim.id) {
   720тЖТ   720тЖТ                // Check consecutive grade constraint
   721тЖТ   721тЖТ                if (wasSameGradeAdjacent(tracking[admin.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTimeInfo, comId: 0 })) continue;

   723тЖТ   723тЖТ                const isT1 = c.t1?.id === victim.id;

   725тЖТ   725тЖТ                if (isT1) { c.t1 = { id: admin.id, name: admin.name }; }
   726тЖТ   726тЖТ                else { c.t2 = { id: admin.id, name: admin.name }; }
   727тЖТ   727тЖТ                // Update tracking: remove from victim, add to admin
   728тЖТ   728тЖТ                tracking[victim.id].totalComm--;
   729тЖТ   729тЖТ                tracking[victim.id].dayComm[day] = Math.max(0, (tracking[victim.id].dayComm[day] || 0) - 1);
   730тЖТ   730тЖТ                tracking[admin.id].totalComm++;
   731тЖТ   731тЖТ                tracking[admin.id].dayComm[day] = (tracking[admin.id].dayComm[day] || 0) + 1;
   732тЖТ   732тЖТ                tracking[admin.id].assignedSlots.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   733тЖТ   733тЖТ                tracking[admin.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   734тЖТ   734тЖТ                break;
   735тЖТ   735тЖТ              }
   736тЖТ   736тЖТ            }
   737тЖТ   737тЖТ          }
   738тЖТ   738тЖТ        }
   739тЖТ   739тЖТ      }
   740тЖТ   740тЖТ    }
   741тЖТ   741тЖТ
   742тЖТ   742тЖТ    // ---- Post-distribution: BALANCE PASS ----
   743тЖТ   743тЖТ    // Iteratively swap assignments from overburdened тЖТ underburdened teachers
   744тЖТ   744тЖТ    // to minimize the max-min spread in hours
   745тЖТ   745тЖТ    const BALANCE_ITERATIONS = 150;
   746тЖТ   746тЖТ    const recalcHours = () => {
   747тЖТ   747тЖТ      teachers.forEach(t => {
   748тЖТ   748тЖТ        const slots = tracking[t.id].assignedSlots || [];
   749тЖТ   749тЖТ        const seen = new Set<string>();
   750тЖТ   750тЖТ        let hrs = 0;
   751тЖТ   751тЖТ        slots.forEach(s => {
   752тЖТ   752тЖТ          const key = s.day + '_' + s.start + '_' + s.end;
   753тЖТ   753тЖТ          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   754тЖТ   754тЖТ        });
   755тЖТ   755тЖТ        tracking[t.id].totalHours = hrs;
   756тЖТ   756тЖТ      });
   757тЖТ   757тЖТ    };
   758тЖТ   758тЖТ    recalcHours(); // recalc after admin swaps first
   759тЖТ   759тЖТ
   760тЖТ   760тЖТ    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   761тЖТ   761тЖТ      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   762тЖТ   762тЖТ      if (active.length < 2) break;
   763тЖТ   763тЖТ      const sorted = [...active].sort((a, b) => tracking[b.id].totalHours - tracking[a.id].totalHours);
   764тЖТ   764тЖТ      const maxH = tracking[sorted[0].id].totalHours;
   765тЖТ   765тЖТ      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   766тЖТ   766тЖТ      if (maxH - minH <= 1.0) break; // spread тЙд 1h тАФ good enough
   767тЖТ   767тЖТ
   768тЖТ   768тЖТ      let swapped = false;
   769тЖТ   769тЖТ      // Try top-3 donors ├Ч bottom-3 recipients
   770тЖТ   770тЖТ      for (let di = 0; di < Math.min(sorted.length, 3) && !swapped; di++) {
   771тЖТ   771тЖТ        const donor = sorted[di];
   772тЖТ   772тЖТ        // Don't strip admins below their minimum
   773тЖТ   773тЖТ        const isDonorAdmin = donor.subject === 'Admin';
   774тЖТ   774тЖТ        if (tracking[donor.id].totalComm <= 1) continue;
   775тЖТ   775тЖТ        if (isDonorAdmin && tracking[donor.id].totalComm <= adminMinTarget) continue;
   776тЖТ   776тЖТ
   777тЖТ   777тЖТ        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 3) && !swapped; ri--) {
   778тЖТ   778тЖТ          const recip = sorted[ri];
   779тЖТ   779тЖТ          if (donor.id === recip.id) continue;
   780тЖТ   780тЖТ          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   781тЖТ   781тЖТ
   782тЖТ   782тЖТ          // Try each day where donor has an assignment and recipient doesn't
   783тЖТ   783тЖТ          for (const day of DAYS) {
   784тЖТ   784тЖТ            if (swapped) break;
   785тЖТ   785тЖТ            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   786тЖТ   786тЖТ            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   787тЖТ   787тЖТ
   788тЖТ   788тЖТ            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   789тЖТ   789тЖТ            if (!donorSlot) continue;
   790тЖТ   790тЖТ
   791тЖТ   791тЖТ            // Find the session matching donor's slot
   792тЖТ   792тЖТ            for (const sess of finalAssignments[day]) {
   793тЖТ   793тЖТ              if (swapped) break;
   794тЖТ   794тЖТ              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   795тЖТ   795тЖТ              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   796тЖТ   796тЖТ
   797тЖТ   797тЖТ              // Check recipient constraints
   798тЖТ   798тЖТ              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   799тЖТ   799тЖТ              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   800тЖТ   800тЖТ              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   801тЖТ   801тЖТ              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   802тЖТ   802тЖТ
   803тЖТ   803тЖТ              // Find donor in this session's committees
   804тЖТ   804тЖТ              for (const c of sess.committees) {
   805тЖТ   805тЖТ                const isT1 = c.t1?.id === donor.id;
   806тЖТ   806тЖТ                const isT2 = c.t2?.id === donor.id;
   807тЖТ   807тЖТ                if (!isT1 && !isT2) continue;
   808тЖТ   808тЖТ
   809тЖТ   809тЖТ                // Execute swap
   810тЖТ   810тЖТ                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   811тЖТ   811тЖТ                else c.t2 = { id: recip.id, name: recip.name };
   812тЖТ   812тЖТ
   813тЖТ   813тЖТ                tracking[donor.id].totalComm--;
   814тЖТ   814тЖТ                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);

   817тЖТ   817тЖТ
   818тЖТ   818тЖТ                tracking[recip.id].totalComm++;
   819тЖТ   819тЖТ                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   820тЖТ   820тЖТ                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   821тЖТ   821тЖТ                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   822тЖТ   822тЖТ
   823тЖТ   823тЖТ                swapped = true;
   824тЖТ   824тЖТ                break;
   825тЖТ   825тЖТ              }
   826тЖТ   826тЖТ            }
   827тЖТ   827тЖТ          }
   828тЖТ   828тЖТ        }
   829тЖТ   829тЖТ      }
   830тЖТ   830тЖТ      if (!swapped) break; // no valid swap found
   831тЖТ   831тЖТ      recalcHours(); // update hours after each swap
   832тЖТ   832тЖТ    }
   833тЖТ   833тЖТ
   834тЖТ   834тЖТ    // Final hour recalculation after all balancing
   835тЖТ   835тЖТ    recalcHours();
   836тЖТ   836тЖТ
   837тЖТ   837тЖТ    // ---- Post-distribution: STANDBY ASSIGNMENT ----
   838тЖТ   838тЖТ    // 2 standby teachers per stage per day from unassigned teachers
   839тЖТ   839тЖТ    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   840тЖТ   840тЖТ    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   841тЖТ   841тЖТ    const STANDBY_PER_STAGE = 2;
   842тЖТ   842тЖТ    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
   843тЖТ   843тЖТ    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
   844тЖТ   844тЖТ
   845тЖТ   845тЖТ    // Track standby count per teacher for rotation fairness
   846тЖТ   846тЖТ    const teacherStandbyCount: Record<string, number> = {};
   847тЖТ   847тЖТ    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
   848тЖТ   848тЖТ
   849тЖТ   849тЖТ    for (let di = 0; di < DAYS.length; di++) {
   850тЖТ   850тЖТ      const day = DAYS[di];
   851тЖТ   851тЖТ
   852тЖТ   852тЖТ      for (const stage of STAGES_LIST) {
   853тЖТ   853тЖТ        // Only assign standby if there are exams for this stage on this day
   854тЖТ   854тЖТ        const stageSessions = (finalAssignments[day] || []).filter(s => getStage(s.grade) === stage);
   855тЖТ   855тЖТ        if (stageSessions.length === 0) continue;
   856тЖТ   856тЖТ
   857тЖТ   857тЖТ        // Find teachers NOT assigned on this day who can supervise this stage
   858тЖТ   858тЖТ        let candidates = teachers.filter(t => {
   859тЖТ   859тЖТ          if ((tracking[t.id].dayComm[day] || 0) >= 1) return false;
   860тЖТ   860тЖТ          if (!canSuperviseStage(t, stage)) return false;
   861тЖТ   861тЖТ          return true;
   862тЖТ   862тЖТ        });
   863тЖТ   863тЖТ
   864тЖТ   864тЖТ        // Rotation: deprioritize teachers who were standby YESTERDAY for ANY stage
   865тЖТ   865тЖТ        const yesterdayStandby = new Set<string>();
   866тЖТ   866тЖТ        if (di > 0) {
   867тЖТ   867тЖТ          const prevDay = DAYS[di - 1];
   868тЖТ   868тЖТ          for (const st of STAGES_LIST) {
   869тЖТ   869тЖТ            (standbys[prevDay][st] || []).forEach(s => yesterdayStandby.add(s.id));
   870тЖТ   870тЖТ          }
   871тЖТ   871тЖТ        }
   872тЖТ   872тЖТ        candidates.sort((a, b) => {
   873тЖТ   873тЖТ          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
   874тЖТ   874тЖТ          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
   875тЖТ   875тЖТ          return (teacherStandbyCount[a.id] + aWasYesterday) - (teacherStandbyCount[b.id] + bWasYesterday);
   876тЖТ   876тЖТ        });
   877тЖТ   877тЖТ
   878тЖТ   878тЖТ        // Pick up to STANDBY_PER_STAGE
   879тЖТ   879тЖТ        const picked = candidates.slice(0, STANDBY_PER_STAGE);
   880тЖТ   880тЖТ        standbys[day][stage] = picked.map(t => ({ id: t.id, name: t.name }));
   881тЖТ   881тЖТ
   882тЖТ   882тЖТ        // Only track rotation count тАФ do NOT update tracking (standby is informational)
   883тЖТ   883тЖТ        for (const s of picked) {
   884тЖТ   884тЖТ          teacherStandbyCount[s.id]++;
