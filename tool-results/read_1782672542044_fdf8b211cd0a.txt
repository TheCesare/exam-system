    45→function parseTimeRange(tStr: string) {
    46→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    47→  try {
    48→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    49→    const parts = clean.split('-');
    50→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    51→    let start = toMinutes(parts[0]);
    52→    let end = toMinutes(parts[1]);
    53→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    54→    if (start < 300 && end > start + 120) start += 720;
    55→    if (end <= start) end += 720;
    56→    let duration = (end - start) / 60;
    57→    // Sanity cap: no exam is longer than 5 hours
    58→    if (duration > 5) duration = 2.0; // fallback to standard
    59→    return { start, end, duration };
    60→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    61→}

    63→function getStage(grade: string) {
    64→  const g = grade.toLowerCase();
    65→  if (g.includes('primary')) return 'primary';
    66→  if (g.includes('prep')) return 'prep';
    67→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    68→  return 'any';
    69→}

   494→      loadSchedule();
   495→    } catch { showToast('Error saving schedule', 'error'); }
   496→  };

   500→    setScheduleBuffer({});

   506→    } catch { /* silent */ }
   507→  };
   508→
   509→  // ========== DISTRIBUTION ENGINE v4 ==========
   510→  const runDistribution = () => {
   511→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   512→
   513→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   514→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   515→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   516→
   517→    // ---- All teachers in ONE pool (admin participates normally) ----
   518→    const adminTeachers = teachers.filter(t => t.subject === 'Admin');
   519→
   520→    // ---- Initialize tracking ----
   521→    const tracking: Record<string, TrackingEntry> = {};
   522→    teachers.forEach(t => {
   523→      tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {} as Record<string,number>, assignedSlots: [], gradeHistory: [] };
   524→      DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   525→    });
   526→
   527→    // ---- Build exam slots grouped by day (chronological order) ----
   528→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   529→    const slotsByDay: Record<string, Slot[]> = {};
   530→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   531→    DAYS.forEach((day, dayIndex) => {
   532→      GRADES.forEach(grade => {
   533→        const cell = scheduleBuffer[`${grade}__${day}`];
   534→        if (cell && cell.committees > 0) {
   535→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   536→          for (let c = 1; c <= cell.committees; c++) {
   537→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   538→          }
   539→        }
   540→      });
   541→    });
   542→
   543→    // ---- Build final assignments structure ----
   544→    const finalAssignments: Record<string, SessionResult[]> = {};
   545→    DAYS.forEach(d => { finalAssignments[d] = []; });
   546→    DAYS.forEach(day => {
   547→      GRADES.forEach(grade => {
   548→        const cell = scheduleBuffer[`${grade}__${day}`];
   549→        if (cell && cell.committees > 0) {
   550→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   551→          if (!existing) {
   552→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   553→          }
   554→        }
   555→      });
   556→    });
   557→
   558→    // ---- Helper: Stage notes — HARD constraint ----
   559→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   560→      if (!ruleNotes) return true;
   561→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   562→      const n = teacher.notes.toLowerCase();
   563→      const allowed: string[] = [];
   564→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   565→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   566→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   567→      if (allowed.length === 0) return true;
   568→      return allowed.includes(slotStage);
   569→    };
   570→
   571→    // ---- Helper: Consecutive-day same-grade check ----
   572→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   573→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   574→    };
   575→
   576→    // ---- Helper: Fisher-Yates shuffle ----
   577→    const shuffle = <T,>(arr: T[]): T[] => {
   578→      const a = [...arr];
   579→      for (let i = a.length - 1; i > 0; i--) {
   580→        const j = Math.floor(Math.random() * (i + 1));
   581→        [a[i], a[j]] = [a[j], a[i]];
   582→      }
   583→      return a;
   584→    };
   585→
   586→    // ---- Admin minimum assignment tracking ----
   587→    const adminAssignmentCount: Record<string, number> = {};
   588→    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   589→    const adminMinTarget = 2; // each admin gets at least 2 assignments
   590→    let totalAdminAssigned = 0;
   591→
   592→    // ---- Core: Find best teacher for a slot ----
   593→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   594→    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   595→    const findBest = (
   596→      blockedId: string | null,
   597→      slot: Slot,
   598→      pool: Teacher[],
   599→      relaxAdj: boolean
   600→    ): Teacher | null => {
   601→      const candidates: { teacher: Teacher; score: number }[] = [];
   602→      for (const t of pool) {
   603→        if (blockedId && t.id === blockedId) continue;
   604→        const tr = tracking[t.id];
   605→        // HARD: No own-subject supervision
   606→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   607→        // HARD: Stage notes filtering
   608→        if (!canSuperviseStage(t, slot.stage)) continue;
   609→        // HARD: No time overlap on same day
   610→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   611→        // HARD: MAX 1 committee per teacher per day (NEVER relaxed)
   612→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   613→        // SOFT: No same grade on consecutive days (can be relaxed)
   614→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   615→        // Scoring: HOURS dominate (500x), then committees (10x), tiny noise for variety
   616→        const isAdmin = t.subject === 'Admin';
   617→        const adminBonus = (adminAssignmentCount[t.id] || 0) < adminMinTarget ? -5 : 0;
   618→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (isAdmin ? 3 + adminBonus : 0) + Math.random() * 0.1;
   619→        candidates.push({ teacher: t, score });
   620→      }
   621→      if (candidates.length === 0) return null;
   622→      candidates.sort((a, b) => a.score - b.score);
   623→      const best = candidates[0].score;
   624→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   625→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   626→    };
   627→
   628→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 1-per-day + time) ----
   629→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   630→      const candidates: { teacher: Teacher; score: number }[] = [];
   631→      for (const t of pool) {
   632→        if (blockedId && t.id === blockedId) continue;
   633→        const tr = tracking[t.id];
   634→        // HARD: Still no own-subject supervision even in relaxed mode
   635→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   636→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   637→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   638→        const score = tr.totalHours * 500 + tr.totalComm * 10 + Math.random() * 0.1;
   639→        candidates.push({ teacher: t, score });
   640→      }
   641→      if (candidates.length === 0) return null;
   642→      candidates.sort((a, b) => a.score - b.score);
   643→      return candidates[0].teacher;
   644→    };
   645→
   646→    // ---- Fallback 3: Allow 2nd assignment same day (only time overlap + own-subject blocked) ----
   647→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   648→      const candidates: { teacher: Teacher; score: number }[] = [];
   649→      for (const t of pool) {
   650→        if (blockedId && t.id === blockedId) continue;
   651→        const tr = tracking[t.id];
   652→        // HARD: Still no own-subject supervision even in force-day mode
   653→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   654→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   655→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (tr.dayComm[slot.day] || 0) * 100 + Math.random() * 0.1;
   656→        candidates.push({ teacher: t, score });
   657→      }
   658→      if (candidates.length === 0) return null;
   659→      candidates.sort((a, b) => a.score - b.score);
   660→      return candidates[0].teacher;
   661→    };
   662→
   663→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   664→    let standbyCount = 0;
   665→    const allPool = [...teachers]; // ALL teachers including admin
   666→
   667→    for (const day of DAYS) {
   668→      const daySlots = shuffle(slotsByDay[day]);
   669→      if (daySlots.length === 0) continue;
   670→
   671→      for (const slot of daySlots) {
   672→        let t1: Teacher | null = null;
   673→        let t2: Teacher | null = null;
   674→
   675→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   676→        t1 = findBest(null, slot, allPool, false);
   677→        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   678→        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   679→        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   680→
   681→        if (t1) {
   682→          const tr = tracking[t1.id];
   683→          tr.totalComm++; tr.dayComm[slot.day]++;
   684→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   685→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   686→        } else { standbyCount++; }
   687→
   688→        // T2: same pool, blocked=T1, cascade fallbacks
   689→        const blocked = t1?.id || null;
   690→        t2 = findBest(blocked, slot, allPool, false);
   691→        if (!t2) t2 = findBest(blocked, slot, allPool, true);
   692→        if (!t2) t2 = findBestRelaxed(blocked, slot, allPool);
   693→        if (!t2) t2 = findBestForceDay(blocked, slot, allPool);
   694→
   695→        if (t2) {
   696→          if (t2.subject === 'Admin') {
   697→            adminAssignmentCount[t2.id] = (adminAssignmentCount[t2.id] || 0) + 1;
   698→            totalAdminAssigned++;
   699→          }
   700→          const tr = tracking[t2.id];
   701→          tr.totalComm++; tr.dayComm[slot.day]++;
   702→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   703→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   704→        } else { standbyCount++; }
   705→
   706→        // Record assignment
   707→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   708→        if (session) {
   709→          session.committees.push({
   710→            serial: slot.comId,
   711→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   712→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   713→          });
   714→        }
   715→      }
   716→    }
   717→
   718→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   719→    // This eliminates any tracking accumulation bugs
   720→    const teacherCommCount: Record<string, number> = {};
   721→    const teacherHoursFromAssign: Record<string, number> = {};
   722→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   723→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   724→    teachers.forEach(t => {
   725→      teacherCommCount[t.id] = 0;
   726→      teacherHoursFromAssign[t.id] = 0;
   727→      teacherDayCommFromAssign[t.id] = {};
   728→      teacherSlotsFromAssign[t.id] = [];
   729→    });
   730→
   731→    for (const day of DAYS) {
   732→      for (const sess of finalAssignments[day]) {
   733→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   734→        for (const c of sess.committees) {
   735→          [c.t1, c.t2].forEach(who => {
   736→            if (!who.id) return;
   737→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   738→            // Deduplicate: same day + same time = counted once
   739→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   740→            const existing = teacherSlotsFromAssign[who.id];
   741→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   742→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   743→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   744→            }
   745→            // Day committee count
   746→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   747→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   748→          });
   749→        }
   750→      }
   751→    }
   752→
   753→    // Override tracking with verified data from assignments
   754→    teachers.forEach(t => {
   755→      if (tracking[t.id]) {
   756→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   757→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   758→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   759→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   760→      }
   761→    });
   762→
   763→    // ---- Post-distribution: Ensure admin gets at least 2 assignments ----
   764→    for (const admin of adminTeachers) {
   765→      const adminComm = tracking[admin.id].totalComm;
   766→      if (adminComm >= adminMinTarget) continue;
   767→      // Try to swap: find a teacher with > average hours who has an assignment
   768→      // that admin could take (passing all constraints)
   769→      const allTeacherHours = teachers.map(t => tracking[t.id]?.totalHours || 0).filter(h => h > 0);
   770→      const avgH = allTeacherHours.length > 0 ? allTeacherHours.reduce((a, b) => a + b, 0) / allTeacherHours.length : 0;
   771→      const overburdened = teachers.filter(t => t.id !== admin.id && tracking[t.id]?.totalHours > avgH);
   772→      // Sort by most overburdened first
   773→      overburdened.sort((a, b) => (tracking[b.id]?.totalHours || 0) - (tracking[a.id]?.totalHours || 0));
   774→
   775→      for (const victim of overburdened) {
   776→        if (tracking[admin.id].totalComm >= adminMinTarget) break;
   777→        // Find an assignment of victim that admin can take
   778→        for (const day of DAYS) {
   779→          if (tracking[admin.id].totalComm >= adminMinTarget) break;
   780→          const daySlots = teacherSlotsFromAssign[victim.id].filter(s => s.day === day);
   781→          if (daySlots.length === 0) continue;
   782→          // Admin already has assignment on this day?
   783→          if ((tracking[admin.id].dayComm[day] || 0) >= 1) continue;
   784→          // Find the session+committee for this slot
   785→          for (const sess of finalAssignments[day]) {
   786→            if (tracking[admin.id].totalComm >= adminMinTarget) break;
   787→            const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   788→            if (daySlots[0].start !== sessTimeInfo.start || daySlots[0].end !== sessTimeInfo.end) continue;
   789→            // Check admin constraints
   790→            if (ruleSubject && sess.subject && admin.subject === sess.subject) continue;
   791→            if (!canSuperviseStage(admin, getStage(sess.grade))) continue;
   792→            // Check admin no time overlap (should be fine since we checked dayComm above)
   793→            if (tracking[admin.id].assignedSlots.some(s => s.day === day && !(sessTimeInfo.end <= s.start || sessTimeInfo.start >= s.end))) continue;
   794→            // Find which position (t1 or t2) victim has in this session
   795→            for (const c of sess.committees) {
   796→              if (c.t1?.id === victim.id || c.t2?.id === victim.id) {
   797→                // Check consecutive grade constraint
   798→                if (wasSameGradeAdjacent(tracking[admin.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTimeInfo, comId: 0 })) continue;
   799→                // Do the swap
   800→                const isT1 = c.t1?.id === victim.id;
   801→                const swappedTeacher = isT1 ? c.t1 : c.t2;
   802→                if (isT1) { c.t1 = { id: admin.id, name: admin.name }; }
   803→                else { c.t2 = { id: admin.id, name: admin.name }; }
   804→                // Update tracking: remove from victim, add to admin
   805→                tracking[victim.id].totalComm--;
   806→                tracking[victim.id].dayComm[day] = Math.max(0, (tracking[victim.id].dayComm[day] || 0) - 1);
   807→                tracking[admin.id].totalComm++;
   808→                tracking[admin.id].dayComm[day] = (tracking[admin.id].dayComm[day] || 0) + 1;
   809→                tracking[admin.id].assignedSlots.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   810→                tracking[admin.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   811→                break;
   812→              }
   813→            }
   814→          }
   815→        }
   816→      }
   817→    }
   818→
   819→    // ---- Post-distribution: BALANCE PASS ----
   820→    // Iteratively swap assignments from overburdened → underburdened teachers
   821→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   822→    const BALANCE_ITERATIONS = 500;
   823→    const recalcHours = () => {
   824→      teachers.forEach(t => {
   825→        const slots = tracking[t.id].assignedSlots || [];
   826→        const seen = new Set<string>();
   827→        let hrs = 0;
   828→        slots.forEach(s => {
   829→          const key = s.day + '_' + s.start + '_' + s.end;
   830→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   831→        });
   832→        tracking[t.id].totalHours = hrs;
   833→      });
   834→    };
   835→    recalcHours(); // recalc after admin swaps first
   836→
   837→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   838→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   839→      if (active.length < 2) break;
   840→      // Sort by HOURS first (primary), then committees (secondary)
   841→      const sorted = [...active].sort((a, b) => {
   842→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   843→        if (Math.abs(hDiff) > 0.1) return hDiff;
   844→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   845→      });
   846→      const maxH = tracking[sorted[0].id].totalHours;
   847→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   848→      if (maxH - minH <= 0.5) break; // spread ≤ 0.5h — tight balance
   849→
   850→      let swapped = false;
   851→      // Try top-5 donors × bottom-5 recipients for wider swap search
   852→      for (let di = 0; di < Math.min(sorted.length, 5) && !swapped; di++) {
   853→        const donor = sorted[di];
   854→        // Don't strip admins below their minimum
   855→        const isDonorAdmin = donor.subject === 'Admin';
   856→        if (tracking[donor.id].totalComm <= 1) continue;
   857→        if (isDonorAdmin && tracking[donor.id].totalComm <= adminMinTarget) continue;
   858→
   859→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 5) && !swapped; ri--) {
   860→          const recip = sorted[ri];
   861→          if (donor.id === recip.id) continue;
   862→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   863→            // Also check if committees would be more balanced after swap
   864→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   865→            const recipCommAfter = tracking[recip.id].totalComm + 1;
   866→            // Skip if swap would make committee imbalance worse (donor already has fewer comms)
   867→
   868→          // Try each day where donor has an assignment and recipient doesn't
   869→          for (const day of DAYS) {
   870→            if (swapped) break;
   871→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   872→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   873→
   874→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   875→            if (!donorSlot) continue;
   876→
   877→            // Find the session matching donor's slot
   878→            for (const sess of finalAssignments[day]) {
   879→              if (swapped) break;
   880→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   881→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   882→
   883→              // Check recipient constraints
   884→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   885→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   886→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   887→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   888→
   889→              // Find donor in this session's committees
   890→              for (const c of sess.committees) {
   891→                const isT1 = c.t1?.id === donor.id;
   892→                const isT2 = c.t2?.id === donor.id;
   893→                if (!isT1 && !isT2) continue;
   894→
   895→                // Execute swap
   896→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   897→                else c.t2 = { id: recip.id, name: recip.name };
   898→
   899→                tracking[donor.id].totalComm--;
   900→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);

   902→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   903→
   904→                tracking[recip.id].totalComm++;
   905→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   906→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   907→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   908→
   909→                swapped = true;
   910→                break;
   911→              }
   912→            }
   913→          }
   914→        }
   915→      }
   916→      if (!swapped) break; // no valid swap found

   975→            (standbys[prevDay][st] || []).forEach(s => yesterdayStandby.add(s.id));
   976→          }

   979→          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
   980→          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
