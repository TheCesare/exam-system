    43→    43→function parseTimeRange(tStr: string) {
    44→    44→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    45→    45→  try {
    46→    46→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    47→    47→    const parts = clean.split('-');
    48→    48→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    49→    49→    let start = toMinutes(parts[0]);
    50→    50→    let end = toMinutes(parts[1]);
    51→    51→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    52→    52→    if (start < 300 && end > start + 120) start += 720;
    53→    53→    if (end <= start) end += 720;
    54→    54→    let duration = (end - start) / 60;
    55→    55→    // Sanity cap: no exam is longer than 5 hours
    56→    56→    if (duration > 5) duration = 2.0; // fallback to standard
    57→    57→    return { start, end, duration };
    58→    58→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    59→    59→}

    61→    61→function getStage(grade: string) {
    62→    62→  const g = grade.toLowerCase();
    63→    63→  if (g.includes('primary')) return 'primary';
    64→    64→  if (g.includes('prep')) return 'prep';
    65→    65→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    66→    66→  return 'any';
    67→    67→}

   484→   484→  // ========== DISTRIBUTION ENGINE v4 ==========
   485→   485→  const runDistribution = () => {
   486→   486→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   487→   487→
   488→   488→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   489→   489→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   490→   490→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   491→   491→
   492→   492→    // ---- All teachers in ONE pool (admin participates normally) ----
   493→   493→    const adminTeachers = teachers.filter(t => t.subject === 'Admin');
   494→   494→
   495→   495→    // ---- Initialize tracking ----
   496→   496→    const tracking: Record<string, TrackingEntry> = {};
   497→   497→    teachers.forEach(t => {
   498→   498→      tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {} as Record<string,number>, assignedSlots: [], gradeHistory: [] };
   499→   499→      DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   500→   500→    });
   501→   501→
   502→   502→    // ---- Build exam slots grouped by day (chronological order) ----
   503→   503→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   504→   504→    const slotsByDay: Record<string, Slot[]> = {};
   505→   505→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   506→   506→    DAYS.forEach((day, dayIndex) => {
   507→   507→      GRADES.forEach(grade => {
   508→   508→        const cell = scheduleBuffer[`${grade}__${day}`];
   509→   509→        if (cell && cell.committees > 0) {
   510→   510→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   511→   511→          for (let c = 1; c <= cell.committees; c++) {
   512→   512→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   513→   513→          }
   514→   514→        }
   515→   515→      });
   516→   516→    });
   517→   517→
   518→   518→    // ---- Build final assignments structure ----
   519→   519→    const finalAssignments: Record<string, SessionResult[]> = {};
   520→   520→    DAYS.forEach(d => { finalAssignments[d] = []; });
   521→   521→    DAYS.forEach(day => {
   522→   522→      GRADES.forEach(grade => {
   523→   523→        const cell = scheduleBuffer[`${grade}__${day}`];
   524→   524→        if (cell && cell.committees > 0) {
   525→   525→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   526→   526→          if (!existing) {
   527→   527→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   528→   528→          }
   529→   529→        }
   530→   530→      });
   531→   531→    });
   532→   532→
   533→   533→    // ---- Helper: Stage notes — HARD constraint ----
   534→   534→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   535→   535→      if (!ruleNotes) return true;
   536→   536→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   537→   537→      const n = teacher.notes.toLowerCase();
   538→   538→      const allowed: string[] = [];
   539→   539→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   540→   540→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   541→   541→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   542→   542→      if (allowed.length === 0) return true;
   543→   543→      return allowed.includes(slotStage);
   544→   544→    };
   545→   545→
   546→   546→    // ---- Helper: Consecutive-day same-grade check ----
   547→   547→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   548→   548→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   549→   549→    };
   550→   550→
   551→   551→    // ---- Helper: Fisher-Yates shuffle ----
   552→   552→    const shuffle = <T,>(arr: T[]): T[] => {
   553→   553→      const a = [...arr];
   554→   554→      for (let i = a.length - 1; i > 0; i--) {
   555→   555→        const j = Math.floor(Math.random() * (i + 1));
   556→   556→        [a[i], a[j]] = [a[j], a[i]];
   557→   557→      }
   558→   558→      return a;
   559→   559→    };
   560→   560→
   561→   561→    // ---- Admin minimum assignment tracking ----
   562→   562→    const adminAssignmentCount: Record<string, number> = {};
   563→   563→    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   564→   564→    const adminMinTarget = 2; // each admin gets at least 2 assignments
   565→   565→    let totalAdminAssigned = 0;
   566→   566→
   567→   567→    // ---- Core: Find best teacher for a slot ----
   568→   568→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   569→   569→    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   570→   570→    const findBest = (
   571→   571→      blockedId: string | null,
   572→   572→      slot: Slot,
   573→   573→      pool: Teacher[],
   574→   574→      relaxAdj: boolean
   575→   575→    ): Teacher | null => {
   576→   576→      const candidates: { teacher: Teacher; score: number }[] = [];
   577→   577→      for (const t of pool) {
   578→   578→        if (blockedId && t.id === blockedId) continue;
   579→   579→        const tr = tracking[t.id];
   580→   580→        // HARD: No own-subject supervision
   581→   581→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   582→   582→        // HARD: Stage notes filtering
   583→   583→        if (!canSuperviseStage(t, slot.stage)) continue;
   584→   584→        // HARD: No time overlap on same day
   585→   585→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   586→   586→        // HARD: MAX 1 committee per teacher per day (NEVER relaxed)
   587→   587→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   588→   588→        // SOFT: No same grade on consecutive days (can be relaxed)
   589→   589→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   590→   590→        // Scoring: hours are king, tiny random for variety between equal candidates
   591→   591→        const isAdmin = t.subject === 'Admin';
   592→   592→        const adminBonus = (adminAssignmentCount[t.id] || 0) < adminMinTarget ? -3 : 0;
   593→   593→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (isAdmin ? 3 + adminBonus : 0) + Math.random() * 0.2;
   594→   594→        candidates.push({ teacher: t, score });
   595→   595→      }
   596→   596→      if (candidates.length === 0) return null;
   597→   597→      candidates.sort((a, b) => a.score - b.score);
   598→   598→      const best = candidates[0].score;
   599→   599→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   600→   600→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   601→   601→    };
   602→   602→
   603→   603→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 1-per-day + time) ----
   604→   604→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   605→   605→      const candidates: { teacher: Teacher; score: number }[] = [];
   606→   606→      for (const t of pool) {
   607→   607→        if (blockedId && t.id === blockedId) continue;
   608→   608→        const tr = tracking[t.id];
   609→   609→        // HARD: Still no own-subject supervision even in relaxed mode
   610→   610→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   611→   611→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   612→   612→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   613→   613→        const score = tr.totalHours * 200 + tr.totalComm * 5 + Math.random() * 0.2;
   614→   614→        candidates.push({ teacher: t, score });
   615→   615→      }
   616→   616→      if (candidates.length === 0) return null;
   617→   617→      candidates.sort((a, b) => a.score - b.score);
   618→   618→      return candidates[0].teacher;
   619→   619→    };
   620→   620→
   621→   621→    // ---- Fallback 3: Allow 2nd assignment same day (only time overlap + own-subject blocked) ----
   622→   622→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   623→   623→      const candidates: { teacher: Teacher; score: number }[] = [];
   624→   624→      for (const t of pool) {
   625→   625→        if (blockedId && t.id === blockedId) continue;
   626→   626→        const tr = tracking[t.id];
   627→   627→        // HARD: Still no own-subject supervision even in force-day mode
   628→   628→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   629→   629→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   630→   630→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (tr.dayComm[slot.day] || 0) * 50 + Math.random() * 0.2;
   631→   631→        candidates.push({ teacher: t, score });
   632→   632→      }
   633→   633→      if (candidates.length === 0) return null;
   634→   634→      candidates.sort((a, b) => a.score - b.score);
   635→   635→      return candidates[0].teacher;
   636→   636→    };
   637→   637→
   638→   638→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   639→   639→    let standbyCount = 0;
   640→   640→    const allPool = [...teachers]; // ALL teachers including admin
   641→   641→
   642→   642→    for (const day of DAYS) {
   643→   643→      const daySlots = shuffle(slotsByDay[day]);
   644→   644→      if (daySlots.length === 0) continue;
   645→   645→
   646→   646→      for (const slot of daySlots) {
   647→   647→        let t1: Teacher | null = null;
   648→   648→        let t2: Teacher | null = null;
   649→   649→
   650→   650→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   651→   651→        t1 = findBest(null, slot, allPool, false);
   652→   652→        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   653→   653→        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   654→   654→        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   655→   655→
   656→   656→        if (t1) {
   657→   657→          const tr = tracking[t1.id];
   658→   658→          tr.totalComm++; tr.dayComm[slot.day]++;
   659→   659→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   660→   660→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   661→   661→        } else { standbyCount++; }
   662→   662→
   663→   663→        // T2: same pool, blocked=T1, cascade fallbacks
   664→   664→        const blocked = t1?.id || null;
   665→   665→        t2 = findBest(blocked, slot, allPool, false);
   666→   666→        if (!t2) t2 = findBest(blocked, slot, allPool, true);
   667→   667→        if (!t2) t2 = findBestRelaxed(blocked, slot, allPool);
   668→   668→        if (!t2) t2 = findBestForceDay(blocked, slot, allPool);
   669→   669→
   670→   670→        if (t2) {
   671→   671→          if (t2.subject === 'Admin') {
   672→   672→            adminAssignmentCount[t2.id] = (adminAssignmentCount[t2.id] || 0) + 1;
   673→   673→            totalAdminAssigned++;
   674→   674→          }
   675→   675→          const tr = tracking[t2.id];
   676→   676→          tr.totalComm++; tr.dayComm[slot.day]++;
   677→   677→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   678→   678→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   679→   679→        } else { standbyCount++; }
   680→   680→
   681→   681→        // Record assignment
   682→   682→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   683→   683→        if (session) {
   684→   684→          session.committees.push({
   685→   685→            serial: slot.comId,
   686→   686→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   687→   687→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   688→   688→          });
   689→   689→        }
   690→   690→      }
   691→   691→    }
   692→   692→
   693→   693→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   694→   694→    // This eliminates any tracking accumulation bugs
   695→   695→    const teacherCommCount: Record<string, number> = {};
   696→   696→    const teacherHoursFromAssign: Record<string, number> = {};
   697→   697→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   698→   698→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   699→   699→    teachers.forEach(t => {
   700→   700→      teacherCommCount[t.id] = 0;
   701→   701→      teacherHoursFromAssign[t.id] = 0;
   702→   702→      teacherDayCommFromAssign[t.id] = {};
   703→   703→      teacherSlotsFromAssign[t.id] = [];
   704→   704→    });
   705→   705→
   706→   706→    for (const day of DAYS) {
   707→   707→      for (const sess of finalAssignments[day]) {
   708→   708→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   709→   709→        for (const c of sess.committees) {
   710→   710→          [c.t1, c.t2].forEach(who => {
   711→   711→            if (!who.id) return;
   712→   712→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   713→   713→            // Deduplicate: same day + same time = counted once
   714→   714→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   715→   715→            const existing = teacherSlotsFromAssign[who.id];
   716→   716→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   717→   717→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   718→   718→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   719→   719→            }
   720→   720→            // Day committee count
   721→   721→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   722→   722→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   723→   723→          });
   724→   724→        }
   725→   725→      }
   726→   726→    }
   727→   727→
   728→   728→    // Override tracking with verified data from assignments
   729→   729→    teachers.forEach(t => {
   730→   730→      if (tracking[t.id]) {
   731→   731→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   732→   732→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   733→   733→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   734→   734→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   735→   735→      }
   736→   736→    });
   737→   737→
   738→   738→    // ---- Post-distribution: Ensure admin gets at least 2 assignments ----
   739→   739→    for (const admin of adminTeachers) {
   740→   740→      const adminComm = tracking[admin.id].totalComm;
   741→   741→      if (adminComm >= adminMinTarget) continue;
   742→   742→      // Try to swap: find a teacher with > average hours who has an assignment
   743→   743→      // that admin could take (passing all constraints)
   744→   744→      const allTeacherHours = teachers.map(t => tracking[t.id]?.totalHours || 0).filter(h => h > 0);
   745→   745→      const avgH = allTeacherHours.length > 0 ? allTeacherHours.reduce((a, b) => a + b, 0) / allTeacherHours.length : 0;
   746→   746→      const overburdened = teachers.filter(t => t.id !== admin.id && tracking[t.id]?.totalHours > avgH);
   747→   747→      // Sort by most overburdened first
   748→   748→      overburdened.sort((a, b) => (tracking[b.id]?.totalHours || 0) - (tracking[a.id]?.totalHours || 0));
   749→   749→
   750→   750→      for (const victim of overburdened) {
   751→   751→        if (tracking[admin.id].totalComm >= adminMinTarget) break;
   752→   752→        // Find an assignment of victim that admin can take
   753→   753→        for (const day of DAYS) {
   754→   754→          if (tracking[admin.id].totalComm >= adminMinTarget) break;
   755→   755→          const daySlots = teacherSlotsFromAssign[victim.id].filter(s => s.day === day);
   756→   756→          if (daySlots.length === 0) continue;
   757→   757→          // Admin already has assignment on this day?
   758→   758→          if ((tracking[admin.id].dayComm[day] || 0) >= 1) continue;
   759→   759→          // Find the session+committee for this slot
   760→   760→          for (const sess of finalAssignments[day]) {
   761→   761→            if (tracking[admin.id].totalComm >= adminMinTarget) break;
   762→   762→            const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   763→   763→            if (daySlots[0].start !== sessTimeInfo.start || daySlots[0].end !== sessTimeInfo.end) continue;
   764→   764→            // Check admin constraints
   765→   765→            if (ruleSubject && sess.subject && admin.subject === sess.subject) continue;
   766→   766→            if (!canSuperviseStage(admin, getStage(sess.grade))) continue;
   767→   767→            // Check admin no time overlap (should be fine since we checked dayComm above)
   768→   768→            if (tracking[admin.id].assignedSlots.some(s => s.day === day && !(sessTimeInfo.end <= s.start || sessTimeInfo.start >= s.end))) continue;
   769→   769→            // Find which position (t1 or t2) victim has in this session
   770→   770→            for (const c of sess.committees) {
   771→   771→              if (c.t1?.id === victim.id || c.t2?.id === victim.id) {
   772→   772→                // Check consecutive grade constraint
   773→   773→                if (wasSameGradeAdjacent(tracking[admin.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTimeInfo, comId: 0 })) continue;
   774→   774→                // Do the swap
   775→   775→                const isT1 = c.t1?.id === victim.id;
   776→   776→                const swappedTeacher = isT1 ? c.t1 : c.t2;
   777→   777→                if (isT1) { c.t1 = { id: admin.id, name: admin.name }; }
   778→   778→                else { c.t2 = { id: admin.id, name: admin.name }; }
   779→   779→                // Update tracking: remove from victim, add to admin
   780→   780→                tracking[victim.id].totalComm--;
   781→   781→                tracking[victim.id].dayComm[day] = Math.max(0, (tracking[victim.id].dayComm[day] || 0) - 1);
   782→   782→                tracking[admin.id].totalComm++;
   783→   783→                tracking[admin.id].dayComm[day] = (tracking[admin.id].dayComm[day] || 0) + 1;
   784→   784→                tracking[admin.id].assignedSlots.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   785→   785→                tracking[admin.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   786→   786→                break;
   787→   787→              }
   788→   788→            }
   789→   789→          }
   790→   790→        }
   791→   791→      }
   792→   792→    }
   793→   793→
   794→   794→    // ---- Post-distribution: BALANCE PASS ----
   795→   795→    // Iteratively swap assignments from overburdened → underburdened teachers
   796→   796→    // to minimize the max-min spread in hours
   797→   797→    const BALANCE_ITERATIONS = 150;
   798→   798→    const recalcHours = () => {
   799→   799→      teachers.forEach(t => {
   800→   800→        const slots = tracking[t.id].assignedSlots || [];
   801→   801→        const seen = new Set<string>();
   802→   802→        let hrs = 0;
   803→   803→        slots.forEach(s => {
   804→   804→          const key = s.day + '_' + s.start + '_' + s.end;
   805→   805→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   806→   806→        });
   807→   807→        tracking[t.id].totalHours = hrs;
   808→   808→      });
   809→   809→    };
   810→   810→    recalcHours(); // recalc after admin swaps first
   811→   811→
   812→   812→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   813→   813→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   814→   814→      if (active.length < 2) break;
   815→   815→      const sorted = [...active].sort((a, b) => tracking[b.id].totalHours - tracking[a.id].totalHours);
   816→   816→      const maxH = tracking[sorted[0].id].totalHours;
   817→   817→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   818→   818→      if (maxH - minH <= 1.0) break; // spread ≤ 1h — good enough
   819→   819→
   820→   820→      let swapped = false;
   821→   821→      // Try top-3 donors × bottom-3 recipients
   822→   822→      for (let di = 0; di < Math.min(sorted.length, 3) && !swapped; di++) {
   823→   823→        const donor = sorted[di];
   824→   824→        // Don't strip admins below their minimum
   825→   825→        const isDonorAdmin = donor.subject === 'Admin';
   826→   826→        if (tracking[donor.id].totalComm <= 1) continue;
   827→   827→        if (isDonorAdmin && tracking[donor.id].totalComm <= adminMinTarget) continue;
   828→   828→
   829→   829→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 3) && !swapped; ri--) {
   830→   830→          const recip = sorted[ri];
   831→   831→          if (donor.id === recip.id) continue;
   832→   832→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   833→   833→
   834→   834→          // Try each day where donor has an assignment and recipient doesn't
   835→   835→          for (const day of DAYS) {
   836→   836→            if (swapped) break;
   837→   837→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   838→   838→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   839→   839→
   840→   840→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   841→   841→            if (!donorSlot) continue;
   842→   842→
   843→   843→            // Find the session matching donor's slot
   844→   844→            for (const sess of finalAssignments[day]) {
   845→   845→              if (swapped) break;
   846→   846→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   847→   847→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   848→   848→
   849→   849→              // Check recipient constraints
   850→   850→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   851→   851→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   852→   852→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   853→   853→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   854→   854→
   855→   855→              // Find donor in this session's committees
   856→   856→              for (const c of sess.committees) {
   857→   857→                const isT1 = c.t1?.id === donor.id;
   858→   858→                const isT2 = c.t2?.id === donor.id;
   859→   859→                if (!isT1 && !isT2) continue;
   860→   860→
   861→   861→                // Execute swap
   862→   862→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   863→   863→                else c.t2 = { id: recip.id, name: recip.name };
   864→   864→
   865→   865→                tracking[donor.id].totalComm--;
   866→   866→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   867→   867→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   868→   868→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   869→   869→
   870→   870→                tracking[recip.id].totalComm++;
   871→   871→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   872→   872→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   873→   873→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   874→   874→
   875→   875→                swapped = true;
   876→   876→                break;
   877→   877→              }
   878→   878→            }
   879→   879→          }
   880→   880→        }
   881→   881→      }
   882→   882→      if (!swapped) break; // no valid swap found
   883→   883→      recalcHours(); // update hours after each swap
   884→   884→    }
   885→   885→
   886→   886→    // Final hour recalculation after all balancing
   887→   887→    recalcHours();
   888→   888→
   889→   889→    // ---- Sort committees by serial within each session (1, 2, 3...) ----
   890→   890→    for (const day of DAYS) {
   891→   891→      for (const sess of finalAssignments[day]) {
   892→   892→        sess.committees.sort((a, b) => a.serial - b.serial);
   893→   893→      }
   894→   894→    }
   895→   895→
   896→   896→    // ---- Post-distribution: STANDBY ASSIGNMENT ----
   897→   897→    // 1 standby teacher per stage per day from unassigned teachers
   898→   898→    // CRITICAL: Standby must NOT be the same subject as any exam that day for that stage
   899→   899→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   900→   900→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   901→   901→    const STANDBY_PER_STAGE = 1;
   902→   902→    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
   903→   903→    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
   904→   904→
   905→   905→    // Track standby count per teacher for rotation fairness
   906→   906→    const teacherStandbyCount: Record<string, number> = {};
   907→   907→    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
   908→   908→
   909→   909→    // Track which teachers are already standby for another stage today
   910→   910→    const todayStandby: Record<string, Set<string>> = {};
   911→   911→    DAYS.forEach(d => { todayStandby[d] = new Set(); });
   912→   912→
   913→   913→    for (let di = 0; di < DAYS.length; di++) {
   914→   914→      const day = DAYS[di];
   915→   915→
   916→   916→      for (const stage of STAGES_LIST) {
   917→   917→        // Only assign standby if there are exams for this stage on this day
   918→   918→        const stageSessions = (finalAssignments[day] || []).filter(s => getStage(s.grade) === stage);
   919→   919→        if (stageSessions.length === 0) continue;
   920→   920→
   921→   921→        // Collect subjects being examined today for this stage
   922→   922→        const dayStageSubjects = new Set<string>();
   923→   923→        stageSessions.forEach(s => { if (s.subject) dayStageSubjects.add(s.subject); });
   924→   924→
   925→   925→        // Find teachers NOT assigned on this day who can supervise this stage
   926→   926→        // AND whose subject doesn't match any exam subject for this stage today
   927→   927→        let candidates = teachers.filter(t => {
   928→   928→          if ((tracking[t.id].dayComm[day] || 0) >= 1) return false;
   929→   929→          if (todayStandby[day].has(t.id)) return false; // already standby for another stage today
   930→   930→          if (!canSuperviseStage(t, stage)) return false;
   931→   931→          // CRITICAL: Don't pick a teacher whose subject is being examined today
   932→   932→          if (ruleSubject && t.subject && dayStageSubjects.has(t.subject)) return false;
   933→   933→          return true;
   934→   934→        });
   935→   935→
   936→   936→        // Rotation: deprioritize teachers who were standby YESTERDAY for ANY stage
   937→   937→        const yesterdayStandby = new Set<string>();
   938→   938→        if (di > 0) {
   939→   939→          const prevDay = DAYS[di - 1];
   940→   940→          for (const st of STAGES_LIST) {
   941→   941→            (standbys[prevDay][st] || []).forEach(s => yesterdayStandby.add(s.id));
   942→   942→          }
   943→   943→        }
   944→   944→        candidates.sort((a, b) => {
   945→   945→          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
   946→   946→          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
   947→   947→          return (teacherStandbyCount[a.id] + aWasYesterday) - (teacherStandbyCount[b.id] + bWasYesterday);
   948→   948→        });
   949→   949→
   950→   950→        // Pick up to STANDBY_PER_STAGE
   951→   951→        const picked = candidates.slice(0, STANDBY_PER_STAGE);
   952→   952→        standbys[day][stage] = picked.map(t => ({ id: t.id, name: t.name }));
   953→   953→
   954→   954→        // Track rotation count and mark as standby for today (so same teacher isn't picked for another stage)
   955→   955→        for (const s of picked) {
   956→   956→          teacherStandbyCount[s.id]++;
   957→   957→          todayStandby[day].add(s.id);
   958→   958→        }
   959→   959→      }
   960→   960→    }
   961→   961→
   962→   962→    // ---- Post-distribution verification ----
   963→   963→    const violations: string[] = [];
   964→   964→    // V-Check 1: Own-subject supervision
   965→   965→    if (ruleSubject) {
   966→   966→      for (const day of DAYS) {
   967→   967→        for (const sess of finalAssignments[day]) {
   968→   968→          for (const c of sess.committees) {
   969→   969→            [c.t1, c.t2].forEach(who => {
   970→   970→              if (who.id) {
   971→   971→                const tch = teachers.find(x => x.id === who.id);
   972→   972→                if (tch && tch.subject === sess.subject) violations.push(`${tch.name} -> own subject (${sess.subject}) on ${day}`);
   973→   973→              }
   974→   974→            });
   975→   975→          }
   976→   976→        }
   977→   977→      }
   978→   978→    }
   979→   979→    // V-Check 2: Time overlap
   980→   980→    for (const t of teachers) {
   981→   981→      const tr = tracking[t.id];
   982→   982→      for (let i = 0; i < tr.assignedSlots.length; i++) {
   983→   983→        for (let j = i + 1; j < tr.assignedSlots.length; j++) {
   984→   984→          const a = tr.assignedSlots[i], b = tr.assignedSlots[j];
   985→   985→          if (a.day === b.day && !(a.end <= b.start || b.end <= a.start)) {
   986→   986→            violations.push(`${t.name} -> time overlap on ${a.day}`);
   987→   987→          }
   988→   988→        }
   989→   989→      }
   990→   990→    }
   991→   991→    // V-Check 3: Same-day double assignment (should NEVER happen now)
   992→   992→    for (const t of teachers) {
   993→   993→      const tr = tracking[t.id];
   994→   994→      for (const day of DAYS) {
   995→   995→        if ((tr.dayComm[day] || 0) > 1) {
   996→   996→          violations.push(`${t.name} -> 2+ committees on ${day} (HARD RULE VIOLATED)`);
   997→   997→        }
   998→   998→      }
   999→   999→    }
  1000→  1000→    // V-Check 4: Stage notes compliance
  1001→  1001→    if (ruleNotes) {
  1002→  1002→      for (const day of DAYS) {
  1003→  1003→        for (const sess of finalAssignments[day]) {
  1004→  1004→          const stage = getStage(sess.grade);
  1005→  1005→          for (const c of sess.committees) {
  1006→  1006→            [c.t1, c.t2].forEach(who => {
  1007→  1007→              if (who.id) {
  1008→  1008→                const tch = teachers.find(x => x.id === who.id);
  1009→  1009→                if (tch && !canSuperviseStage(tch, stage)) violations.push(`${tch.name} -> wrong stage (${stage}) notes="${tch.notes}" on ${day}`);
  1010→  1010→              }
  1011→  1011→            });
  1012→  1012→          }
  1013→  1013→        }
  1014→  1014→      }
  1015→  1015→    }
  1016→  1016→
  1017→  1017→    // ---- Build summary statistics (from verified tracking) ----
  1018→  1018→    const allH = teachers.map(t => tracking[t.id]?.totalHours || 0);
  1019→  1019→    const avgAll = allH.length ? allH.reduce((a, b) => a + b, 0) / allH.length : 0;
  1020→  1020→    const spread = allH.length ? Math.sqrt(allH.reduce((s, h) => s + (h - avgAll) ** 2, 0) / allH.length) : 0;
  1021→  1021→    let msg = `v8 | Avg: ${avgAll.toFixed(1)}h | Spread: ${spread.toFixed(1)} | Min: ${allH.length ? Math.min(...allH).toFixed(1) : 0}h | Max: ${allH.length ? Math.max(...allH).toFixed(1) : 0}h`;
  1022→  1022→    // Count total standby assigned
  1023→  1023→    const totalStandby = Object.values(standbys).reduce((a, daySt) => a + Object.values(daySt).reduce((b, stList) => b + stList.length, 0), 0);
  1024→  1024→    if (totalStandby > 0) msg += ` | ${totalStandby} standby (${STANDBY_PER_STAGE}/stage/day)`;
  1025→  1025→    if (standbyCount > 0) msg += ` | ${standbyCount} unfilled`;
  1026→  1026→    if (violations.length > 0) {
  1027→  1027→      msg += ` | ${violations.length} violations (check console)`;
  1028→  1028→      console.warn('[Distribution v8] Violations:', violations);
  1029→  1029→    } else {
  1030→  1030→      console.log('[Distribution v8] All constraints passed!');
  1031→  1031→    }
  1032→  1032→
  1033→  1033→    const newResults: DistributionResults = { _version: 8, assignments: finalAssignments, standbys, tracking };
  1034→  1034→    setResults(newResults);
  1035→  1035→    fetch('/api/results', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ data: newResults }) });
  1036→  1036→    showToast(msg, standbyCount > 0 || violations.length > 0 ? 'error' : 'success');
  1037→  1037→    setActivePage('results');
  1038→  1038→  };
