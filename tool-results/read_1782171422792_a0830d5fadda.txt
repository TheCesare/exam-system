   806→function parseTimeRange(tStr) {
   807→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
   808→  try {
   809→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
   810→    const parts = clean.split('-');
   811→    const toMinutes = (timeString) => {
   812→      const p = timeString.split(':');
   813→      let hrs = parseInt(p[0]) || 0;
   814→      let mins = parseInt(p[1]) || 0;
   815→      return (hrs * 60) + mins;
   816→    };
   817→    let start = toMinutes(parts[0]);
   818→    let end = toMinutes(parts[1]);
   819→    if (end <= start) end += 12 * 60;
   820→    return { start, end, duration: (end - start) / 60 };
   821→  } catch(e) {
   822→    return { start: 540, end: 630, duration: 1.5 };
   823→  }
   824→}

   837→function runDistribution() {
   838→  if (state.teachers.length === 0) { toast('Error: Registry empty!', 'error'); return; }
   839→
   840→  const ruleSubject = document.getElementById('rule-subject').checked;
   841→  const ruleDayLimit = document.getElementById('rule-daylimit').checked;
   842→  const ruleNotes = document.getElementById('rule-notes').checked;
   843→
   844→  let tracking = {};
   845→  state.teachers.forEach(t => {
   846→    tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {}, assignedSlots: [] };
   847→    DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   848→  });
   849→
   850→  let allSlots = [];
   851→  DAYS.forEach(day => {
   852→    GRADES.forEach(grade => {
   853→      const cell = state.schedule[grade]?.[day];
   854→      if (cell && cell.committees > 0) {
   855→        let lowerGrade = grade.toLowerCase();
   856→        let stage = "any";
   857→        if(lowerGrade.includes("primary")) stage = "primary";
   858→        else if(lowerGrade.includes("prep")) stage = "prep";
   859→        else if(lowerGrade.includes("secondary") || lowerGrade.includes("sec")) stage = "sec";
   860→
   861→        let timeInfo = parseTimeRange(cell.time);
   862→        for (let c = 1; c <= cell.committees; c++) {
   863→          allSlots.push({ day, grade, stage, subject: cell.subject, time: cell.time, timeInfo, comId: c });
   864→        }
   865→      }
   866→    });
   867→  });
   868→
   869→  // Heaviest duration loads calculated first
   870→  allSlots.sort((a, b) => b.timeInfo.duration - a.timeInfo.duration);
   871→
   872→  let finalAssignments = {};
   873→  DAYS.forEach(d => { finalAssignments[d] = []; });
   874→  DAYS.forEach(day => {
   875→    GRADES.forEach(grade => {
   876→      const cell = state.schedule[grade]?.[day];
   877→      if (cell && cell.committees > 0) {
   878→        let existing = finalAssignments[day].find(s => s.grade === grade && s.time === cell.time);
   879→        if (!existing) {
   880→          finalAssignments[day].push({ grade, time: cell.time, subject: cell.subject, committees: [] });
   881→        }
   882→      }
   883→    });
   884→  });
   885→
   886→  allSlots.forEach(slot => {
   887→    
   888→    let queryTeacher = (blockedId) => {
   889→      let pool = [];
   890→
   891→      state.teachers.forEach(t => {
   892→        if (blockedId && t.id === blockedId) return; 
   893→        
   894→        let tr = tracking[t.id];
   895→
   896→        // 1. Subject Block
   897→        if (ruleSubject && slot.subject && t.subject === slot.subject) return;
   898→
   899→        // 2. STRICT time overlap - cannot be in 2 places at same time
   900→        let hasOverlap = tr.assignedSlots.some(s => {
   901→          return s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end);
   902→        });
   903→        if (hasOverlap) return;
   904→
   905→        // 3. Day limit rule
   906→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) return;
   907→
   908→        // 4. Max 5h per day - count unique hours today
   909→        let todayHours = 0;
   910→        let seenToday = new Set();
   911→        tr.assignedSlots.filter(s => s.day === slot.day).forEach(s => {
   912→          let k = s.start + '_' + s.end;
   913→          if (!seenToday.has(k)) { seenToday.add(k); todayHours += (s.end - s.start) / 60; }
   914→        });
   915→        if (todayHours + slot.timeInfo.duration > 5) return;
   916→
   917→        let matchScore = 0;
   918→        if (ruleNotes && t.notes && slot.stage !== "any" && t.notes.toLowerCase().includes(slot.stage)) {
   919→          matchScore = 1;
   920→        }
   921→
   922→        pool.push({ teacher: t, matchScore, totalComm: tr.totalComm, totalHours: tr.totalHours });
   923→      });
   924→
   925→      // Emergency Fallback: relax day limit only, KEEP overlap + max hours
   926→      if (pool.length === 0) {
   927→        state.teachers.forEach(t => {
   928→          if (blockedId && t.id === blockedId) return;
   929→          let tr = tracking[t.id];
   930→          if (ruleSubject && slot.subject && t.subject === slot.subject) return;
   931→          let hasOverlap = tr.assignedSlots.some(s => {
   932→            return s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end);
   933→          });
   934→          if (hasOverlap) return;
   935→          let todayH = 0; let seenT = new Set();
   936→          tr.assignedSlots.filter(s => s.day === slot.day).forEach(s => {
   937→            let k = s.start+'_'+s.end; if(!seenT.has(k)){seenT.add(k); todayH += (s.end-s.start)/60;}
   938→          });
   939→          if (todayH + slot.timeInfo.duration > 5) return;
   940→          pool.push({ teacher: t, matchScore: 0, totalComm: tr.totalComm, totalHours: tr.totalHours });
   941→        });
   942→      }
   943→
   944→      if (pool.length === 0) return null;
   945→
   946→      pool.sort((a, b) => {
   947→        if (a.totalComm !== b.totalComm) return a.totalComm - b.totalComm;
   948→        if (Math.abs(a.totalHours - b.totalHours) > 0.001) return a.totalHours - b.totalHours;
   949→        if (b.matchScore !== a.matchScore) return b.matchScore - a.matchScore;
   950→        return a.teacher.id - b.teacher.id;
   951→      });
   952→
   953→      return pool[0].teacher;
   954→    };
   955→
   956→    let t1 = queryTeacher(null);
   957→    if (t1) {
   958→      tracking[t1.id].totalComm++;
   959→      tracking[t1.id].dayComm[slot.day]++;
   960→      tracking[t1.id].totalHours += slot.timeInfo.duration;
   961→      tracking[t1.id].assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   962→    }
   963→
   964→    let t2 = queryTeacher(t1 ? t1.id : null);
   965→    if (t2) {
   966→      tracking[t2.id].totalComm++;
   967→      tracking[t2.id].dayComm[slot.day]++;
   968→      tracking[t2.id].totalHours += slot.timeInfo.duration;
   969→      tracking[t2.id].assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   970→    }
   971→
   972→    let dayGroup = finalAssignments[slot.day];
   973→    let sessionGroup = dayGroup.find(s => s.grade === slot.grade && s.time === slot.time);
   974→    if (sessionGroup) {
   975→      sessionGroup.committees.push({
   976→        serial: slot.comId,
   977→        t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: 'Standby Monitor' },
   978→        t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: 'Standby Monitor' }
   979→      });
   980→    }
   981→  });
   982→
   983→  state.results = { assignments: finalAssignments, tracking };
   984→  saveState();
   985→  toast('Overlap Lock Secured! Schedules fully cleared. ✓', 'success');
   986→  showPage('results');
   987→}

  1075→  });

  1084→    if ((tt.totalComm || 0) === 0) notUsedCount++;

  1089→  // Sort by total hours descending
