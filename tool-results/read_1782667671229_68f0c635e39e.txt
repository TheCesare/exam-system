    45→    45→function parseTimeRange(tStr: string) {
    46→    46→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    47→    47→  try {
    48→    48→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    49→    49→    const parts = clean.split('-');
    50→    50→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    51→    51→    let start = toMinutes(parts[0]);
    52→    52→    let end = toMinutes(parts[1]);
    53→    53→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    54→    54→    if (start < 300 && end > start + 120) start += 720;
    55→    55→    if (end <= start) end += 720;
    56→    56→    let duration = (end - start) / 60;
    57→    57→    // Sanity cap: no exam is longer than 5 hours
    58→    58→    if (duration > 5) duration = 2.0; // fallback to standard
    59→    59→    return { start, end, duration };
    60→    60→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    61→    61→}

   485→   485→
   486→   486→  // ========== DISTRIBUTION ENGINE v4 ==========
   487→   487→  const runDistribution = () => {
   488→   488→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   489→   489→
   490→   490→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   491→   491→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   492→   492→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   493→   493→
   494→   494→    // ---- All teachers in ONE pool (admin participates normally) ----
   495→   495→    const adminTeachers = teachers.filter(t => t.subject === 'Admin');
   496→   496→
   497→   497→    // ---- Initialize tracking ----
   498→   498→    const tracking: Record<string, TrackingEntry> = {};
   499→   499→    teachers.forEach(t => {
   500→   500→      tracking[t.id] = { totalComm: 0, totalHours: 0, dayComm: {} as Record<string,number>, assignedSlots: [], gradeHistory: [] };
   501→   501→      DAYS.forEach(d => { tracking[t.id].dayComm[d] = 0; });
   502→   502→    });
   503→   503→
   504→   504→    // ---- Build exam slots grouped by day (chronological order) ----
   505→   505→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   506→   506→    const slotsByDay: Record<string, Slot[]> = {};
   507→   507→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   508→   508→    DAYS.forEach((day, dayIndex) => {
   509→   509→      GRADES.forEach(grade => {
   510→   510→        const cell = scheduleBuffer[`${grade}__${day}`];
   511→   511→        if (cell && cell.committees > 0) {
   512→   512→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   513→   513→          for (let c = 1; c <= cell.committees; c++) {
   514→   514→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   515→   515→          }
   516→   516→        }
   517→   517→      });
   518→   518→    });
   519→   519→
   520→   520→    // ---- Build final assignments structure ----
   521→   521→    const finalAssignments: Record<string, SessionResult[]> = {};
   522→   522→    DAYS.forEach(d => { finalAssignments[d] = []; });
   523→   523→    DAYS.forEach(day => {
   524→   524→      GRADES.forEach(grade => {
   525→   525→        const cell = scheduleBuffer[`${grade}__${day}`];
   526→   526→        if (cell && cell.committees > 0) {
   527→   527→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   528→   528→          if (!existing) {
   529→   529→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   530→   530→          }
   531→   531→        }
   532→   532→      });
   533→   533→    });
   534→   534→
   535→   535→    // ---- Helper: Stage notes — HARD constraint ----
   536→   536→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   537→   537→      if (!ruleNotes) return true;
   538→   538→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   539→   539→      const n = teacher.notes.toLowerCase();
   540→   540→      const allowed: string[] = [];
   541→   541→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   542→   542→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   543→   543→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   544→   544→      if (allowed.length === 0) return true;
   545→   545→      return allowed.includes(slotStage);
   546→   546→    };
   547→   547→
   548→   548→    // ---- Helper: Consecutive-day same-grade check ----
   549→   549→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   550→   550→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   551→   551→    };
   552→   552→
   553→   553→    // ---- Helper: Fisher-Yates shuffle ----
   554→   554→    const shuffle = <T,>(arr: T[]): T[] => {
   555→   555→      const a = [...arr];
   556→   556→      for (let i = a.length - 1; i > 0; i--) {
   557→   557→        const j = Math.floor(Math.random() * (i + 1));
   558→   558→        [a[i], a[j]] = [a[j], a[i]];
   559→   559→      }
   560→   560→      return a;
   561→   561→    };
   562→   562→
   563→   563→    // ---- Admin minimum assignment tracking ----
   564→   564→    const adminAssignmentCount: Record<string, number> = {};
   565→   565→    adminTeachers.forEach(t => { adminAssignmentCount[t.id] = 0; });
   566→   566→    const adminMinTarget = 2; // each admin gets at least 2 assignments
   567→   567→    let totalAdminAssigned = 0;
   568→   568→
   569→   569→    // ---- Core: Find best teacher for a slot ----
   570→   570→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   571→   571→    // Scoring: hours dominate (200x), tiny noise (0.2) for variety
   572→   572→    const findBest = (
   573→   573→      blockedId: string | null,
   574→   574→      slot: Slot,
   575→   575→      pool: Teacher[],
   576→   576→      relaxAdj: boolean
   577→   577→    ): Teacher | null => {
   578→   578→      const candidates: { teacher: Teacher; score: number }[] = [];
   579→   579→      for (const t of pool) {
   580→   580→        if (blockedId && t.id === blockedId) continue;
   581→   581→        const tr = tracking[t.id];
   582→   582→        // HARD: No own-subject supervision
   583→   583→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   584→   584→        // HARD: Stage notes filtering
   585→   585→        if (!canSuperviseStage(t, slot.stage)) continue;
   586→   586→        // HARD: No time overlap on same day
   587→   587→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   588→   588→        // HARD: MAX 1 committee per teacher per day (NEVER relaxed)
   589→   589→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   590→   590→        // SOFT: No same grade on consecutive days (can be relaxed)
   591→   591→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   592→   592→        // Scoring: hours are king, tiny random for variety between equal candidates
   593→   593→        const isAdmin = t.subject === 'Admin';
   594→   594→        const adminBonus = (adminAssignmentCount[t.id] || 0) < adminMinTarget ? -3 : 0;
   595→   595→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (isAdmin ? 3 + adminBonus : 0) + Math.random() * 0.2;
   596→   596→        candidates.push({ teacher: t, score });
   597→   597→      }
   598→   598→      if (candidates.length === 0) return null;
   599→   599→      candidates.sort((a, b) => a.score - b.score);
   600→   600→      const best = candidates[0].score;
   601→   601→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   602→   602→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   603→   603→    };
   604→   604→
   605→   605→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 1-per-day + time) ----
   606→   606→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   607→   607→      const candidates: { teacher: Teacher; score: number }[] = [];
   608→   608→      for (const t of pool) {
   609→   609→        if (blockedId && t.id === blockedId) continue;
   610→   610→        const tr = tracking[t.id];
   611→   611→        // HARD: Still no own-subject supervision even in relaxed mode
   612→   612→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   613→   613→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   614→   614→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) continue;
   615→   615→        const score = tr.totalHours * 200 + tr.totalComm * 5 + Math.random() * 0.2;
   616→   616→        candidates.push({ teacher: t, score });
   617→   617→      }
   618→   618→      if (candidates.length === 0) return null;
   619→   619→      candidates.sort((a, b) => a.score - b.score);
   620→   620→      return candidates[0].teacher;
   621→   621→    };
   622→   622→
   623→   623→    // ---- Fallback 3: Allow 2nd assignment same day (only time overlap + own-subject blocked) ----
   624→   624→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   625→   625→      const candidates: { teacher: Teacher; score: number }[] = [];
   626→   626→      for (const t of pool) {
   627→   627→        if (blockedId && t.id === blockedId) continue;
   628→   628→        const tr = tracking[t.id];
   629→   629→        // HARD: Still no own-subject supervision even in force-day mode
   630→   630→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   631→   631→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   632→   632→        const score = tr.totalHours * 200 + tr.totalComm * 5 + (tr.dayComm[slot.day] || 0) * 50 + Math.random() * 0.2;
   633→   633→        candidates.push({ teacher: t, score });
   634→   634→      }
   635→   635→      if (candidates.length === 0) return null;
   636→   636→      candidates.sort((a, b) => a.score - b.score);
   637→   637→      return candidates[0].teacher;
   638→   638→    };
   639→   639→
   640→   640→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   641→   641→    let standbyCount = 0;
   642→   642→    const allPool = [...teachers]; // ALL teachers including admin
   643→   643→
   644→   644→    for (const day of DAYS) {
   645→   645→      const daySlots = shuffle(slotsByDay[day]);
   646→   646→      if (daySlots.length === 0) continue;
   647→   647→
   648→   648→      for (const slot of daySlots) {
   649→   649→        let t1: Teacher | null = null;
   650→   650→        let t2: Teacher | null = null;
   651→   651→
   652→   652→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day
   653→   653→        t1 = findBest(null, slot, allPool, false);
   654→   654→        if (!t1) t1 = findBest(null, slot, allPool, true); // relax consecutive-day
   655→   655→        if (!t1) t1 = findBestRelaxed(null, slot, allPool); // relax subject + stage
   656→   656→        if (!t1) t1 = findBestForceDay(null, slot, allPool); // allow 2nd assignment same day
   657→   657→
   658→   658→        if (t1) {
   659→   659→          const tr = tracking[t1.id];
   660→   660→          tr.totalComm++; tr.dayComm[slot.day]++;
   661→   661→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   662→   662→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   663→   663→        } else { standbyCount++; }
   664→   664→
   665→   665→        // T2: same pool, blocked=T1, cascade fallbacks
   666→   666→        const blocked = t1?.id || null;
   667→   667→        t2 = findBest(blocked, slot, allPool, false);
   668→   668→        if (!t2) t2 = findBest(blocked, slot, allPool, true);
   669→   669→        if (!t2) t2 = findBestRelaxed(blocked, slot, allPool);
   670→   670→        if (!t2) t2 = findBestForceDay(blocked, slot, allPool);
   671→   671→
   672→   672→        if (t2) {
   673→   673→          if (t2.subject === 'Admin') {
   674→   674→            adminAssignmentCount[t2.id] = (adminAssignmentCount[t2.id] || 0) + 1;
   675→   675→            totalAdminAssigned++;
   676→   676→          }
   677→   677→          const tr = tracking[t2.id];
   678→   678→          tr.totalComm++; tr.dayComm[slot.day]++;
   679→   679→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   680→   680→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   681→   681→        } else { standbyCount++; }
   682→   682→
   683→   683→        // Record assignment
   684→   684→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   685→   685→        if (session) {
   686→   686→          session.committees.push({
   687→   687→            serial: slot.comId,
   688→   688→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   689→   689→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   690→   690→          });
   691→   691→        }
   692→   692→      }
   693→   693→    }
   694→   694→
   695→   695→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   696→   696→    // This eliminates any tracking accumulation bugs
   697→   697→    const teacherCommCount: Record<string, number> = {};
   698→   698→    const teacherHoursFromAssign: Record<string, number> = {};
   699→   699→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   700→   700→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   701→   701→    teachers.forEach(t => {
   702→   702→      teacherCommCount[t.id] = 0;
   703→   703→      teacherHoursFromAssign[t.id] = 0;
   704→   704→      teacherDayCommFromAssign[t.id] = {};
   705→   705→      teacherSlotsFromAssign[t.id] = [];
   706→   706→    });
   707→   707→
   708→   708→    for (const day of DAYS) {
   709→   709→      for (const sess of finalAssignments[day]) {
   710→   710→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   711→   711→        for (const c of sess.committees) {
   712→   712→          [c.t1, c.t2].forEach(who => {
   713→   713→            if (!who.id) return;
   714→   714→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   715→   715→            // Deduplicate: same day + same time = counted once
   716→   716→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   717→   717→            const existing = teacherSlotsFromAssign[who.id];
   718→   718→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   719→   719→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   720→   720→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   721→   721→            }
   722→   722→            // Day committee count
   723→   723→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   724→   724→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   725→   725→          });
   726→   726→        }
   727→   727→      }
   728→   728→    }
   729→   729→
   730→   730→    // Override tracking with verified data from assignments
   731→   731→    teachers.forEach(t => {
   732→   732→      if (tracking[t.id]) {
   733→   733→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   734→   734→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   735→   735→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   736→   736→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   737→   737→      }
   738→   738→    });
   739→   739→
   740→   740→    // ---- Post-distribution: Ensure admin gets at least 2 assignments ----
   741→   741→    for (const admin of adminTeachers) {
   742→   742→      const adminComm = tracking[admin.id].totalComm;
   743→   743→      if (adminComm >= adminMinTarget) continue;
   744→   744→      // Try to swap: find a teacher with > average hours who has an assignment
   745→   745→      // that admin could take (passing all constraints)
   746→   746→      const allTeacherHours = teachers.map(t => tracking[t.id]?.totalHours || 0).filter(h => h > 0);
   747→   747→      const avgH = allTeacherHours.length > 0 ? allTeacherHours.reduce((a, b) => a + b, 0) / allTeacherHours.length : 0;
   748→   748→      const overburdened = teachers.filter(t => t.id !== admin.id && tracking[t.id]?.totalHours > avgH);
   749→   749→      // Sort by most overburdened first
   750→   750→      overburdened.sort((a, b) => (tracking[b.id]?.totalHours || 0) - (tracking[a.id]?.totalHours || 0));
   751→   751→
   752→   752→      for (const victim of overburdened) {
   753→   753→        if (tracking[admin.id].totalComm >= adminMinTarget) break;
   754→   754→        // Find an assignment of victim that admin can take
   755→   755→        for (const day of DAYS) {
   756→   756→          if (tracking[admin.id].totalComm >= adminMinTarget) break;
   757→   757→          const daySlots = teacherSlotsFromAssign[victim.id].filter(s => s.day === day);
   758→   758→          if (daySlots.length === 0) continue;
   759→   759→          // Admin already has assignment on this day?
   760→   760→          if ((tracking[admin.id].dayComm[day] || 0) >= 1) continue;
   761→   761→          // Find the session+committee for this slot
   762→   762→          for (const sess of finalAssignments[day]) {
   763→   763→            if (tracking[admin.id].totalComm >= adminMinTarget) break;
   764→   764→            const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   765→   765→            if (daySlots[0].start !== sessTimeInfo.start || daySlots[0].end !== sessTimeInfo.end) continue;
   766→   766→            // Check admin constraints
   767→   767→            if (ruleSubject && sess.subject && admin.subject === sess.subject) continue;
   768→   768→            if (!canSuperviseStage(admin, getStage(sess.grade))) continue;
   769→   769→            // Check admin no time overlap (should be fine since we checked dayComm above)
   770→   770→            if (tracking[admin.id].assignedSlots.some(s => s.day === day && !(sessTimeInfo.end <= s.start || sessTimeInfo.start >= s.end))) continue;
   771→   771→            // Find which position (t1 or t2) victim has in this session
   772→   772→            for (const c of sess.committees) {
   773→   773→              if (c.t1?.id === victim.id || c.t2?.id === victim.id) {
   774→   774→                // Check consecutive grade constraint
   775→   775→                if (wasSameGradeAdjacent(tracking[admin.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTimeInfo, comId: 0 })) continue;
   776→   776→                // Do the swap
   777→   777→                const isT1 = c.t1?.id === victim.id;
   778→   778→                const swappedTeacher = isT1 ? c.t1 : c.t2;
   779→   779→                if (isT1) { c.t1 = { id: admin.id, name: admin.name }; }
   780→   780→                else { c.t2 = { id: admin.id, name: admin.name }; }
   781→   781→                // Update tracking: remove from victim, add to admin
   782→   782→                tracking[victim.id].totalComm--;
   783→   783→                tracking[victim.id].dayComm[day] = Math.max(0, (tracking[victim.id].dayComm[day] || 0) - 1);
   784→   784→                tracking[admin.id].totalComm++;
   785→   785→                tracking[admin.id].dayComm[day] = (tracking[admin.id].dayComm[day] || 0) + 1;
   786→   786→                tracking[admin.id].assignedSlots.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   787→   787→                tracking[admin.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   788→   788→                break;
   789→   789→              }
   790→   790→            }
   791→   791→          }
   792→   792→        }
   793→   793→      }
   794→   794→    }
   795→   795→
   796→   796→    // ---- Post-distribution: BALANCE PASS ----
   797→   797→    // Iteratively swap assignments from overburdened → underburdened teachers
   798→   798→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   799→   799→    const BALANCE_ITERATIONS = 300;
   800→   800→    const recalcHours = () => {
   801→   801→      teachers.forEach(t => {
   802→   802→        const slots = tracking[t.id].assignedSlots || [];
   803→   803→        const seen = new Set<string>();
   804→   804→        let hrs = 0;
   805→   805→        slots.forEach(s => {
   806→   806→          const key = s.day + '_' + s.start + '_' + s.end;
   807→   807→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   808→   808→        });
   809→   809→        tracking[t.id].totalHours = hrs;
   810→   810→      });
   811→   811→    };
   812→   812→    recalcHours(); // recalc after admin swaps first
   813→   813→
   814→   814→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   815→   815→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   816→   816→      if (active.length < 2) break;
   817→   817→      // Sort by HOURS first (primary), then committees (secondary)
   818→   818→      const sorted = [...active].sort((a, b) => {
   819→   819→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   820→   820→        if (Math.abs(hDiff) > 0.1) return hDiff;
   821→   821→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   822→   822→      });
   823→   823→      const maxH = tracking[sorted[0].id].totalHours;
   824→   824→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   825→   825→      if (maxH - minH <= 1.0) break; // spread ≤ 1h — good enough
   826→   826→
   827→   827→      let swapped = false;
   828→   828→      // Try top-3 donors × bottom-3 recipients
   829→   829→      for (let di = 0; di < Math.min(sorted.length, 3) && !swapped; di++) {
   830→   830→        const donor = sorted[di];
   831→   831→        // Don't strip admins below their minimum
   832→   832→        const isDonorAdmin = donor.subject === 'Admin';
   833→   833→        if (tracking[donor.id].totalComm <= 1) continue;
   834→   834→        if (isDonorAdmin && tracking[donor.id].totalComm <= adminMinTarget) continue;
   835→   835→
   836→   836→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 3) && !swapped; ri--) {
   837→   837→          const recip = sorted[ri];
   838→   838→          if (donor.id === recip.id) continue;
   839→   839→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   840→   840→            // Also check if committees would be more balanced after swap
   841→   841→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   842→   842→            const recipCommAfter = tracking[recip.id].totalComm + 1;
   843→   843→            // Skip if swap would make committee imbalance worse (donor already has fewer comms)
   844→   844→
   845→   845→          // Try each day where donor has an assignment and recipient doesn't
   846→   846→          for (const day of DAYS) {
   847→   847→            if (swapped) break;
   848→   848→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   849→   849→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   850→   850→
   851→   851→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   852→   852→            if (!donorSlot) continue;
   853→   853→
   854→   854→            // Find the session matching donor's slot
   855→   855→            for (const sess of finalAssignments[day]) {
   856→   856→              if (swapped) break;
   857→   857→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   858→   858→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   859→   859→
   860→   860→              // Check recipient constraints
   861→   861→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   862→   862→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   863→   863→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   864→   864→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   865→   865→
   866→   866→              // Find donor in this session's committees
   867→   867→              for (const c of sess.committees) {
   868→   868→                const isT1 = c.t1?.id === donor.id;
   869→   869→                const isT2 = c.t2?.id === donor.id;
   870→   870→                if (!isT1 && !isT2) continue;
   871→   871→
   872→   872→                // Execute swap
   873→   873→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   874→   874→                else c.t2 = { id: recip.id, name: recip.name };
   875→   875→
   876→   876→                tracking[donor.id].totalComm--;
   877→   877→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   878→   878→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   879→   879→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   880→   880→
   881→   881→                tracking[recip.id].totalComm++;
   882→   882→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   883→   883→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   884→   884→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   885→   885→
   886→   886→                swapped = true;
   887→   887→                break;
   888→   888→              }
   889→   889→            }
   890→   890→          }
   891→   891→        }
   892→   892→      }
   893→   893→      if (!swapped) break; // no valid swap found
   894→   894→      recalcHours(); // update hours after each swap
   895→   895→    }
   896→   896→
   897→   897→    // Final hour recalculation after all balancing
   898→   898→    recalcHours();
   899→   899→
   900→   900→    // ---- Sort committees by serial within each session (1, 2, 3...) ----
   901→   901→    for (const day of DAYS) {
   902→   902→      for (const sess of finalAssignments[day]) {
   903→   903→        sess.committees.sort((a, b) => a.serial - b.serial);
   904→   904→      }
   905→   905→    }
   906→   906→
   907→   907→    // ---- Post-distribution: STANDBY ASSIGNMENT ----
   908→   908→    // 1 standby teacher per stage per day from unassigned teachers
   909→   909→    // CRITICAL: Standby must NOT be the same subject as any exam that day for that stage
   910→   910→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
   911→   911→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
   912→   912→    const STANDBY_PER_STAGE = 1;
   913→   913→    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
   914→   914→    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
   915→   915→
   916→   916→    // Track standby count per teacher for rotation fairness
   917→   917→    const teacherStandbyCount: Record<string, number> = {};
   918→   918→    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
   919→   919→
   920→   920→    // Track which teachers are already standby for another stage today
   921→   921→    const todayStandby: Record<string, Set<string>> = {};
   922→   922→    DAYS.forEach(d => { todayStandby[d] = new Set(); });
   923→   923→
   924→   924→    for (let di = 0; di < DAYS.length; di++) {
   925→   925→      const day = DAYS[di];
   926→   926→
   927→   927→      for (const stage of STAGES_LIST) {
   928→   928→        // Only assign standby if there are exams for this stage on this day
   929→   929→        const stageSessions = (finalAssignments[day] || []).filter(s => getStage(s.grade) === stage);
   930→   930→        if (stageSessions.length === 0) continue;
   931→   931→
   932→   932→        // Collect subjects being examined today for this stage
   933→   933→        const dayStageSubjects = new Set<string>();
   934→   934→        stageSessions.forEach(s => { if (s.subject) dayStageSubjects.add(s.subject); });
   935→   935→
   936→   936→        // Find teachers NOT assigned on this day who can supervise this stage
   937→   937→        // AND whose subject doesn't match any exam subject for this stage today
   938→   938→        let candidates = teachers.filter(t => {
   939→   939→          if ((tracking[t.id].dayComm[day] || 0) >= 1) return false;
   940→   940→          if (todayStandby[day].has(t.id)) return false; // already standby for another stage today
   941→   941→          if (!canSuperviseStage(t, stage)) return false;
   942→   942→          // CRITICAL: Don't pick a teacher whose subject is being examined today
   943→   943→          if (ruleSubject && t.subject && dayStageSubjects.has(t.subject)) return false;
   944→   944→          return true;
   945→   945→        });
   946→   946→
   947→   947→        // Rotation: deprioritize teachers who were standby YESTERDAY for ANY stage
   948→   948→        const yesterdayStandby = new Set<string>();
   949→   949→        if (di > 0) {
   950→   950→          const prevDay = DAYS[di - 1];
   951→   951→          for (const st of STAGES_LIST) {
   952→   952→            (standbys[prevDay][st] || []).forEach(s => yesterdayStandby.add(s.id));
   953→   953→          }
   954→   954→        }
   955→   955→        candidates.sort((a, b) => {
   956→   956→          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
   957→   957→          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
   958→   958→          return (teacherStandbyCount[a.id] + aWasYesterday) - (teacherStandbyCount[b.id] + bWasYesterday);
   959→   959→        });
   960→   960→
   961→   961→        // Pick up to STANDBY_PER_STAGE
   962→   962→        const picked = candidates.slice(0, STANDBY_PER_STAGE);
   963→   963→        standbys[day][stage] = picked.map(t => ({ id: t.id, name: t.name }));
   964→   964→
   965→   965→        // Track rotation count and mark as standby for today (so same teacher isn't picked for another stage)
   966→   966→        for (const s of picked) {
   967→   967→          teacherStandbyCount[s.id]++;
   968→   968→          todayStandby[day].add(s.id);
   969→   969→        }
   970→   970→      }
   971→   971→    }
   972→   972→
   973→   973→    // ---- Post-distribution verification ----
   974→   974→    const violations: string[] = [];
   975→   975→    // V-Check 1: Own-subject supervision
   976→   976→    if (ruleSubject) {
   977→   977→      for (const day of DAYS) {
   978→   978→        for (const sess of finalAssignments[day]) {
   979→   979→          for (const c of sess.committees) {
   980→   980→            [c.t1, c.t2].forEach(who => {
   981→   981→              if (who.id) {
   982→   982→                const tch = teachers.find(x => x.id === who.id);
   983→   983→                if (tch && tch.subject === sess.subject) violations.push(`${tch.name} -> own subject (${sess.subject}) on ${day}`);
   984→   984→              }
   985→   985→            });
   986→   986→          }
   987→   987→        }
   988→   988→      }
   989→   989→    }
   990→   990→    // V-Check 2: Time overlap
   991→   991→    for (const t of teachers) {
   992→   992→      const tr = tracking[t.id];
   993→   993→      for (let i = 0; i < tr.assignedSlots.length; i++) {
   994→   994→        for (let j = i + 1; j < tr.assignedSlots.length; j++) {
   995→   995→          const a = tr.assignedSlots[i], b = tr.assignedSlots[j];
   996→   996→          if (a.day === b.day && !(a.end <= b.start || b.end <= a.start)) {
   997→   997→            violations.push(`${t.name} -> time overlap on ${a.day}`);
   998→   998→          }
   999→   999→        }
  1000→  1000→      }
  1001→  1001→    }
  1002→  1002→    // V-Check 3: Same-day double assignment (should NEVER happen now)
  1003→  1003→    for (const t of teachers) {
  1004→  1004→      const tr = tracking[t.id];
  1005→  1005→      for (const day of DAYS) {
  1006→  1006→        if ((tr.dayComm[day] || 0) > 1) {
  1007→  1007→          violations.push(`${t.name} -> 2+ committees on ${day} (HARD RULE VIOLATED)`);
  1008→  1008→        }
  1009→  1009→      }
  1010→  1010→    }
  1011→  1011→    // V-Check 4: Stage notes compliance
  1012→  1012→    if (ruleNotes) {
  1013→  1013→      for (const day of DAYS) {
  1014→  1014→        for (const sess of finalAssignments[day]) {
  1015→  1015→          const stage = getStage(sess.grade);
  1016→  1016→          for (const c of sess.committees) {
  1017→  1017→            [c.t1, c.t2].forEach(who => {
  1018→  1018→              if (who.id) {
  1019→  1019→                const tch = teachers.find(x => x.id === who.id);
  1020→  1020→                if (tch && !canSuperviseStage(tch, stage)) violations.push(`${tch.name} -> wrong stage (${stage}) notes="${tch.notes}" on ${day}`);
  1021→  1021→              }
  1022→  1022→            });
  1023→  1023→          }
  1024→  1024→        }
  1025→  1025→      }
  1026→  1026→    }
  1027→  1027→
  1028→  1028→    // ---- Build summary statistics (from verified tracking) ----
  1029→  1029→    const allH = teachers.map(t => tracking[t.id]?.totalHours || 0);
  1030→  1030→    const avgAll = allH.length ? allH.reduce((a, b) => a + b, 0) / allH.length : 0;
  1031→  1031→    const spread = allH.length ? Math.sqrt(allH.reduce((s, h) => s + (h - avgAll) ** 2, 0) / allH.length) : 0;
  1032→  1032→    let msg = `v11 | Avg: ${avgAll.toFixed(1)}h | Spread: ${spread.toFixed(1)} | Min: ${allH.length ? Math.min(...allH).toFixed(1) : 0}h | Max: ${allH.length ? Math.max(...allH).toFixed(1) : 0}h`;
  1033→  1033→    // Count total standby assigned
  1034→  1034→    const totalStandby = Object.values(standbys).reduce((a, daySt) => a + Object.values(daySt).reduce((b, stList) => b + stList.length, 0), 0);
  1035→  1035→    if (totalStandby > 0) msg += ` | ${totalStandby} standby (${STANDBY_PER_STAGE}/stage/day)`;
  1036→  1036→    if (standbyCount > 0) msg += ` | ${standbyCount} unfilled`;
  1037→  1037→    if (violations.length > 0) {
  1038→  1038→      msg += ` | ${violations.length} violations (check console)`;
  1039→  1039→      console.warn('[Distribution v11] Violations:', violations);
  1040→  1040→    } else {
  1041→  1041→      console.log('[Distribution v11] All constraints passed!');
  1042→  1042→    }
  1043→  1043→
  1044→  1044→    const newResults: DistributionResults = { _version: 11, assignments: finalAssignments, standbys, tracking };
  1045→  1045→    setResults(newResults);
  1046→  1046→    fetch('/api/results', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ data: newResults }) });
  1047→  1047→    showToast(msg, standbyCount > 0 || violations.length > 0 ? 'error' : 'success');
  1048→  1048→    setActivePage('results');
  1049→  1049→  };
