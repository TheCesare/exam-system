    44→function parseTimeRange(tStr: string) {
    45→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    46→  try {
    47→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    48→    const parts = clean.split('-');
    49→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    50→    let start = toMinutes(parts[0]);
    51→    let end = toMinutes(parts[1]);
    52→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    53→    if (start < 300 && end > start + 120) start += 720;
    54→    if (end <= start) end += 720;
    55→    let duration = (end - start) / 60;
    56→    // Sanity cap: no exam is longer than 5 hours
    57→    if (duration > 5) duration = 2.0; // fallback to standard
    58→    return { start, end, duration };
    59→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    60→}

    62→function getStage(grade: string) {
    63→  const g = grade.toLowerCase();
    64→  if (g.includes('primary')) return 'primary';
    65→  if (g.includes('prep')) return 'prep';
    66→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    67→  return 'any';

   483→  const importCSV = () => {
   484→    const input = document.createElement('input');
   485→    input.type = 'file'; input.accept = '.csv';
   486→    input.onchange = async (e) => {
   487→      const file = (e.target as HTMLInputElement).files?.[0];
   488→      if (!file) return;
   489→      const text = await file.text();
   490→      const lines = text.split('\n');
   491→      const imported: { name: string; subject: string; notes: string }[] = [];
   492→      lines.forEach(line => {
   493→        const parts = line.split(',').map(p => p.trim());
   494→        if (parts.length >= 2 && parts[0]) imported.push({ name: parts[0], subject: parts[1], notes: parts[2] || '' });

   538→  // ========== DISTRIBUTION ENGINE v4 ==========
   539→  const runDistribution = () => {
   540→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   541→
   542→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   543→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   544→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   545→    const includeAdminW2 = (document.getElementById('rule-admin-w2') as HTMLInputElement)?.checked ?? false;
   546→
   547→    // ---- Admin handling: Admin subject excluded by default, included only for W2 peak days ----
   548→
   549→    // ---- Initialize tracking ----

   554→    });
   555→
   556→    // ---- Build exam slots grouped by day (chronological order) ----
   557→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   558→    const slotsByDay: Record<string, Slot[]> = {};
   559→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   560→    DAYS.forEach((day, dayIndex) => {
   561→      GRADES.forEach(grade => {
   562→        const cell = scheduleBuffer[`${grade}__${day}`];
   563→        if (cell && cell.committees > 0) {
   564→          const timeInfo = parseTimeRange(cell.time || DEFAULT_TIMES[grade] || '9:00-10:30');
   565→          for (let c = 1; c <= cell.committees; c++) {
   566→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   567→          }
   568→        }
   569→      });
   570→    });
   571→
   572→    // ---- Build final assignments structure ----
   573→    const finalAssignments: Record<string, SessionResult[]> = {};
   574→    DAYS.forEach(d => { finalAssignments[d] = []; });
   575→    DAYS.forEach(day => {
   576→      GRADES.forEach(grade => {
   577→        const cell = scheduleBuffer[`${grade}__${day}`];
   578→        if (cell && cell.committees > 0) {
   579→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   580→          if (!existing) {
   581→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   582→          }
   583→        }
   584→      });
   585→    });
   586→
   587→    // ---- Helper: Stage notes — HARD constraint ----
   588→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   589→      if (!ruleNotes) return true;
   590→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   591→      const n = teacher.notes.toLowerCase();
   592→      const allowed: string[] = [];
   593→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   594→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   595→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   596→      if (allowed.length === 0) return true;
   597→      return allowed.includes(slotStage);
   598→    };
   599→
   600→    // ---- Helper: Consecutive-day same-grade check ----
   601→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   602→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   603→    };
   604→
   605→    // ---- Helper: Special subject pairing rules ----
   606→    // Religion exam: Arabic + Religion teachers blocked
   607→    // Arabic exam: Religion teachers blocked from primary only
   608→    const isBlockedBySpecialRule = (teacher: Teacher, slot: Slot): boolean => {
   609→      if (slot.subject === 'Religion' && (teacher.subject === 'Arabic' || teacher.subject === 'Religion')) return true;
   610→      if (slot.subject === 'Arabic' && teacher.subject === 'Religion' && slot.stage === 'primary') return true;
   611→      return false;
   612→    };
   613→
   614→    // ---- Helper: Fisher-Yates shuffle ----
   615→    const shuffle = <T,>(arr: T[]): T[] => {
   616→      const a = [...arr];
   617→      for (let i = a.length - 1; i > 0; i--) {
   618→        const j = Math.floor(Math.random() * (i + 1));
   619→        [a[i], a[j]] = [a[j], a[i]];
   620→      }
   621→      return a;
   622→    };
   623→
   624→    // ---- Helper: Special subject exclusion rules (Religion/Arabic) ----
   625→    // Rule 1: If exam subject is Religion (الدين), teachers with subject Arabic (العربي) are excluded (Religion teachers already caught by own-subject)
   626→    // Rule 2: If exam subject is Arabic (العربي), teachers with subject Religion (الدين) are excluded for primary stage only
   627→    const isSubjectExcluded = (t: Teacher, slot: Slot): boolean => {
   628→      if (!ruleSubject || !slot.subject) return false;
   629→      if (slot.subject === 'الدين') {
   630→        return t.subject === 'العربي';
   631→      }
   632→      if (slot.subject === 'العربي' && slot.stage === 'primary') {
   633→        return t.subject === 'الدين';
   634→      }
   635→      return false;
   636→    };
   637→
   638→    // ---- Core: Find best teacher for a slot ----
   639→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   640→    // Scoring: hours dominate (500x), tiny noise (0.1) for variety
   641→    const findBest = (
   642→      blockedId: string | null,
   643→      slot: Slot,
   644→      pool: Teacher[],
   645→      relaxAdj: boolean
   646→    ): Teacher | null => {
   647→      const candidates: { teacher: Teacher; score: number }[] = [];
   648→      for (const t of pool) {
   649→        if (blockedId && t.id === blockedId) continue;
   650→        const tr = tracking[t.id];
   651→        // HARD: No own-subject supervision
   652→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   653→        // HARD: Special subject exclusion (Religion/Arabic rules)
   654→        if (isSubjectExcluded(t, slot)) continue;
   655→        // HARD: Stage notes filtering
   656→        if (!canSuperviseStage(t, slot.stage)) continue;
   657→        // HARD: No time overlap on same day
   658→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   659→        // HARD: MAX 2 committees per teacher per day (NEVER relaxed)
   660→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   661→        // SOFT: No same grade on consecutive days (can be relaxed)
   662→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   663→        // Scoring: HOURS dominate (500x), then committees (10x), tiny noise for variety
   664→        // Admin subject = HEAVY penalty (always last choice)
   665→        const isAdmin = t.subject === 'Admin';
   666→        const adminPenalty = isAdmin ? 50000 : 0;
   667→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   668→        candidates.push({ teacher: t, score });
   669→      }
   670→      if (candidates.length === 0) return null;
   671→      candidates.sort((a, b) => a.score - b.score);
   672→      const best = candidates[0].score;
   673→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   674→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   675→    };
   676→
   677→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 2-per-day + time) ----
   678→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   679→      const candidates: { teacher: Teacher; score: number }[] = [];
   680→      for (const t of pool) {
   681→        if (blockedId && t.id === blockedId) continue;
   682→        const tr = tracking[t.id];
   683→        // HARD: Still no own-subject supervision even in relaxed mode
   684→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   685→        // HARD: Special subject exclusion (Religion/Arabic rules) — never relaxed
   686→        if (isSubjectExcluded(t, slot)) continue;
   687→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   688→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   689→        const isAdmin = t.subject === 'Admin';
   690→        const adminPenalty = isAdmin ? 50000 : 0;
   691→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   692→        candidates.push({ teacher: t, score });
   693→      }
   694→      if (candidates.length === 0) return null;
   695→      candidates.sort((a, b) => a.score - b.score);
   696→      return candidates[0].teacher;
   697→    };
   698→
   699→    // ---- Fallback 3: Allow 2nd assignment same day (time overlap + own-subject + max 2/day blocked) ----
   700→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   701→      const candidates: { teacher: Teacher; score: number }[] = [];
   702→      for (const t of pool) {
   703→        if (blockedId && t.id === blockedId) continue;
   704→        const tr = tracking[t.id];
   705→        // HARD: Still no own-subject supervision even in force-day mode
   706→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   707→        // HARD: Special subject exclusion (Religion/Arabic rules) — never relaxed
   708→        if (isSubjectExcluded(t, slot)) continue;
   709→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   710→        // HARD: ABSOLUTE MAX 2 committees per teacher per day (never exceeded)
   711→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   712→        const isAdmin = t.subject === 'Admin';
   713→        const adminPenalty = isAdmin ? 50000 : 0;
   714→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (tr.dayComm[slot.day] || 0) * 100 + adminPenalty + Math.random() * 0.1;
   715→        candidates.push({ teacher: t, score });
   716→      }
   717→      if (candidates.length === 0) return null;
   718→      candidates.sort((a, b) => a.score - b.score);
   719→      return candidates[0].teacher;
   720→    };
   721→
   722→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   723→    let standbyCount = 0;
   724→
   725→    // ---- 'Old' teacher handling: senior teachers excluded from default pool, used as last resort ----
   726→    const isOldTeacher = (t: Teacher) => !!(t.notes && t.notes.toLowerCase().includes('old'));
   727→    const oldTeachers = teachers.filter(t => t.subject !== 'Admin' && isOldTeacher(t));
   728→    const nonAdminTeachers = teachers.filter(t => t.subject !== 'Admin' && !isOldTeacher(t));
   729→
   730→    // Track old teacher supervision days (max 3, min 1)
   731→    const oldTeacherDays: Record<string, number> = {};
   732→    oldTeachers.forEach(t => { oldTeacherDays[t.id] = 0; });
   733→
   734→    // Calculate Week 2 peak days (days with above-average committee count)
   735→    const w2SlotCounts = WEEK2_DAYS.map(d => slotsByDay[d].length).filter(c => c > 0);
   736→    const w2Avg = w2SlotCounts.length > 0 ? w2SlotCounts.reduce((a, b) => a + b, 0) / w2SlotCounts.length : 0;
   737→    const peakW2Days = new Set(WEEK2_DAYS.filter(d => slotsByDay[d].length >= w2Avg && slotsByDay[d].length > 0));
   738→
   739→    for (const day of DAYS) {
   740→      // Determine pool: Admin only included on W2 peak days when toggle is ON
   741→      const isW2Peak = WEEK2_DAYS.includes(day) && peakW2Days.has(day);
   742→      const dayPool = (includeAdminW2 && isW2Peak) ? [...teachers] : nonAdminTeachers;
   743→      const daySlots = shuffle(slotsByDay[day]);
   744→      if (daySlots.length === 0) continue;
   745→
   746→      for (const slot of daySlots) {
   747→        let t1: Teacher | null = null;
   748→        let t2: Teacher | null = null;
   749→
   750→        // Helper: filter old teachers not exceeding 3-day max and not already assigned today
   751→        const availableOldTeachers = (blockedId: string | null, day: string) => {
   752→          return oldTeachers.filter(ot => {
   753→            if (blockedId && ot.id === blockedId) return false;
   754→            if ((oldTeacherDays[ot.id] || 0) >= 3) return false;
   755→            const tr = tracking[ot.id];
   756→            if (tr.assignedSlots.some(s => s.day === day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) return false;
   757→            return true;
   758→          });
   759→        };
   760→
   761→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day → OLD teacher last resort
   762→        t1 = findBest(null, slot, dayPool, false);
   763→        if (!t1) t1 = findBest(null, slot, dayPool, true); // relax consecutive-day
   764→        if (!t1) t1 = findBestRelaxed(null, slot, dayPool); // relax subject + stage
   765→        if (!t1) t1 = findBestForceDay(null, slot, dayPool); // allow 2nd assignment same day
   766→        // Last resort: try old teachers (max 3 days)
   767→        if (!t1) t1 = findBest(null, slot, availableOldTeachers(null, slot.day), false);
   768→        if (!t1) t1 = findBestRelaxed(null, slot, availableOldTeachers(null, slot.day));
   769→
   770→        if (t1) {
   771→          const tr = tracking[t1.id];
   772→          tr.totalComm++; tr.dayComm[slot.day]++;
   773→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   774→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   775→          // Track old teacher days
   776→          if (isOldTeacher(t1) && !tr.assignedSlots.some((s, idx) => idx < tr.assignedSlots.length - 1 && s.day === slot.day)) {
   777→            oldTeacherDays[t1.id] = (oldTeacherDays[t1.id] || 0) + 1;
   778→          }
   779→        } else { standbyCount++; }
   780→
   781→        // T2: pairing rule — classified teacher (has notes) must pair with unclassified
   782→        const blocked = t1?.id || null;
   783→        const t1Classified = t1 ? !!(t1.notes && t1.notes.trim() !== '') : false;
   784→        const t2Filtered = t1Classified ? dayPool.filter(t => t.id !== blocked && (!t.notes || t.notes.trim() === '')) : dayPool;
   785→
   786→        t2 = findBest(blocked, slot, t2Filtered, false);
   787→        if (!t2) t2 = findBest(blocked, slot, t2Filtered, true);
   788→        if (!t2) t2 = findBestRelaxed(blocked, slot, t2Filtered);
   789→        if (!t2) t2 = findBestForceDay(blocked, slot, t2Filtered);
   790→        // Relax pairing rule if no unclassified teacher available
   791→        if (!t2 && t1Classified) {
   792→          t2 = findBest(blocked, slot, dayPool, false);
   793→          if (!t2) t2 = findBest(blocked, slot, dayPool, true);
   794→          if (!t2) t2 = findBestRelaxed(blocked, slot, dayPool);
   795→          if (!t2) t2 = findBestForceDay(blocked, slot, dayPool);
   796→        }
   797→        // Last resort: try old teachers for T2
   798→        if (!t2) {
   799→          const oldPool = availableOldTeachers(blocked, slot.day);
   800→          t2 = findBest(blocked, slot, oldPool, false);
   801→          if (!t2) t2 = findBestRelaxed(blocked, slot, oldPool);
   802→        }
   803→
   804→        if (t2) {
   805→          const tr = tracking[t2.id];
   806→          tr.totalComm++; tr.dayComm[slot.day]++;
   807→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   808→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   809→          // Track old teacher days
   810→          if (isOldTeacher(t2) && !tr.assignedSlots.some((s, idx) => idx < tr.assignedSlots.length - 1 && s.day === slot.day)) {
   811→            oldTeacherDays[t2.id] = (oldTeacherDays[t2.id] || 0) + 1;
   812→          }
   813→        } else { standbyCount++; }
   814→
   815→        // Record assignment
   816→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   817→        if (session) {
   818→          session.committees.push({
   819→            serial: slot.comId,
   820→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   821→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   822→          });
   823→        }
   824→      }
   825→    }
   826→
   827→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   828→    // This eliminates any tracking accumulation bugs
   829→    const teacherCommCount: Record<string, number> = {};
   830→    const teacherHoursFromAssign: Record<string, number> = {};
   831→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   832→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   833→    teachers.forEach(t => {
   834→      teacherCommCount[t.id] = 0;
   835→      teacherHoursFromAssign[t.id] = 0;
   836→      teacherDayCommFromAssign[t.id] = {};
   837→      teacherSlotsFromAssign[t.id] = [];
   838→    });
   839→
   840→    for (const day of DAYS) {
   841→      for (const sess of finalAssignments[day]) {
   842→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   843→        for (const c of sess.committees) {
   844→          [c.t1, c.t2].forEach(who => {
   845→            if (!who.id) return;
   846→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   847→            // Deduplicate: same day + same time = counted once
   848→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   849→            const existing = teacherSlotsFromAssign[who.id];
   850→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   851→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   852→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   853→            }
   854→            // Day committee count
   855→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   856→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   857→          });
   858→        }
   859→      }
   860→    }
   861→
   862→    // Override tracking with verified data from assignments
   863→    teachers.forEach(t => {
   864→      if (tracking[t.id]) {
   865→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   866→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   867→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   868→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   869→      }
   870→    });
   871→
   872→    // ---- Post-distribution: Admin already handled by pool exclusion ----
   873→
   874→    // ---- Post-distribution: BALANCE PASS ----
   875→    // Iteratively swap assignments from overburdened → underburdened teachers
   876→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   877→    const BALANCE_ITERATIONS = 500;
   878→    const recalcHours = () => {
   879→      teachers.forEach(t => {
   880→        const slots = tracking[t.id].assignedSlots || [];
   881→        const seen = new Set<string>();
   882→        let hrs = 0;
   883→        slots.forEach(s => {
   884→          const key = s.day + '_' + s.start + '_' + s.end;
   885→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   886→        });
   887→        tracking[t.id].totalHours = hrs;
   888→      });
   889→    };
   890→    recalcHours(); // recalc after admin swaps first
   891→
   892→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   893→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   894→      if (active.length < 2) break;
   895→      // Sort by HOURS first (primary), then committees (secondary)
   896→      const sorted = [...active].sort((a, b) => {
   897→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   898→        if (Math.abs(hDiff) > 0.1) return hDiff;
   899→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   900→      });
   901→      const maxH = tracking[sorted[0].id].totalHours;
   902→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   903→      if (maxH - minH <= 0.5) break; // spread ≤ 0.5h — tight balance

   920→          if (isOldTeacher(recip) && (oldTeacherDays[recip.id] || 0) >= 3) continue;

   923→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   924→            const recipCommAfter = tracking[recip.id].totalComm + 1;

   927→          // Try each day where donor has an assignment and recipient doesn't

   929→            if (swapped) break;
   930→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   931→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   932→
   933→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   934→            if (!donorSlot) continue;
   935→
   936→            // Find the session matching donor's slot
   937→            for (const sess of finalAssignments[day]) {
   938→              if (swapped) break;

   948→
   949→              // Find donor in this session's committees

   951→                const isT1 = c.t1?.id === donor.id;
   952→                const isT2 = c.t2?.id === donor.id;
