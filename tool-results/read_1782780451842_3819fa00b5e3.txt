    46→function parseTimeRange(tStr: string) {
    47→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
    48→  try {
    49→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
    50→    const parts = clean.split('-');
    51→    const toMinutes = (s: string) => { const p = s.split(':'); return (parseInt(p[0])||0)*60 + (parseInt(p[1])||0); };
    52→    let start = toMinutes(parts[0]);
    53→    let end = toMinutes(parts[1]);
    54→    // Fix common 12h format issue: "1:00" should be "13:00" if start < 300 (before 5am) and end > start
    55→    if (start < 300 && end > start + 120) start += 720;
    56→    if (end <= start) end += 720;
    57→    let duration = (end - start) / 60;
    58→    // Sanity cap: no exam is longer than 5 hours
    59→    if (duration > 5) duration = 2.0; // fallback to standard
    60→    return { start, end, duration };
    61→  } catch { return { start: 540, end: 630, duration: 1.5 }; }
    62→}

    64→function getStage(grade: string) {
    65→  const g = grade.toLowerCase();
    66→  if (g.includes('primary')) return 'primary';
    67→  if (g.includes('prep')) return 'prep';
    68→  if (g.includes('secondary') || g.includes('sec')) return 'sec';
    69→  return 'any';

   540→  // ========== DISTRIBUTION ENGINE v4 ==========
   541→  const runDistribution = () => {
   542→    if (teachers.length === 0) { showToast('Error: Registry empty!', 'error'); return; }
   543→
   544→    const ruleSubject = (document.getElementById('rule-subject') as HTMLInputElement)?.checked ?? true;
   545→    const ruleDayLimit = (document.getElementById('rule-daylimit') as HTMLInputElement)?.checked ?? true;
   546→    const ruleNotes = (document.getElementById('rule-notes') as HTMLInputElement)?.checked ?? true;
   547→    const includeAdminW2 = (document.getElementById('rule-admin-w2') as HTMLInputElement)?.checked ?? false;
   548→
   549→    // ---- Admin handling: Admin subject excluded by default, included only for W2 peak days ----
   550→
   551→    // ---- Initialize tracking ----

   557→
   558→    // ---- Build exam slots grouped by day (chronological order) ----
   559→    type Slot = { day: string; dayIndex: number; grade: string; stage: string; subject: string; time: string; timeInfo: ReturnType<typeof parseTimeRange>; comId: number };
   560→    const slotsByDay: Record<string, Slot[]> = {};
   561→    DAYS.forEach((d, i) => { slotsByDay[d] = []; });
   562→    DAYS.forEach((day, dayIndex) => {

   567→          for (let c = 1; c <= cell.committees; c++) {
   568→            slotsByDay[day].push({ day, dayIndex, grade, stage: getStage(grade), subject: cell.subject || '', time: cell.time || DEFAULT_TIMES[grade] || '9:00-10:30', timeInfo, comId: c });
   569→          }
   570→        }
   571→      });
   572→    });
   573→
   574→    // ---- Build final assignments structure ----
   575→    const finalAssignments: Record<string, SessionResult[]> = {};
   576→    DAYS.forEach(d => { finalAssignments[d] = []; });
   577→    DAYS.forEach(day => {
   578→      GRADES.forEach(grade => {
   579→        const cell = scheduleBuffer[`${grade}__${day}`];
   580→        if (cell && cell.committees > 0) {
   581→          const existing = finalAssignments[day].find(s => s.grade === grade && s.time === (cell.time || DEFAULT_TIMES[grade]));
   582→          if (!existing) {
   583→            finalAssignments[day].push({ grade, time: cell.time || DEFAULT_TIMES[grade] || '', subject: cell.subject || '', committees: [] });
   584→          }
   585→        }
   586→      });
   587→    });
   588→
   589→    // ---- Helper: Stage notes — HARD constraint ----
   590→    const canSuperviseStage = (teacher: Teacher, slotStage: string): boolean => {
   591→      if (!ruleNotes) return true;
   592→      if (!teacher.notes || teacher.notes.trim() === '') return true;
   593→      const n = teacher.notes.toLowerCase();
   594→      const allowed: string[] = [];
   595→      if (/\b(primary|pri)\b/i.test(n) || n.includes('\u0627\u0628\u062a\u062f\u0627\u0626\u064a')) allowed.push('primary');
   596→      if (/\bprep\b/i.test(n) || n.includes('\u0627\u0639\u062f\u0627\u062f\u064a')) allowed.push('prep');
   597→      if (/\b(secondary|sec)\b/i.test(n) || n.includes('\u062b\u0627\u0646\u0648\u064a')) allowed.push('sec');
   598→      if (allowed.length === 0) return true;
   599→      return allowed.includes(slotStage);
   600→    };
   601→
   602→    // ---- Helper: Consecutive-day same-grade check ----
   603→    const wasSameGradeAdjacent = (tr: TrackingEntry, slot: Slot): boolean => {
   604→      return tr.gradeHistory.some(h => Math.abs(h.dayIndex - slot.dayIndex) === 1 && h.grade === slot.grade);
   605→    };
   606→
   607→    // ---- Helper: Special subject pairing rules ----
   608→    // Religion exam: Arabic + Religion teachers blocked
   609→    // Arabic exam: Religion teachers blocked from primary only
   610→    const isBlockedBySpecialRule = (teacher: Teacher, slot: Slot): boolean => {
   611→      if (slot.subject === 'Religion' && (teacher.subject === 'Arabic' || teacher.subject === 'Religion')) return true;
   612→      if (slot.subject === 'Arabic' && teacher.subject === 'Religion' && slot.stage === 'primary') return true;
   613→      return false;
   614→    };
   615→
   616→    // ---- Helper: Fisher-Yates shuffle ----
   617→    const shuffle = <T,>(arr: T[]): T[] => {
   618→      const a = [...arr];
   619→      for (let i = a.length - 1; i > 0; i--) {
   620→        const j = Math.floor(Math.random() * (i + 1));
   621→        [a[i], a[j]] = [a[j], a[i]];
   622→      }
   623→      return a;
   624→    };
   625→
   626→    // ---- Helper: Special subject exclusion rules (Religion/Arabic) ----
   627→    // Rule 1: If exam subject is Religion (الدين), teachers with subject Arabic (العربي) are excluded (Religion teachers already caught by own-subject)
   628→    // Rule 2: If exam subject is Arabic (العربي), teachers with subject Religion (الدين) are excluded for primary stage only
   629→    const isSubjectExcluded = (t: Teacher, slot: Slot): boolean => {
   630→      if (!ruleSubject || !slot.subject) return false;
   631→      if (slot.subject === 'الدين') {
   632→        return t.subject === 'العربي';
   633→      }
   634→      if (slot.subject === 'العربي' && slot.stage === 'primary') {
   635→        return t.subject === 'الدين';
   636→      }
   637→      return false;
   638→    };
   639→
   640→    // ---- Core: Find best teacher for a slot ----
   641→    // HARD RULES: own-subject, stage notes, NO same-day double, time overlap
   642→    // Scoring: hours dominate (500x), tiny noise (0.1) for variety
   643→    const findBest = (
   644→      blockedId: string | null,
   645→      slot: Slot,
   646→      pool: Teacher[],
   647→      relaxAdj: boolean
   648→    ): Teacher | null => {
   649→      const candidates: { teacher: Teacher; score: number }[] = [];
   650→      for (const t of pool) {
   651→        if (blockedId && t.id === blockedId) continue;
   652→        const tr = tracking[t.id];
   653→        // HARD: No own-subject supervision
   654→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   655→        // HARD: Special subject exclusion (Religion/Arabic rules)
   656→        if (isSubjectExcluded(t, slot)) continue;
   657→        // HARD: Stage notes filtering
   658→        if (!canSuperviseStage(t, slot.stage)) continue;
   659→        // HARD: No time overlap on same day
   660→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   661→        // HARD: MAX 2 committees per teacher per day (NEVER relaxed)
   662→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   663→        // SOFT: No same grade on consecutive days (can be relaxed)
   664→        if (!relaxAdj && wasSameGradeAdjacent(tr, slot)) continue;
   665→        // Scoring: HOURS dominate (500x), then committees (10x), tiny noise for variety
   666→        // Admin subject = HEAVY penalty (always last choice)
   667→        const isAdmin = t.subject === 'Admin';
   668→        const adminPenalty = isAdmin ? 50000 : 0;
   669→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   670→        candidates.push({ teacher: t, score });
   671→      }
   672→      if (candidates.length === 0) return null;
   673→      candidates.sort((a, b) => a.score - b.score);
   674→      const best = candidates[0].score;
   675→      const topGroup = candidates.filter(c => c.score <= best + 0.5);
   676→      return topGroup[Math.floor(Math.random() * topGroup.length)].teacher;
   677→    };
   678→
   679→    // ---- Fallback 2: Relax stage constraints only (keep own-subject, 2-per-day + time) ----
   680→    const findBestRelaxed = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   681→      const candidates: { teacher: Teacher; score: number }[] = [];
   682→      for (const t of pool) {
   683→        if (blockedId && t.id === blockedId) continue;
   684→        const tr = tracking[t.id];
   685→        // HARD: Still no own-subject supervision even in relaxed mode
   686→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   687→        // HARD: Special subject exclusion (Religion/Arabic rules) — never relaxed
   688→        if (isSubjectExcluded(t, slot)) continue;
   689→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   690→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   691→        const isAdmin = t.subject === 'Admin';
   692→        const adminPenalty = isAdmin ? 50000 : 0;
   693→        const score = tr.totalHours * 500 + tr.totalComm * 10 + adminPenalty + Math.random() * 0.1;
   694→        candidates.push({ teacher: t, score });
   695→      }
   696→      if (candidates.length === 0) return null;
   697→      candidates.sort((a, b) => a.score - b.score);
   698→      return candidates[0].teacher;
   699→    };
   700→
   701→    // ---- Fallback 3: Allow 2nd assignment same day (time overlap + own-subject + max 2/day blocked) ----
   702→    const findBestForceDay = (blockedId: string | null, slot: Slot, pool: Teacher[]): Teacher | null => {
   703→      const candidates: { teacher: Teacher; score: number }[] = [];
   704→      for (const t of pool) {
   705→        if (blockedId && t.id === blockedId) continue;
   706→        const tr = tracking[t.id];
   707→        // HARD: Still no own-subject supervision even in force-day mode
   708→        if (ruleSubject && slot.subject && t.subject === slot.subject) continue;
   709→        // HARD: Special subject exclusion (Religion/Arabic rules) — never relaxed
   710→        if (isSubjectExcluded(t, slot)) continue;
   711→        if (tr.assignedSlots.some(s => s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) continue;
   712→        // HARD: ABSOLUTE MAX 2 committees per teacher per day (never exceeded)
   713→        if (ruleDayLimit && tr.dayComm[slot.day] >= 2) continue;
   714→        const isAdmin = t.subject === 'Admin';
   715→        const adminPenalty = isAdmin ? 50000 : 0;
   716→        const score = tr.totalHours * 500 + tr.totalComm * 10 + (tr.dayComm[slot.day] || 0) * 100 + adminPenalty + Math.random() * 0.1;
   717→        candidates.push({ teacher: t, score });
   718→      }
   719→      if (candidates.length === 0) return null;
   720→      candidates.sort((a, b) => a.score - b.score);
   721→      return candidates[0].teacher;
   722→    };
   723→
   724→    // ---- Process each day chronologically, shuffle slots within day for variety ----
   725→    let standbyCount = 0;
   726→
   727→    // ---- 'Old' teacher handling: senior teachers excluded from default pool, used as last resort ----
   728→    const isOldTeacher = (t: Teacher) => !!(t.notes && t.notes.toLowerCase().includes('old'));
   729→    const oldTeachers = teachers.filter(t => t.subject !== 'Admin' && isOldTeacher(t));
   730→    const nonAdminTeachers = teachers.filter(t => t.subject !== 'Admin' && !isOldTeacher(t));
   731→
   732→    // Track old teacher supervision days (max 3, min 1)
   733→    const oldTeacherDays: Record<string, number> = {};
   734→    oldTeachers.forEach(t => { oldTeacherDays[t.id] = 0; });
   735→
   736→    // Calculate Week 2 peak days (days with above-average committee count)
   737→    const w2SlotCounts = WEEK2_DAYS.map(d => slotsByDay[d].length).filter(c => c > 0);
   738→    const w2Avg = w2SlotCounts.length > 0 ? w2SlotCounts.reduce((a, b) => a + b, 0) / w2SlotCounts.length : 0;
   739→    const peakW2Days = new Set(WEEK2_DAYS.filter(d => slotsByDay[d].length >= w2Avg && slotsByDay[d].length > 0));
   740→
   741→    for (const day of DAYS) {
   742→      // Determine pool: Admin only included on W2 peak days when toggle is ON
   743→      const isW2Peak = WEEK2_DAYS.includes(day) && peakW2Days.has(day);
   744→      const dayPool = (includeAdminW2 && isW2Peak) ? [...teachers] : nonAdminTeachers;
   745→      const daySlots = shuffle(slotsByDay[day]);
   746→      if (daySlots.length === 0) continue;
   747→
   748→      for (const slot of daySlots) {
   749→        let t1: Teacher | null = null;
   750→        let t2: Teacher | null = null;
   751→
   752→        // Helper: filter old teachers not exceeding 3-day max and not already assigned today
   753→        const availableOldTeachers = (blockedId: string | null, day: string) => {
   754→          return oldTeachers.filter(ot => {
   755→            if (blockedId && ot.id === blockedId) return false;
   756→            if ((oldTeacherDays[ot.id] || 0) >= 3) return false;
   757→            const tr = tracking[ot.id];
   758→            if (tr.assignedSlots.some(s => s.day === day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end))) return false;
   759→            return true;
   760→          });
   761→        };
   762→
   763→        // T1: strict → relax consecutive → relax subject+stage → allow 2nd same-day → OLD teacher last resort
   764→        t1 = findBest(null, slot, dayPool, false);
   765→        if (!t1) t1 = findBest(null, slot, dayPool, true); // relax consecutive-day
   766→        if (!t1) t1 = findBestRelaxed(null, slot, dayPool); // relax subject + stage
   767→        if (!t1) t1 = findBestForceDay(null, slot, dayPool); // allow 2nd assignment same day
   768→        // Last resort: try old teachers (max 3 days)
   769→        if (!t1) t1 = findBest(null, slot, availableOldTeachers(null, slot.day), false);
   770→        if (!t1) t1 = findBestRelaxed(null, slot, availableOldTeachers(null, slot.day));
   771→
   772→        if (t1) {
   773→          const tr = tracking[t1.id];
   774→          tr.totalComm++; tr.dayComm[slot.day]++;
   775→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   776→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   777→          // Track old teacher days
   778→          if (isOldTeacher(t1) && !tr.assignedSlots.some((s, idx) => idx < tr.assignedSlots.length - 1 && s.day === slot.day)) {
   779→            oldTeacherDays[t1.id] = (oldTeacherDays[t1.id] || 0) + 1;
   780→          }
   781→        } else { standbyCount++; }
   782→
   783→        // T2: pairing rule — classified teacher (has notes) must pair with unclassified
   784→        const blocked = t1?.id || null;
   785→        const t1Classified = t1 ? !!(t1.notes && t1.notes.trim() !== '') : false;
   786→        const t2Filtered = t1Classified ? dayPool.filter(t => t.id !== blocked && (!t.notes || t.notes.trim() === '')) : dayPool;
   787→
   788→        t2 = findBest(blocked, slot, t2Filtered, false);
   789→        if (!t2) t2 = findBest(blocked, slot, t2Filtered, true);
   790→        if (!t2) t2 = findBestRelaxed(blocked, slot, t2Filtered);
   791→        if (!t2) t2 = findBestForceDay(blocked, slot, t2Filtered);
   792→        // Relax pairing rule if no unclassified teacher available
   793→        if (!t2 && t1Classified) {
   794→          t2 = findBest(blocked, slot, dayPool, false);
   795→          if (!t2) t2 = findBest(blocked, slot, dayPool, true);
   796→          if (!t2) t2 = findBestRelaxed(blocked, slot, dayPool);
   797→          if (!t2) t2 = findBestForceDay(blocked, slot, dayPool);
   798→        }
   799→        // Last resort: try old teachers for T2
   800→        if (!t2) {
   801→          const oldPool = availableOldTeachers(blocked, slot.day);
   802→          t2 = findBest(blocked, slot, oldPool, false);
   803→          if (!t2) t2 = findBestRelaxed(blocked, slot, oldPool);
   804→        }
   805→
   806→        if (t2) {
   807→          const tr = tracking[t2.id];
   808→          tr.totalComm++; tr.dayComm[slot.day]++;
   809→          tr.assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   810→          tr.gradeHistory.push({ dayIndex: slot.dayIndex, grade: slot.grade });
   811→          // Track old teacher days
   812→          if (isOldTeacher(t2) && !tr.assignedSlots.some((s, idx) => idx < tr.assignedSlots.length - 1 && s.day === slot.day)) {
   813→            oldTeacherDays[t2.id] = (oldTeacherDays[t2.id] || 0) + 1;
   814→          }
   815→        } else { standbyCount++; }
   816→
   817→        // Record assignment
   818→        const session = finalAssignments[slot.day].find(s => s.grade === slot.grade && s.time === slot.time);
   819→        if (session) {
   820→          session.committees.push({
   821→            serial: slot.comId,
   822→            t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: '—' },
   823→            t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: '—' }
   824→          });
   825→        }
   826→      }
   827→    }
   828→
   829→    // ---- Post-distribution: Recalculate hours from ACTUAL assignments (source of truth) ----
   830→    // This eliminates any tracking accumulation bugs
   831→    const teacherCommCount: Record<string, number> = {};
   832→    const teacherHoursFromAssign: Record<string, number> = {};
   833→    const teacherDayCommFromAssign: Record<string, Record<string, number>> = {};
   834→    const teacherSlotsFromAssign: Record<string, {day:string;start:number;end:number}[]> = {};
   835→    teachers.forEach(t => {
   836→      teacherCommCount[t.id] = 0;
   837→      teacherHoursFromAssign[t.id] = 0;
   838→      teacherDayCommFromAssign[t.id] = {};
   839→      teacherSlotsFromAssign[t.id] = [];
   840→    });
   841→
   842→    for (const day of DAYS) {
   843→      for (const sess of finalAssignments[day]) {
   844→        const sessTimeInfo = parseTimeRange(sess.time || '9:00-10:30');
   845→        for (const c of sess.committees) {
   846→          [c.t1, c.t2].forEach(who => {
   847→            if (!who.id) return;
   848→            teacherCommCount[who.id] = (teacherCommCount[who.id] || 0) + 1;
   849→            // Deduplicate: same day + same time = counted once
   850→            const slotKey = day + '_' + sessTimeInfo.start + '_' + sessTimeInfo.end;
   851→            const existing = teacherSlotsFromAssign[who.id];
   852→            if (!existing.some(s => s.day === day && s.start === sessTimeInfo.start && s.end === sessTimeInfo.end)) {
   853→              existing.push({ day, start: sessTimeInfo.start, end: sessTimeInfo.end });
   854→              teacherHoursFromAssign[who.id] += sessTimeInfo.duration;
   855→            }
   856→            // Day committee count
   857→            if (!teacherDayCommFromAssign[who.id]) teacherDayCommFromAssign[who.id] = {};
   858→            teacherDayCommFromAssign[who.id][day] = (teacherDayCommFromAssign[who.id][day] || 0) + 1;
   859→          });
   860→        }
   861→      }
   862→    }
   863→
   864→    // Override tracking with verified data from assignments
   865→    teachers.forEach(t => {
   866→      if (tracking[t.id]) {
   867→        tracking[t.id].totalComm = teacherCommCount[t.id] || 0;
   868→        tracking[t.id].totalHours = teacherHoursFromAssign[t.id] || 0;
   869→        tracking[t.id].dayComm = teacherDayCommFromAssign[t.id] || ({} as Record<string,number>);
   870→        tracking[t.id].assignedSlots = teacherSlotsFromAssign[t.id] || [];
   871→      }
   872→    });
   873→
   874→    // ---- Post-distribution: Admin already handled by pool exclusion ----
   875→
   876→    // ---- Post-distribution: BALANCE PASS ----
   877→    // Iteratively swap assignments from overburdened → underburdened teachers
   878→    // to minimize the max-min spread in BOTH hours (primary) and committees (secondary)
   879→    const BALANCE_ITERATIONS = 500;
   880→    const recalcHours = () => {
   881→      teachers.forEach(t => {
   882→        const slots = tracking[t.id].assignedSlots || [];
   883→        const seen = new Set<string>();
   884→        let hrs = 0;
   885→        slots.forEach(s => {
   886→          const key = s.day + '_' + s.start + '_' + s.end;
   887→          if (!seen.has(key)) { seen.add(key); hrs += (s.end - s.start) / 60; }
   888→        });
   889→        tracking[t.id].totalHours = hrs;
   890→      });
   891→    };
   892→    recalcHours(); // recalc after admin swaps first
   893→
   894→    for (let iter = 0; iter < BALANCE_ITERATIONS; iter++) {
   895→      const active = teachers.filter(t => tracking[t.id].totalComm > 0);
   896→      if (active.length < 2) break;
   897→      // Sort by HOURS first (primary), then committees (secondary)
   898→      const sorted = [...active].sort((a, b) => {
   899→        const hDiff = tracking[b.id].totalHours - tracking[a.id].totalHours;
   900→        if (Math.abs(hDiff) > 0.1) return hDiff;
   901→        return tracking[b.id].totalComm - tracking[a.id].totalComm;
   902→      });
   903→      const maxH = tracking[sorted[0].id].totalHours;
   904→      const minH = tracking[sorted[sorted.length - 1].id].totalHours;
   905→      if (maxH - minH <= 0.5) break; // spread ≤ 0.5h — tight balance
   906→
   907→      let swapped = false;
   908→      // Try top-5 donors × bottom-5 recipients for wider swap search
   909→      for (let di = 0; di < Math.min(sorted.length, 5) && !swapped; di++) {
   910→        const donor = sorted[di];
   911→        // Don't strip admins below 1 assignment in balance pass
   912→        // Never swap OUT old teachers (they are last-resort only)
   913→        if (tracking[donor.id].totalComm <= 1) continue;
   914→        if (isOldTeacher(donor)) continue;
   915→
   916→        for (let ri = sorted.length - 1; ri >= Math.max(0, sorted.length - 5) && !swapped; ri--) {
   917→          const recip = sorted[ri];
   918→          if (donor.id === recip.id) continue;
   919→          // NEVER give admin more assignments during balance pass
   920→          // Old teachers CAN be recipients (swapped IN) but check their day limit
   921→          if (recip.subject === 'Admin') continue;
   922→          if (isOldTeacher(recip) && (oldTeacherDays[recip.id] || 0) >= 3) continue;
   923→          if (tracking[donor.id].totalHours <= tracking[recip.id].totalHours + 0.3) continue;
   924→            // Also check if committees would be more balanced after swap
   925→            const donorCommAfter = tracking[donor.id].totalComm - 1;
   926→            const recipCommAfter = tracking[recip.id].totalComm + 1;
   927→            // Skip if swap would make committee imbalance worse (donor already has fewer comms)
   928→
   929→          // Try each day where donor has an assignment and recipient doesn't
   930→          for (const day of DAYS) {
   931→            if (swapped) break;
   932→            if ((tracking[donor.id].dayComm[day] || 0) === 0) continue;
   933→            if ((tracking[recip.id].dayComm[day] || 0) >= 1) continue;
   934→
   935→            const donorSlot = tracking[donor.id].assignedSlots.find(s => s.day === day);
   936→            if (!donorSlot) continue;
   937→
   938→            // Find the session matching donor's slot
   939→            for (const sess of finalAssignments[day]) {
   940→              if (swapped) break;
   941→              const sessTI = parseTimeRange(sess.time || '9:00-10:30');
   942→              if (donorSlot.start !== sessTI.start || donorSlot.end !== sessTI.end) continue;
   943→
   944→              // Check recipient constraints
   945→              if (ruleSubject && sess.subject && recip.subject === sess.subject) continue;
   946→              if (isSubjectExcluded(recip, { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   947→              if (!canSuperviseStage(recip, getStage(sess.grade))) continue;
   948→              if (tracking[recip.id].assignedSlots.some(s => s.day === day && !(sessTI.end <= s.start || sessTI.start >= s.end))) continue;
   949→              if (wasSameGradeAdjacent(tracking[recip.id], { day, dayIndex: DAYS.indexOf(day), grade: sess.grade, stage: getStage(sess.grade), subject: sess.subject, time: sess.time, timeInfo: sessTI, comId: 0 })) continue;
   950→
   951→              // Find donor in this session's committees
   952→              for (const c of sess.committees) {
   953→                const isT1 = c.t1?.id === donor.id;
   954→                const isT2 = c.t2?.id === donor.id;
   955→                if (!isT1 && !isT2) continue;
   956→
   957→                // Execute swap
   958→                if (isT1) c.t1 = { id: recip.id, name: recip.name };
   959→                else c.t2 = { id: recip.id, name: recip.name };
   960→
   961→                tracking[donor.id].totalComm--;
   962→                tracking[donor.id].dayComm[day] = Math.max(0, (tracking[donor.id].dayComm[day] || 0) - 1);
   963→                tracking[donor.id].assignedSlots = tracking[donor.id].assignedSlots.filter(s => !(s.day === day && s.start === donorSlot.start && s.end === donorSlot.end));
   964→                tracking[donor.id].gradeHistory = tracking[donor.id].gradeHistory.filter(h => !(h.dayIndex === DAYS.indexOf(day) && h.grade === sess.grade));
   965→
   966→                tracking[recip.id].totalComm++;
   967→                tracking[recip.id].dayComm[day] = (tracking[recip.id].dayComm[day] || 0) + 1;
   968→                tracking[recip.id].assignedSlots.push({ day, start: sessTI.start, end: sessTI.end });
   969→                tracking[recip.id].gradeHistory.push({ dayIndex: DAYS.indexOf(day), grade: sess.grade });
   970→
   971→                // Track old teacher day if recipient is old
   972→                if (isOldTeacher(recip)) {
   973→                  oldTeacherDays[recip.id] = (oldTeacherDays[recip.id] || 0) + 1;
   974→                }
   975→
   976→                swapped = true;
   977→                break;
   978→              }
   979→            }
   980→          }
   981→        }
   982→      }
   983→      if (!swapped) break; // no valid swap found
   984→      recalcHours(); // update hours after each swap
   985→    }
   986→
   987→    // Final hour recalculation after all balancing
   988→    recalcHours();
   989→
   990→    // ---- Sort committees by serial within each session (1, 2, 3...) ----
   991→    for (const day of DAYS) {
   992→      for (const sess of finalAssignments[day]) {
   993→        sess.committees.sort((a, b) => a.serial - b.serial);
   994→      }
   995→    }

   998→    // 1 standby teacher per stage per day from unassigned teachers
   999→    // CRITICAL: Standby must NOT be the same subject as any exam that day for that stage
  1000→    // Rotation: prefer teachers who weren't standby yesterday, fewest total standby days
  1001→    const STAGES_LIST = ['primary', 'prep', 'sec'] as const;
  1002→    const STANDBY_PER_STAGE = 1;
  1003→    const standbys: Record<string, Record<string, StandbyEntry[]>> = {};
  1004→    DAYS.forEach(d => { standbys[d] = {}; STAGES_LIST.forEach(s => { standbys[d][s] = []; }); });
  1005→
  1006→    // Track standby count per teacher for rotation fairness
  1007→    const teacherStandbyCount: Record<string, number> = {};
  1008→    teachers.forEach(t => { teacherStandbyCount[t.id] = 0; });
  1009→
  1010→    // Track which teachers are already standby for another stage today
  1011→    const todayStandby: Record<string, Set<string>> = {};
  1012→    DAYS.forEach(d => { todayStandby[d] = new Set(); });
  1013→
  1014→    for (let di = 0; di < DAYS.length; di++) {
  1015→      const day = DAYS[di];
  1016→
  1017→      // Determine pool for standby (same logic as distribution)
  1018→      const isW2Peak = WEEK2_DAYS.includes(day) && peakW2Days.has(day);
  1019→      const dayStandbyPool = (includeAdminW2 && isW2Peak) ? [...teachers] : nonAdminTeachers;
  1020→

  1022→        // Only assign standby if there are exams for this stage on this day
  1023→        const stageSessions = (finalAssignments[day] || []).filter(s => getStage(s.grade) === stage);
  1024→        if (stageSessions.length === 0) continue;
  1025→
  1026→        // Collect subjects being examined today for this stage
  1027→        const dayStageSubjects = new Set<string>();
  1028→        stageSessions.forEach(s => { if (s.subject) dayStageSubjects.add(s.subject); });
  1029→
  1030→        // Find teachers NOT assigned on this day who can supervise this stage
  1031→        // AND whose subject doesn't match any exam subject for this stage today
  1032→        let candidates = dayStandbyPool.filter(t => {
  1033→          if ((tracking[t.id].dayComm[day] || 0) >= 2) return false;
  1034→          if (todayStandby[day].has(t.id)) return false; // already standby for another stage today
  1035→          if (!canSuperviseStage(t, stage)) return false;
  1036→          // CRITICAL: Don't pick a teacher whose subject is being examined today
  1037→          if (ruleSubject && t.subject && dayStageSubjects.has(t.subject)) return false;
  1038→          // CRITICAL: Special subject exclusion (Religion/Arabic) for standby
  1039→          if (ruleSubject && dayStageSubjects.has('الدين') && (t.subject === 'الدين' || t.subject === 'العربي')) return false;
  1040→          if (ruleSubject && dayStageSubjects.has('العربي') && t.subject === 'الدين' && stage === 'primary') return false;
  1041→          return true;
  1042→        });
  1043→
  1044→        // Rotation: deprioritize teachers who were standby YESTERDAY for ANY stage
  1045→        const yesterdayStandby = new Set<string>();
  1046→        if (di > 0) {
  1047→          const prevDay = DAYS[di - 1];
  1048→          for (const st of STAGES_LIST) {
  1049→            (standbys[prevDay][st] || []).forEach(s => yesterdayStandby.add(s.id));
  1050→          }
  1051→        }
  1052→        candidates.sort((a, b) => {
  1053→          const aWasYesterday = yesterdayStandby.has(a.id) ? 100 : 0;
  1054→          const bWasYesterday = yesterdayStandby.has(b.id) ? 100 : 0;
  1055→          return (teacherStandbyCount[a.id] + aWasYesterday) - (teacherStandbyCount[b.id] + bWasYesterday);
  1056→        });
  1057→
  1058→        // Pick up to STANDBY_PER_STAGE
  1059→        const picked = candidates.slice(0, STANDBY_PER_STAGE);
  1060→        standbys[day][stage] = picked.map(t => ({ id: t.id, name: t.name }));
  1061→
  1062→        // Track rotation count and mark as standby for today (so same teacher isn't picked for another stage)
  1063→        for (const s of picked) {
  1064→          teacherStandbyCount[s.id]++;
  1065→          todayStandby[day].add(s.id);
  1066→        }
  1067→      }
  1068→    }
  1069→
  1070→    // ---- Post-distribution verification ----
  1071→    const violations: string[] = [];

  1073→    if (ruleSubject) {
  1074→      for (const day of DAYS) {

  1079→                const tch = teachers.find(x => x.id === who.id);

  1095→          }
  1096→        }
  1097→      }

  1100→    for (const t of teachers) {
  1101→      const tr = tracking[t.id];
  1102→      for (const day of DAYS) {
  1103→        if ((tr.dayComm[day] || 0) > 2) {
  1104→          violations.push(`${t.name} -> 3+ committees on ${day} (HARD RULE VIOLATED)`);

  1122→      }
  1123→    }

  1126→    for (const day of DAYS) {
  1127→      for (const sess of finalAssignments[day]) {

  1133→            const t2Has = t2Info && t2Info.notes && t2Info.notes.trim() !== '';
  1134→            if (t1Has && t2Has) {

  1136→            }
  1137→          }
  1138→        }
  1139→      }
  1140→    }
  1141→

  1143→    for (const day of DAYS) {
  1144→      for (const sess of finalAssignments[day]) {
  1145→        const sessStage = getStage(sess.grade);

  1151→            if (sess.subject === 'الدين' && (tch.subject === 'العربي' || tch.subject === 'الدين')) {

  1163→    for (const ot of oldTeachers) {
  1164→      const days = oldTeacherDays[ot.id] || 0;
  1165→      if (days > 3) {

  1190→    setActivePage('results');
