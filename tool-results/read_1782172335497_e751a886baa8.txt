   806→   806→function parseTimeRange(tStr) {
   807→   807→  if (!tStr || !tStr.includes('-')) return { start: 540, end: 630, duration: 1.5 };
   808→   808→  try {
   809→   809→    let clean = tStr.replace(/\s+/g, '').replace(/—/g, '-').replace(/–/g, '-');
   810→   810→    const parts = clean.split('-');
   811→   811→    const toMinutes = (timeString) => {
   812→   812→      const p = timeString.split(':');
   813→   813→      let hrs = parseInt(p[0]) || 0;
   814→   814→      let mins = parseInt(p[1]) || 0;
   815→   815→      return (hrs * 60) + mins;
   816→   816→    };
   817→   817→    let start = toMinutes(parts[0]);
   818→   818→    let end = toMinutes(parts[1]);
   819→   819→    if (end <= start) end += 12 * 60;
   820→   820→    return { start, end, duration: (end - start) / 60 };
   821→   821→  } catch(e) {
   822→   822→    return { start: 540, end: 630, duration: 1.5 };
   823→   823→  }
   824→   824→}

   859→   859→        else if(lowerGrade.includes("secondary") || lowerGrade.includes("sec")) stage = "sec";
   860→   860→
   861→   861→        let timeInfo = parseTimeRange(cell.time);
   862→   862→        for (let c = 1; c <= cell.committees; c++) {
   863→   863→          allSlots.push({ day, grade, stage, subject: cell.subject, time: cell.time, timeInfo, comId: c });
   864→   864→        }
   865→   865→      }
   866→   866→    });
   867→   867→  });
   868→   868→
   869→   869→  // Heaviest duration loads calculated first
   870→   870→  allSlots.sort((a, b) => b.timeInfo.duration - a.timeInfo.duration);
   871→   871→
   872→   872→  let finalAssignments = {};
   873→   873→  DAYS.forEach(d => { finalAssignments[d] = []; });
   874→   874→  DAYS.forEach(day => {
   875→   875→    GRADES.forEach(grade => {
   876→   876→      const cell = state.schedule[grade]?.[day];
   877→   877→      if (cell && cell.committees > 0) {
   878→   878→        let existing = finalAssignments[day].find(s => s.grade === grade && s.time === cell.time);
   879→   879→        if (!existing) {
   880→   880→          finalAssignments[day].push({ grade, time: cell.time, subject: cell.subject, committees: [] });
   881→   881→        }
   882→   882→      }
   883→   883→    });
   884→   884→  });
   885→   885→
   886→   886→  allSlots.forEach(slot => {
   887→   887→    
   888→   888→    let queryTeacher = (blockedId) => {
   889→   889→      let pool = [];
   890→   890→
   891→   891→      state.teachers.forEach(t => {
   892→   892→        if (blockedId && t.id === blockedId) return; 
   893→   893→        
   894→   894→        let tr = tracking[t.id];
   895→   895→
   896→   896→        // 1. Subject Block
   897→   897→        if (ruleSubject && slot.subject && t.subject === slot.subject) return;
   898→   898→
   899→   899→        // 2. STRICT time overlap - cannot be in 2 places at same time
   900→   900→        let hasOverlap = tr.assignedSlots.some(s => {
   901→   901→          return s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end);
   902→   902→        });
   903→   903→        if (hasOverlap) return;
   904→   904→
   905→   905→        // 3. Day limit rule
   906→   906→        if (ruleDayLimit && tr.dayComm[slot.day] >= 1) return;
   907→   907→
   908→   908→        // 4. Max 5h per day - count unique hours today
   909→   909→        let todayHours = 0;
   910→   910→        let seenToday = new Set();

   912→   912→          let k = s.start + '_' + s.end;
   913→   913→          if (!seenToday.has(k)) { seenToday.add(k); todayHours += (s.end - s.start) / 60; }
   914→   914→        });
   915→   915→        if (todayHours + slot.timeInfo.duration > 5) return;
   916→   916→
   917→   917→        let matchScore = 0;
   918→   918→        if (ruleNotes && t.notes && slot.stage !== "any" && t.notes.toLowerCase().includes(slot.stage)) {
   919→   919→          matchScore = 1;
   920→   920→        }
   921→   921→
   922→   922→        pool.push({ teacher: t, matchScore, totalComm: tr.totalComm, totalHours: tr.totalHours });
   923→   923→      });
   924→   924→
   925→   925→      // Emergency Fallback: relax day limit only, KEEP overlap + max hours
   926→   926→      if (pool.length === 0) {
   927→   927→        state.teachers.forEach(t => {
   928→   928→          if (blockedId && t.id === blockedId) return;
   929→   929→          let tr = tracking[t.id];
   930→   930→          if (ruleSubject && slot.subject && t.subject === slot.subject) return;
   931→   931→          let hasOverlap = tr.assignedSlots.some(s => {
   932→   932→            return s.day === slot.day && !(slot.timeInfo.end <= s.start || slot.timeInfo.start >= s.end);
   933→   933→          });
   934→   934→          if (hasOverlap) return;
   935→   935→          let todayH = 0; let seenT = new Set();
   936→   936→          tr.assignedSlots.filter(s => s.day === slot.day).forEach(s => {
   937→   937→            let k = s.start+'_'+s.end; if(!seenT.has(k)){seenT.add(k); todayH += (s.end-s.start)/60;}
   938→   938→          });
   939→   939→          if (todayH + slot.timeInfo.duration > 5) return;
   940→   940→          pool.push({ teacher: t, matchScore: 0, totalComm: tr.totalComm, totalHours: tr.totalHours });
   941→   941→        });
   942→   942→      }
   943→   943→
   944→   944→      if (pool.length === 0) return null;
   945→   945→
   946→   946→      pool.sort((a, b) => {
   947→   947→        if (a.totalComm !== b.totalComm) return a.totalComm - b.totalComm;
   948→   948→        if (Math.abs(a.totalHours - b.totalHours) > 0.001) return a.totalHours - b.totalHours;
   949→   949→        if (b.matchScore !== a.matchScore) return b.matchScore - a.matchScore;
   950→   950→        return a.teacher.id - b.teacher.id;
   951→   951→      });
   952→   952→
   953→   953→      return pool[0].teacher;
   954→   954→    };
   955→   955→
   956→   956→    let t1 = queryTeacher(null);
   957→   957→    if (t1) {
   958→   958→      tracking[t1.id].totalComm++;
   959→   959→      tracking[t1.id].dayComm[slot.day]++;
   960→   960→      tracking[t1.id].totalHours += slot.timeInfo.duration;
   961→   961→      tracking[t1.id].assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   962→   962→    }
   963→   963→
   964→   964→    let t2 = queryTeacher(t1 ? t1.id : null);
   965→   965→    if (t2) {
   966→   966→      tracking[t2.id].totalComm++;
   967→   967→      tracking[t2.id].dayComm[slot.day]++;
   968→   968→      tracking[t2.id].totalHours += slot.timeInfo.duration;
   969→   969→      tracking[t2.id].assignedSlots.push({ day: slot.day, start: slot.timeInfo.start, end: slot.timeInfo.end });
   970→   970→    }
   971→   971→
   972→   972→    let dayGroup = finalAssignments[slot.day];
   973→   973→    let sessionGroup = dayGroup.find(s => s.grade === slot.grade && s.time === slot.time);
   974→   974→    if (sessionGroup) {
   975→   975→      sessionGroup.committees.push({
   976→   976→        serial: slot.comId,
   977→   977→        t1: t1 ? { id: t1.id, name: t1.name } : { id: null, name: 'Standby Monitor' },
   978→   978→        t2: t2 ? { id: t2.id, name: t2.name } : { id: null, name: 'Standby Monitor' }
   979→   979→      });
   980→   980→    }
   981→   981→  });
   982→   982→
   983→   983→  state.results = { assignments: finalAssignments, tracking };
   984→   984→  saveState();
   985→   985→  toast('Overlap Lock Secured! Schedules fully cleared. ✓', 'success');
   986→   986→  showPage('results');
   987→   987→}
